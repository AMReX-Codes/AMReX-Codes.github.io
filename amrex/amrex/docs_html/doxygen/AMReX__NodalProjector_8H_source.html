13 // *************************** DEFAULT MODE ***************************
14 //
15 // Solves
16 //
17 // div(sigma*grad(phi)) = div(vel) + S_nd + S_cc
18 //
19 // and performs the projection
20 //
21 // vel = vel - sigma * grad(phi)
22 //
23 // vel, sigma, and S_cc are cell-centered variables, while
24 // phi and S_nd are nodal-centered variables.
25 //
26 // *************************** CUSTOM MODE ***************************
27 //
28 // Solves
29 //
30 // div(sigma*grad(phi)) = rhs
31 //
32 // and performs the projection
33 //
34 // vel = vel - (sigma/alpha) * grad(phi)
35 //
36 // alpha is a cell-centered variable, while rhs is nodal-centered.
37 //
38 // In this mode, the user provides rhs and alpha
39 //
40 // By default alpha is assumed to be 1. Use setAlpha to change the default.
41 //
42 // Use setCustomRHS to provide a custom RHS, else "div(vel) + S_nd + S_cc"
43 // is used.
44 //
45 // Example: rhs = div(alpha*vel)
46 //
