219 template <class T, class H>
220 std::size_t removeDupDoit (Vector<T>& vec, std::size_t start, std::size_t stop)
221 {
222 std::size_t N = stop-start;
223 if (N < 2) { return stop; }
224
225 T* const data = vec.data() + start;
226 T const sentinel = data[0]; // duplicates will be set to sentinel and removed later
227 H const hasher;
228 for (std::size_t i = 1; i < N; ) {
229 if (data[i] == sentinel) {
230 ++i;
231 continue;
232 }
233
234 std::size_t const hash = hasher(data[i]) % N;
235 if (i == hash) { // data[i] in correct hash position
236 ++i;
237 continue;
238 }
239
240 if (data[i] == data[hash]) {
241 data[i] = sentinel; // because it's a duplicate
242 ++i;
243 continue;
244 }
245
246 if (data[hash] == sentinel) {
247 std::swap(data[hash], data[i]);
248 // after swap, new data[i] holds sentinel
249 // newdata[hash] in correct hash poitiion
250 ++i;
251 continue;
252 }
253
254 std::size_t const hashhash = hasher(data[hash]) % N;
255 if (hashhash != hash) { // data[hash] not in correct has poision, thus will yield it's position
256 std::swap(data[i], data[hash]);
257 // after swap, new data[hash] in correct hash position
258 // new data[i] not sure
259 if (hash < i) { // we have seen new data[i]
260 ++i;
261 } // else next iteration we will work on data[i]
262 } else { // data[hash] in correct hash position, but data[i] is not because of hash collision
263 ++i;
264 }
265 }
266
267 // Now there are three types for data[i]
268 // (1) sentinel
269 // (2) data[i] != sentinel and hash(data[i]) == i
270 // (3) data[i] != sentinel and hash(data[i]) != i because of hash collision
271 // All type 2s are unique, and all sentinels except one can be removed.
272 // We will move all type 2s to the beginning, all sentinels to the end.
273 // This will leave all type 3s in the middle. Then we will work on the middle
274 // part plus one sentinel.
275
276 std::size_t swapPos = 0;
277 for (std::size_t i = 0; i < N; ++i) {
278 // move type 2 to the beginning pointed to by swapPos
279 if (data[i] != sentinel && i == hasher(data[i]) % N) {
280 std::swap(data[i], data[swapPos++]);
281 }
282 }
283
284 // Now we have moved all type 2 elements to the beginning, [0,swapPos)
285
286 std::size_t sentinelPos = N;
287 for (std::size_t i = swapPos; i < sentinelPos; ) {
288 // move type 1 to the end
289 if(data[i] == sentinel) {
290 std::swap(data[i], data[--sentinelPos]);
291 } else {
292 ++i;
293 }
294 }
295
296 // recursively work on the middle part
297 return detail::removeDupDoit<T,H>(vec, start+swapPos, start+sentinelPos+1);
298 }
299 }
