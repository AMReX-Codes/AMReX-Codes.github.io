64  } else if (fb == 0.0) {

66  }

69  int i;
70  for (i = 0; i < MAXITER; ++i)

72  if (fb*fc > 0)

74  // Rename a, b, c and adjust bounding interval d
75  c = aPt[rangedir];
76  fc = fa;
77  d = bPt[rangedir] - aPt[rangedir];
78  e = d;
79  }
80 
81  if (std::abs(fc) < std::abs(fb))
82  {
83  aPt[rangedir] = bPt[rangedir];
84  bPt[rangedir] = c;
85  c = aPt[rangedir];
86  fa = fb;
87  fb = fc;
88  fc = fa;
89  }
90 
91  // Convergence check
92  Real tol1 = 2.0 * EPS * std::abs(bPt[rangedir]) + 0.5 * tol;
93  Real xm = 0.5 * (c - bPt[rangedir]);
94 
95  if (std::abs(xm) <= tol1 || fb == 0.0)
96  {
97  break;
98  }
99 
100  if (std::abs(e) >= tol1 && std::abs(fa) > std::abs(fb))
101  {
102  // Attempt inverse quadratic interpolation
103  s = fb / fa;
104  if (aPt[rangedir] == c)
105  {
106  p = 2.0 * xm * s;
107  q = 1.0 - s;
108  }
109  else
110  {
111  q = fa / fc;
112  r = fb / fc;
113  p = s * (2.0 * xm * q * (q-r) - (bPt[rangedir]-aPt[rangedir]) * (r-1.0));
114  q = (q-1.0) * (r-1.0) * (s-1.0);
115  }
116 
117  // Check whether in bounds
118  if (p > 0) q = -q;
119 
120  p = std::abs(p);
121 
122  if (2.0 * p < amrex::min(3.0*xm*q-std::abs(tol1*q), 1.0*std::abs(e*q)))
123  {
124  // Accept interpolation
125  e = d;
126  d = p / q;
127  }
128  else
129  {
130  // Interpolation failed, use bisection
131  d = xm;
132  e = d;
133  }
134  }
135  else
136  {
137  // Bounds decreasing too slowly, use bisection
138  d = xm;
139  e = d;
140  }
141 
142  // Move last best guess to a
143  aPt[rangedir] = bPt[rangedir];
144  fa = fb;
145 
146  // Evaluate new trial root
147  if (std::abs(d) > tol1)
148  {
149  bPt[rangedir] = bPt[rangedir] + d;
150  }
151  else

155  }

158  }
