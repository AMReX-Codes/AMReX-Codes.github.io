41 BrentRootFinder (GpuArray<Real,AMREX_SPACEDIM> const& lo,
42  GpuArray<Real,AMREX_SPACEDIM> const& hi,
43  int rangedir, F const& f) noexcept

46  const int MAXITER = 100;
47  const Real EPS = 3.0e-15;
48 
49  Real p, q, r, s;
50 
51  GpuArray<Real,AMREX_SPACEDIM> aPt = lo;
52  GpuArray<Real,AMREX_SPACEDIM> bPt = hi;
53 
54  Real fa = IF_f(f, aPt);
55  Real fb = IF_f(f, bPt);
56  Real c = bPt[rangedir];
57  Real fc = fb;
58 
59  if (fb*fa > 0.0) {
60 // amrex::AllPrint() << "fa " << fa << " fb " << fb << "\n";
61  amrex::Error("BrentRootFinder. Root must be bracketed, but instead the supplied end points have the same sign.");
62  return 0.0;
63  } else if (fa == 0.0) {
64  return aPt[rangedir];
65  } else if (fb == 0.0) {
66  return bPt[rangedir];
67  }
68 
69  Real d = 0.0, e = 0.0;
70  int i;
71  for (i = 0; i < MAXITER; ++i)
72  {
73  if (fb*fc > 0)
74  {
75  // Rename a, b, c and adjust bounding interval d
76  c = aPt[rangedir];
77  fc = fa;
78  d = bPt[rangedir] - aPt[rangedir];
79  e = d;
80  }
81 
82  if (amrex::Math::abs(fc) < amrex::Math::abs(fb))
83  {
84  aPt[rangedir] = bPt[rangedir];
85  bPt[rangedir] = c;
86  c = aPt[rangedir];
87  fa = fb;
88  fb = fc;
89  fc = fa;
90  }
91 
92  // Convergence check
93  Real tol1 = 2.0 * EPS * amrex::Math::abs(bPt[rangedir]) + 0.5 * tol;
94  Real xm = 0.5 * (c - bPt[rangedir]);
95 
96  if (amrex::Math::abs(xm) <= tol1 || fb == 0.0)
97  {
98  break;
99  }
100 
101  if (amrex::Math::abs(e) >= tol1 && amrex::Math::abs(fa) > amrex::Math::abs(fb))
102  {
103  // Attempt inverse quadratic interpolation
104  s = fb / fa;
105  if (aPt[rangedir] == c)
106  {
107  p = 2.0 * xm * s;
108  q = 1.0 - s;
109  }
110  else
111  {
112  q = fa / fc;
113  r = fb / fc;
114  p = s * (2.0 * xm * q * (q-r) - (bPt[rangedir]-aPt[rangedir]) * (r-1.0));
115  q = (q-1.0) * (r-1.0) * (s-1.0);
116  }
117 
118  // Check whether in bounds
119  if (p > 0) q = -q;
120 
121  p = amrex::Math::abs(p);
122 
123  if (2.0 * p < amrex::min(3.0*xm*q-amrex::Math::abs(tol1*q), 1.0*amrex::Math::abs(e*q)))
124  {
125  // Accept interpolation
126  e = d;
127  d = p / q;
128  }
129  else
130  {
131  // Interpolation failed, use bisection
132  d = xm;
133  e = d;
134  }
135  }
136  else
137  {
138  // Bounds decreasing too slowly, use bisection
139  d = xm;
140  e = d;
141  }
142 
143  // Move last best guess to a
144  aPt[rangedir] = bPt[rangedir];
145  fa = fb;
146 
147  // Evaluate new trial root
148  if (amrex::Math::abs(d) > tol1)
149  {
150  bPt[rangedir] = bPt[rangedir] + d;
151  }
152  else
153  {
154  if (xm < 0) bPt[rangedir] = bPt[rangedir] - tol1;
155  else bPt[rangedir] = bPt[rangedir] + tol1;
156  }
157 
158  fb = IF_f(f, bPt);
159  }
160 
161  if (i >= MAXITER)
162  {
163  amrex::Error("BrentRootFinder: exceeding maximum iterations.");
164  }
165 
166  return bPt[rangedir];
