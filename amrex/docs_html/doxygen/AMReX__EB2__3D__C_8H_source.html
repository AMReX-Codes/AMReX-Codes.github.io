278AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
279Real coarsen_edge_cent (Real f1, Real f2)
280{
281 if (f1 == Real(1.0) && f2 == Real(1.0)) {
282 return Real(1.0);
283 } else if (f1 == Real(-1.0) && f2 == Real(-1.0)) {
284 return Real(-1.0);
285 } else {
286 if (f1 == Real(-1.0)) {
287 f1 = Real(0.0);
288 } else if (f1 == Real(1.0)) {
289 f1 = Real(-0.25);
290 } else {
291 f1 = Real(0.5)*f1 - Real(0.25);
292 }
293 if (f2 == Real(-1.0)) {
294 f2 = Real(0.0);
295 } else if (f2 == Real(1.0)) {
296 f2 = Real(0.25);
297 } else {
298 f2 = Real(0.5)*f2 + Real(0.25);
299 }
300 Real r = (f2*f2-f1*f1)/(f2-f1+Real(1.e-30));
301 return amrex::min(Real(0.5),amrex::max(Real(-0.5),r));
302 }
303}
