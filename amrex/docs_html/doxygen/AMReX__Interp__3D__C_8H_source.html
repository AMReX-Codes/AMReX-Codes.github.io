496 nodebilin_slopes (Box const& bx, Array4<T> const& slope, Array4<T const> const& u,
497  const int icomp, const int ncomp, IntVect const& ratio) noexcept
498 {
499  const auto lo = amrex::lbound(bx);
500  const auto hi = amrex::ubound(bx);
501 
502  const Real rx = Real(1.)/ratio[0];
503  const Real ry = Real(1.)/ratio[1];
504  const Real rz = Real(1.)/ratio[2];
505 
506  for (int n = 0; n < ncomp; ++n) {
507  const int nu = n + icomp;
508  for (int k = lo.z; k <= hi.z; ++k) {
509  for (int j = lo.y; j <= hi.y; ++j) {
510  AMREX_PRAGMA_SIMD
511  for (int i = lo.x; i <= hi.x; ++i) {
512  T dx00 = u(i+1,j,k,nu) - u(i,j,k,nu);
513  T d0x0 = u(i,j+1,k,nu) - u(i,j,k,nu);
514  T d00x = u(i,j,k+1,nu) - u(i,j,k,nu);
515 
516  T dx10 = u(i+1,j+1,k,nu) - u(i,j+1,k,nu);
517  T dx01 = u(i+1,j,k+1,nu) - u(i,j,k+1,nu);
518  T d0x1 = u(i,j+1,k+1,nu) - u(i,j,k+1,nu);
519 
520  T dx11 = u(i+1,j+1,k+1,nu) - u(i,j+1,k+1,nu);
521 
522  slope(i,j,k,n+ncomp*ix ) = rx*dx00;
523  slope(i,j,k,n+ncomp*iy ) = ry*d0x0;
524  slope(i,j,k,n+ncomp*iz ) = rz*d00x;
525  slope(i,j,k,n+ncomp*ixy ) = rx*ry*(dx10 - dx00);
526  slope(i,j,k,n+ncomp*ixz ) = rx*rz*(dx01 - dx00);
527  slope(i,j,k,n+ncomp*iyz ) = ry*rz*(d0x1 - d0x0);
528  slope(i,j,k,n+ncomp*ixyz) = rx*ry*rz*(dx11 - dx01 - dx10 + dx00);
529  }
530  }
531  }
532  }
