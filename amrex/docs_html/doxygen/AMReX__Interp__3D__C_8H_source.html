495 nodebilin_slopes (Box const& bx, Array4<T> const& slope, Array4<T const> const& u,
496  const int icomp, const int ncomp, IntVect const& ratio) noexcept
497 {
498  const auto lo = amrex::lbound(bx);
499  const auto hi = amrex::ubound(bx);
500 
501  const Real rx = 1.0_rt/ratio[0];
502  const Real ry = 1.0_rt/ratio[1];
503  const Real rz = 1.0_rt/ratio[2];
504 
505  for (int n = 0; n < ncomp; ++n) {
506  const int nu = n + icomp;
507  for (int k = lo.z; k <= hi.z; ++k) {
508  for (int j = lo.y; j <= hi.y; ++j) {
509  AMREX_PRAGMA_SIMD
510  for (int i = lo.x; i <= hi.x; ++i) {
511  T dx00 = u(i+1,j,k,nu) - u(i,j,k,nu);
512  T d0x0 = u(i,j+1,k,nu) - u(i,j,k,nu);
513  T d00x = u(i,j,k+1,nu) - u(i,j,k,nu);
514 
515  T dx10 = u(i+1,j+1,k,nu) - u(i,j+1,k,nu);
516  T dx01 = u(i+1,j,k+1,nu) - u(i,j,k+1,nu);
517  T d0x1 = u(i,j+1,k+1,nu) - u(i,j,k+1,nu);
518 
519  T dx11 = u(i+1,j+1,k+1,nu) - u(i,j+1,k+1,nu);
520 
521  slope(i,j,k,n+ncomp*ix ) = rx*dx00;
522  slope(i,j,k,n+ncomp*iy ) = ry*d0x0;
523  slope(i,j,k,n+ncomp*iz ) = rz*d00x;
524  slope(i,j,k,n+ncomp*ixy ) = rx*ry*(dx10 - dx00);
525  slope(i,j,k,n+ncomp*ixz ) = rx*rz*(dx01 - dx00);
526  slope(i,j,k,n+ncomp*iyz ) = ry*rz*(d0x1 - d0x0);
527  slope(i,j,k,n+ncomp*ixyz) = rx*ry*rz*(dx11 - dx01 - dx10 + dx00);
528  }
529  }

535 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE void
536 nodebilin_interp (Box const& bx, Array4<T> const& fine, const int fcomp, const int ncomp,
537  Array4<T const> const& slope, Array4<T const> const& crse,
538  const int ccomp, IntVect const& ratio) noexcept

547  const Real fz = k - kc*ratio[2];
548  for (int j = lo.y; j <= hi.y; ++j) {
549  const int jc = amrex::min(amrex::coarsen(j,ratio[1]),chi.y);
550  const Real fy = j - jc*ratio[1];

552  for (int i = lo.x; i <= hi.x; ++i) {

554  const Real fx = i - ic*ratio[0];

558  + fz*slope(ic,jc,kc,n+ncomp*iz)
559  + fx*fy*slope(ic,jc,kc,n+ncomp*ixy)
560  + fx*fz*slope(ic,jc,kc,n+ncomp*ixz)
561  + fy*fz*slope(ic,jc,kc,n+ncomp*iyz)
