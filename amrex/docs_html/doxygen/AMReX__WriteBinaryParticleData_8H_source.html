156template <typename P, typename I>
157AMREX_GPU_HOST_DEVICE
158void packParticleIDs (I* idata, const P& p, bool is_checkpoint) noexcept
159{
160 if (is_checkpoint) {
161 std::int32_t xi, yi;
162 std::uint32_t xu, yu;
163 xu = (std::uint32_t)((p.m_idcpu & 0xFFFFFFFF00000000LL) >> 32);
164 yu = (std::uint32_t)( p.m_idcpu & 0xFFFFFFFFLL);
165 amrex::Gpu::memcpy(&xi, &xu, sizeof(xu));
166 amrex::Gpu::memcpy(&yi, &yu, sizeof(yu));
167 idata[0] = xi;
168 idata[1] = yi;
169 } else {
170 idata[0] = p.id();
171 idata[1] = p.cpu();
172 }

258 for (int j = real_start_offset; j < PC::SuperParticleType::NReal; j++) {

260 if (write_real_comp_d_ptr[write_comp_index]) {
261 rdata_d_ptr[rout_index] = p.rdata(j);
262 rout_index++;
263 }

266 for (int j = 0; j < ptd.m_num_runtime_real; j++) {
267 if (write_real_comp_d_ptr[PC::SuperParticleType::NReal+j-real_start_offset]) {

276 idata.begin() + typename PC::IntVector::difference_type(poffset));
277 Gpu::copyAsync(Gpu::deviceToHost, rdata_d.begin(), rdata_d.end(),
278 rdata.begin() + typename PC::RealVector::difference_type(poffset));
279 Gpu::Device::streamSynchronize();
280
281 poffset += num_copies;
282 }
