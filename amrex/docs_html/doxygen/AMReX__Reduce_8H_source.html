315  int& nblocks = reduce_data.nBlocks(stream);
316  int ncells = box.numPts();
317  const auto lo = amrex::lbound(box);
318  const auto len = amrex::length(box);
319  const auto lenxy = len.x*len.y;
320  const auto lenx = len.x;
321  IndexType ixtype = box.ixType();
322  constexpr int nitems_per_thread = 4;
323  int nblocks_ec = (ncells + nitems_per_thread*AMREX_GPU_MAX_THREADS-1)
324  / (nitems_per_thread*AMREX_GPU_MAX_THREADS);
325  nblocks_ec = std::min(nblocks_ec, static_cast<int>(Gpu::Device::maxBlocksPerLaunch()));
326 #ifdef AMREX_USE_DPCPP

330  [=] AMREX_GPU_DEVICE (Gpu::Handler const& gh) noexcept

332  ReduceTuple r;
333  Reduce::detail::for_each_init<0, ReduceTuple, Ps...>(r);
334  ReduceTuple& dst = *(dp+gh.blockIdx());

336  dst = r;
