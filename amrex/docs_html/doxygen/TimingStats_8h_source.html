71 void printTimingStats(MPI_Comm comm, // comm for MPI_Allreduce()
72  const char *preamble, // text at beginning of line
73  double dt) // delta t in seconds
74 {
75  int myrank, nranks;
76  double max, min, sum, avg, var, stdev;
77 
78  MPI_Comm_rank(comm, &myrank);
79  MPI_Comm_size(comm, &nranks);
80 
81  MPI_Allreduce(&dt, &max, 1, MPI_DOUBLE, MPI_MAX, comm);
82  MPI_Allreduce(&dt, &min, 1, MPI_DOUBLE, MPI_MIN, comm);
83  MPI_Allreduce(&dt, &sum, 1, MPI_DOUBLE, MPI_SUM, comm);
84  avg = sum/nranks;
85 
86  dt -= avg;
87  dt *= dt;
88  MPI_Allreduce(&dt, &var, 1, MPI_DOUBLE, MPI_SUM, comm);
89  var *= 1.0/nranks;
90  stdev = sqrt(var);
91 
92  if(myrank==0) {
93  printf("%s max %.3es avg %.3es min %.3es dev %.3es\n",
94  preamble, max, avg, min, stdev);
95  }
96 
97  MPI_Barrier(comm);
98 
99  return;
100 }
