479  }
480  else if (apy(i,j,k) == 1 && cvol(i,j,k) == 1 && cvol(i,j-1,k) == 1)
481  {
482  edg_y(i,j,k,n) = 0.5 * (phi(i,j ,k,n) + phi(i,j-1,k,n));
483  }
484  else
485  {
486  int jj = j - 1;
487 
488  // We must add an additional test to avoid the case where fcy is very close to zero, but (i-1) or (i+1)
489  // might be covered cells -- this can happen when the EB is exactly aligned with the horizontal grid lines
490  int ii;
491  if (amrex::Math::abs(fcy(i,j,k)) > 1.e-8)
492  ii = (fcy(i,j,k) < 0.0) ? i - 1 : i + 1;
493  else if (apy(i-1,j,k) > 0.)
494  ii = i-1;
495  else

498  Real d0 = 0.5 + ccent(i, j,k,1); // distance in y-dir from centroid(i,j ) to face(i,j)
499  Real d1 = 0.5 - ccent(i,jj,k,1); // distance in y-dir from centroid(i,j-1) to face(i,j)
500 
501  Real d0_ii = 0.5 + ccent(ii, j,k,0); // distance from centroid(ii,j ) to face(ii,j)
502  Real d1_ii = 0.5 - ccent(ii,jj,k,0); // distance from centroid(ii,j-1) to face(ii,j)
503 
504  Real a0 = d1 / (d0 + d1); // coefficient of phi(i,j ,k,n)
505  Real a1 = d0 / (d0 + d1); // coefficient of phi(i,j-1,k,n)
506 
507  Real a0_ii = d1_ii / (d0_ii + d1_ii); // coefficient of phi(i ,jj,k,n)
508  Real a1_ii = d0_ii / (d0_ii + d1_ii); // coefficient of phi(i-1,jj,k,n)
509 
510  Real phi01 = a0 * phi( i,j,k,n) + a1 * phi( i,jj,k,n); // interpolation of phi in y-direction
511  Real x01 = a0 * ccent( i,j,k,0) + a1 * ccent( i,jj,k,0); // interpolation of x-loc in y-direction
512 
513  Real phi01_ii = a0_ii * phi(ii,j,k,n) + a1_ii * phi(ii,jj,k,n); // interpolation of phi in y-direction
514  Real x01_ii = a0_ii * ccent(ii,j,k,0) + a1_ii * ccent(ii,jj,k,0); // interpolation of x-loc in y-direction
515 
516  edg_y(i,j,k,n) = ( ( fcy(i,j,k) - x01 ) * phi01_ii +
517  (1. - fcy(i,j,k) + x01_ii) * phi01 ) / (1. - x01 + x01_ii);
518  }
519  });
