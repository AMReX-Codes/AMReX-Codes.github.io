48 template <typename MF, typename BC>
49 EnableIf_t<IsFabArray<MF>::value>
50 FillPatchSingleLevel (MF& mf, IntVect const& nghost, Real time,
51  const Vector<MF*>& smf, const Vector<Real>& stime,
52  int scomp, int dcomp, int ncomp,
53  const Geometry& geom,
54  BC& physbcf, int bcfcomp)
55 {
56  BL_PROFILE("FillPatchSingleLevel");
57 
58  AMREX_ASSERT(scomp+ncomp <= smf[0]->nComp());
59  AMREX_ASSERT(dcomp+ncomp <= mf.nComp());
60  AMREX_ASSERT(smf.size() == stime.size());
61  AMREX_ASSERT(smf.size() != 0);
62  AMREX_ASSERT(nghost.allLE(mf.nGrowVect()));
63 
64  if (smf.size() == 1)
65  {
66  if (&mf == smf[0] && scomp == dcomp) {
67  mf.FillBoundary(dcomp, ncomp, nghost, geom.periodicity());
68  } else {
69  mf.ParallelCopy(*smf[0], scomp, dcomp, ncomp, IntVect{0}, nghost, geom.periodicity());
70  }
71  }
72  else if (smf.size() == 2)
73  {
74  BL_ASSERT(smf[0]->boxArray() == smf[1]->boxArray());
75  MF raii;
76  MF * dmf;
77  int destcomp;
78  bool sameba;
79  if (mf.boxArray() == smf[0]->boxArray() &&
80  mf.DistributionMap() == smf[0]->DistributionMap())
81  {
82  dmf = &mf;
83  destcomp = dcomp;
84  sameba = true;
85  } else {
86  raii.define(smf[0]->boxArray(), smf[0]->DistributionMap(), ncomp, 0,
87  MFInfo(), smf[0]->Factory());
88 
89  dmf = &raii;
90  destcomp = 0;
91  sameba = false;
92  }
93 
94  if ((dmf != smf[0] && dmf != smf[1]) || scomp != dcomp)
95  {
96 #ifdef AMREX_USE_OMP
97 #pragma omp parallel if (Gpu::notInLaunchRegion())
98 #endif
99  for (MFIter mfi(*dmf,TilingIfNotGPU()); mfi.isValid(); ++mfi)
100  {
101  const Box& bx = mfi.tilebox();
102  const Real t0 = stime[0];
103  const Real t1 = stime[1];
104  auto const sfab0 = smf[0]->array(mfi);
105  auto const sfab1 = smf[1]->array(mfi);
106  auto dfab = dmf->array(mfi);
107 
108  if (time == t0)
109  {
110  AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
111  {
112  dfab(i,j,k,n+destcomp) = sfab0(i,j,k,n+scomp);
113  });
114  }
115  else if (time == t1)
116  {
117  AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
118  {
119  dfab(i,j,k,n+destcomp) = sfab1(i,j,k,n+scomp);
120  });
121  }
122  else if (! amrex::almostEqual(t0,t1))
123  {
124  Real alpha = (t1-time)/(t1-t0);
125  Real beta = (time-t0)/(t1-t0);
126  AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
127  {
128  dfab(i,j,k,n+destcomp) = alpha*sfab0(i,j,k,n+scomp)
129  + beta*sfab1(i,j,k,n+scomp);
130  });
131  }
132  else
133  {
134  AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
135  {
136  dfab(i,j,k,n+destcomp) = sfab0(i,j,k,n+scomp);
137  });
138  }
139  }
140  }
141 
142  if (sameba)
143  {
144  // Note that when sameba is true mf's BoxArray is nonoverlapping.
145  // So FillBoundary is safe.
146  mf.FillBoundary(dcomp, ncomp, nghost, geom.periodicity());
147  }
148  else
149  {
150  IntVect src_ngrow = IntVect::TheZeroVector();
151  IntVect dst_ngrow = nghost;
152 
153  mf.ParallelCopy(*dmf, 0, dcomp, ncomp, src_ngrow, dst_ngrow, geom.periodicity());
154  }
155  }
156  else {
157  amrex::Abort("FillPatchSingleLevel: high-order interpolation in time not implemented yet");
158  }
159 
160  physbcf(mf, dcomp, ncomp, nghost, time, bcfcomp);
161 }
