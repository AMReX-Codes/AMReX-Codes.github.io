189

191  for (int j = jlo; j <= jhi; ++j) {
192  for (int i = ilo; i <= ihi; ++i) {
193  if ((fine_state(i,j,0,n) + fine(i,j,0,n)) < 0.0) {
194  redo_me = true;
195  }
196  }
197  }
198 
199  /*

201  * interpolated correction, then we do nothing here.
202  *
203  * If any of the fine values are negative after the original
204  * interpolated correction, then we do our best.
205  */

209  * Calculate coarse cell volumes.
210  */

213  const Real* cs_dx = cs_geomdata.CellSize();
214  if (cs_geomdata.coord == CoordSys::cartesian) {
215  // Cartesian

217  } else {

220  Real rp = cs_problo[0] + (ic+0.5_rt)*cs_dx[0];
221  Real rm = cs_problo[0] + (ic-0.5_rt)*cs_dx[0];
222  cvol = (rp*rp - rm*rm) * cs_dx[1];
223  }
224

227  *
228  * crseTot = volume-weighted sum of all interpolated values
229  * of the correction, which is equivalent to
230  * the total volume-weighted coarse correction
231  *
232  * SumN = volume-weighted sum of all negative values of fine_state
233  *
234  * SumP = volume-weighted sum of all positive values of fine_state
235  */
236  Real crseTot = 0.0;
237  Real SumN = 0.0;
238  Real SumP = 0.0;
239  for (int j = jlo; j <= jhi; ++j) {
240  for (int i = ilo; i <= ihi; ++i) {
241

243  // Calculate fine cell volume
244  const Real* fn_dx = fn_geomdata.CellSize();
245  if (fn_geomdata.coord == CoordSys::cartesian) {
246  // Cartesian
247  fvol = fn_dx[0] * fn_dx[1];
248  } else {
249  // RZ
250  const Real* fn_problo = fn_geomdata.ProbLo();
251  Real rp = fn_problo[0] + (i+0.5_rt)*fn_dx[0];
252  Real rm = fn_problo[0] + (i-0.5_rt)*fn_dx[0];
253  fvol = (rp*rp - rm*rm) * fn_dx[1];
254  }
255 
256  // Calculate sums
257  crseTot += fvol * fine(i,j,0,n);
258  if (fine_state(i,j,0,n) <= 0.0) {
259  SumN += fvol * fine_state(i,j,0,n);
260  } else {
261  SumP += fvol * fine_state(i,j,0,n);
262  }
263  }
264  }
265 
266  if ( (crseTot > 0) && (crseTot > std::abs(SumN)) ) {
267 
268  /*
269  * Special case 1:
270  *
271  * Coarse correction > 0, and fine_state has some cells
272  * with negative values which will be filled before
273  * adding to the other cells.
274  *
275  * Use the correction to bring negative cells to zero,
276  * then distribute the remaining positive proportionally.
277  */
278  for (int j = jlo; j <= jhi; ++j) {
279  for (int i = ilo; i <= ihi; ++i) {
280

282  if (fine_state(i,j,0,n) < 0.0) {
283  fine(i,j,0,n) = -fine_state(i,j,0,n);
284  }
285

287  if (SumP > 0.0) {

289  Real alpha = (crseTot - std::abs(SumN)) / SumP;
290  fine(i,j,0,n) = alpha * fine_state(i,j,0,n);
291  }
292  } else { /* (SumP < 0) */

294  fine(i,j,0,n) += posVal;
295  }

297  }
298  }
299

304  *

306  * them all positive.

322  }
323 
324  }
325  }
326 
327  } else if ( (crseTot < 0) && (std::abs(crseTot) > SumP) ) {
328 
329  /*
330  * Special case 3:
331  *
332  * Coarse correction < 0, and fine_state DOES NOT have
333  * enough positive states to absorb it.
334  *
335  * Here we distribute the remaining negative amount
336  * in such a way as to make them all as close to the
337  * same negative value as possible.
338  */
339  for (int j = jlo; j <= jhi; ++j) {
340  for (int i = ilo; i <= ihi; ++i) {
341

343  Real negVal = (SumP + SumN + crseTot) / cvol;
344  fine(i,j,0,n) = negVal - fine_state(i,j,0,n);
345

356  * positive states to absorb all the negative
357  * correction *and* to redistribute to make
358  * negative cells positive.
359  */

366  } else {

368  Real alpha = (crseTot + SumN) / SumP;
369  fine(i,j,0,n) = alpha * fine_state(i,j,0,n);
370  }

372  }
373  }

383  * already negative.
384  *

386  * positive cell and don't touch the negatives.

389  for (int i = ilo; i <= ihi; ++i) {

394  } else {

397  fine(i,j,0,n) = alpha * fine_state(i,j,0,n);
398  }
399 
400  }
401  }
402 
403  } // special cases
404  } // redo_me
