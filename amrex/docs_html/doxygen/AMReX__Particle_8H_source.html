67  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
68  ParticleIDWrapper& operator= (const Long id) noexcept
69  {
70  // zero out the 40 leftmost bits, which store the sign and the abs of the id;
71  m_idata &= 0x00FFFFFF;
72 
73  uint64_t val;
74  uint64_t sign = id >= 0;
75  if (sign)
76  {
77  // 2**39-1, the max value representable in this fashion
78  AMREX_ASSERT(id <= 549755813887L);
79  val = id;
80  }
81  else
82  {
83  // -2**39-1, the min value representable in this fashion
84  AMREX_ASSERT(id >= -549755813887L);
85  val = -id;
86  }
87 
88  m_idata |= (sign << 63); // put the sign in the leftmost bit
89  m_idata |= (val << 24); // put the val in the next 39
90  return *this;
91  }
