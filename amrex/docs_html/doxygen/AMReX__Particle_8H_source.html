43  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
44  ParticleIDWrapper& operator= (const Long id) noexcept
45  {
46  // zero out the 40 leftmost bits, which store the sign and the abs of the id;
47  m_idata &= 0x00FFFFFF;
48 
49  uint64_t val;
50  uint64_t sign = id >= 0;
51  if (sign)
52  {
53  // 2**39-1, the max value representable in this fashion
54  AMREX_ASSERT(id <= 549755813887L);
55  val = id;
56  }
57  else
58  {
59  // -2**39-1, the min value representable in this fashion
60  AMREX_ASSERT(id >= -549755813887L);
61  val = -id;
62  }
63 
64  m_idata |= (sign << 63); // put the sign in the leftmost bit
65  m_idata |= (val << 24); // put the val in the next 39
66  return *this;
67  }
68 
69  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
70  operator Long () const noexcept
71  {
72  Long r = 0;
73 
74  uint64_t sign = m_idata >> 63; // extract leftmost sign bit
75  uint64_t val = ((m_idata >> 24) & 0x7FFFFFFFFF); // extract next 39 id bits
76 
77  Long lval = static_cast<Long>(val); // bc we take -
78  r = (sign) ? lval : -lval;
79  return r;
