234 AMREX_GPU_HOST_DEVICE AMREX_INLINE
235 static Real Volume (const IntVect& point, const GeometryData& geomdata)
236 {
237 const auto *dx = geomdata.CellSize();
238
239 Real vol;
240
241#if AMREX_SPACEDIM == 1
242
243 auto coord = geomdata.Coord();
244
245 if (coord == CoordSys::cartesian) {
246 // Cartesian
247
248 vol = dx[0];
249 }
250 else {
251 // Spherical
252
253 Real rl = geomdata.ProbLo()[0] + static_cast<Real>(point[0]) * dx[0];
254 Real rr = rl + dx[0];
255
256 constexpr Real pi = std::numbers::pi_v<Real>;
257 vol = (4.0_rt / 3.0_rt) * pi * dx[0] * (rl * rl + rl * rr + rr * rr);
258 }
259
260#elif AMREX_SPACEDIM == 2
261
262 auto coord = geomdata.Coord();
263
264 if (coord == CoordSys::cartesian) {
265 // Cartesian
266
267 vol = dx[0] * dx[1];
268 }
269 else if (coord == CoordSys::RZ) {
270 // Cylindrical
271
272 Real r_l = geomdata.ProbLo()[0] + static_cast<Real>(point[0]) * dx[0];
273 Real r_r = geomdata.ProbLo()[0] + static_cast<Real>(point[0]+1) * dx[0];
274
275 constexpr Real pi = std::numbers::pi_v<Real>;
276 vol = pi * (r_l + r_r) * dx[0] * dx[1];
277 }
278 else {
279 // Spherical
280
281 Real r_l = geomdata.ProbLo()[0] + static_cast<Real>(point[0]) * dx[0];
282 Real r_r = geomdata.ProbLo()[0] + static_cast<Real>(point[0]+1) * dx[0];
283
284 Real theta_l = geomdata.ProbLo()[1] + static_cast<Real>(point[1]) * dx[1];
285 Real theta_r = geomdata.ProbLo()[1] + static_cast<Real>(point[1]+1) * dx[1];
286
287 constexpr Real twoThirdsPi = static_cast<Real>((2.0 / 3.0) * std::numbers::pi_v<Real>);
288 vol = twoThirdsPi * (std::cos(theta_l) - std::cos(theta_r)) * dx[0] *
289 (r_r*r_r + r_r*r_l + r_l*r_l);
290 }
291
292#else
293
294 amrex::ignore_unused(point);
295
296 // Cartesian
297
298 vol = dx[0] * dx[1] * dx[2];
299
300#endif
301
302 return vol;
303 }
