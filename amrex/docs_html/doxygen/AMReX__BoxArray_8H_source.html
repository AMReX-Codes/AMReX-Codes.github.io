171 struct BATbndryReg
172 {
173  BATbndryReg (Orientation a_face, IndexType a_typ,
174  int a_in_rad, int a_out_rad, int a_extent_rad)
175  : m_face(a_face), m_typ(a_typ), m_crse_ratio(1)
176  {
177  m_loshft = IntVect(-a_extent_rad);
178  m_hishft = IntVect( a_extent_rad);
179  IntVect nodal = a_typ.ixType();
180  m_hishft += nodal;
181  const int d = a_face.coordDir();
182  if (m_face.isLow()) {
183  m_loshft[d] = nodal[d] - a_out_rad;
184  m_hishft[d] = nodal[d] + a_in_rad - 1;
185  } else {
186  m_loshft[d] = 1 - a_in_rad;
187  m_hishft[d] = a_out_rad;
188  }
189  m_doilo = IntVect(a_extent_rad);
190  m_doihi = IntVect(a_extent_rad);
191  m_doihi += nodal;
192  if (m_face.isLow()) { // domain of influence in index space
193  m_doilo[d] = std::max(0, a_out_rad - nodal[d]);
194  m_doihi[d] = 0;
195  } else {
196  m_doilo[d] = 0;
197  m_doihi[d] = a_out_rad;
198  }
199  }

201  Box operator() (const Box& a_bx) const noexcept {
202  IntVect lo = amrex::coarsen(a_bx.smallEnd(), m_crse_ratio);
203  IntVect hi = amrex::coarsen(a_bx.bigEnd(), m_crse_ratio);
204  const int d = m_face.coordDir();
205  if (m_face.isLow()) {
206  hi[d] = lo[d];
207  } else {
208  lo[d] = hi[d];
209  }
210  lo += m_loshft;
211  hi += m_hishft;
212  return Box(lo,hi,m_typ);
213  }
