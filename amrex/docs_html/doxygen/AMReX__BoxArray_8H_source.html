195struct BATbndryReg
196{
197 BATbndryReg (Orientation a_face, IndexType a_typ,
198 int a_in_rad, int a_out_rad, int a_extent_rad)
199 : m_face(a_face), m_typ(a_typ), m_crse_ratio(1)
200 {
201 m_loshft = IntVect(-a_extent_rad);
202 m_hishft = IntVect( a_extent_rad);
203 IntVect nodal = a_typ.ixType();
204 m_hishft += nodal;
205 const int d = a_face.coordDir();
206 if (nodal[d]) {
207 // InterpFaceRegister & SyncRegister in IAMR
208 if (m_face.isLow()) {
209 m_loshft[d] = 0;
210 m_hishft[d] = 0;
211 } else {
212 m_loshft[d] = 1;
213 m_hishft[d] = 1;
214 }
215 m_doilo = IntVect(0);
216 m_doihi = nodal;
217 } else {
218 // BndryRegister
219 if (m_face.isLow()) {
220 m_loshft[d] = nodal[d] - a_out_rad;
221 m_hishft[d] = nodal[d] + a_in_rad - 1;
222 } else {
223 m_loshft[d] = 1 - a_in_rad;
224 m_hishft[d] = a_out_rad;
225 }
226 m_doilo = IntVect(a_extent_rad);
227 m_doihi = IntVect(a_extent_rad);
228 m_doihi += nodal;
229 if (m_face.isLow()) { // domain of influence in index space
230 m_doilo[d] = a_out_rad;
231 m_doihi[d] = 0;
232 } else {
233 m_doilo[d] = 0;
234 m_doihi[d] = a_out_rad;
235 }
236 }
237 }
238
239 [[nodiscard]] Box operator() (const Box& a_bx) const noexcept {
240 IntVect lo = amrex::coarsen(a_bx.smallEnd(), m_crse_ratio);
241 IntVect hi = amrex::coarsen(a_bx.bigEnd(), m_crse_ratio);
242 const int d = m_face.coordDir();
243 if (m_face.isLow()) {
244 hi[d] = lo[d];
245 } else {
246 lo[d] = hi[d];
247 }
248 lo += m_loshft;
249 hi += m_hishft;
250 return Box(lo,hi,m_typ);
251 }
252
253 [[nodiscard]] Box coarsen (Box const& a_box) const noexcept { return amrex::coarsen(a_box,m_crse_ratio); }
254
255 [[nodiscard]] IntVect doiLo () const noexcept { return m_doilo; }
256 [[nodiscard]] IntVect doiHi () const noexcept { return m_doihi; }
257
258 [[nodiscard]] IndexType index_type () const noexcept { return m_typ; }
259 [[nodiscard]] IntVect coarsen_ratio () const noexcept { return m_crse_ratio; }
260
261 friend bool operator== (BATbndryReg const& a, BATbndryReg const& b) noexcept {
262 return a.m_face == b.m_face && a.m_typ == b.m_typ && a.m_crse_ratio == b.m_crse_ratio
263 && a.m_loshft == b.m_loshft && a.m_hishft == b.m_hishft
264 && a.m_doilo == b.m_doilo && a.m_doihi == b.m_doihi;
265 }
