171 struct BATbndryReg
172 {
173  BATbndryReg (Orientation a_face, IndexType a_typ,
174  int a_in_rad, int a_out_rad, int a_extent_rad)
175  : m_face(a_face), m_typ(a_typ), m_crse_ratio(1)
176  {
177  m_loshft = IntVect(-a_extent_rad);
178  m_hishft = IntVect( a_extent_rad);
179  IntVect nodal = a_typ.ixType();
180  m_hishft += nodal;
181  const int d = a_face.coordDir();
182  if (nodal[d]) {
183  // InterpFaceRegister & SyncRegister in IAMR
184  if (m_face.isLow()) {
185  m_loshft[d] = 0;
186  m_hishft[d] = 0;
187  } else {
188  m_loshft[d] = 1;
189  m_hishft[d] = 1;
190  }
191  m_doilo = IntVect(0);
192  m_doihi = nodal;
193  } else {
194  // BndryRegister
195  if (m_face.isLow()) {
196  m_loshft[d] = nodal[d] - a_out_rad;
197  m_hishft[d] = nodal[d] + a_in_rad - 1;
198  } else {
199  m_loshft[d] = 1 - a_in_rad;
200  m_hishft[d] = a_out_rad;
201  }
202  m_doilo = IntVect(a_extent_rad);
203  m_doihi = IntVect(a_extent_rad);
204  m_doihi += nodal;
205  if (m_face.isLow()) { // domain of influence in index space
206  m_doilo[d] = a_out_rad;
207  m_doihi[d] = 0;
208  } else {
209  m_doilo[d] = 0;
210  m_doihi[d] = a_out_rad;
211  }
212  }
213  }
214 
215  Box operator() (const Box& a_bx) const noexcept {
216  IntVect lo = amrex::coarsen(a_bx.smallEnd(), m_crse_ratio);
217  IntVect hi = amrex::coarsen(a_bx.bigEnd(), m_crse_ratio);
218  const int d = m_face.coordDir();
219  if (m_face.isLow()) {
220  hi[d] = lo[d];
221  } else {
222  lo[d] = hi[d];
223  }
224  lo += m_loshft;
225  hi += m_hishft;
226  return Box(lo,hi,m_typ);
227  }
