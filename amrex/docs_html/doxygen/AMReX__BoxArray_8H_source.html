192 struct BATbndryReg
193 {
194  BATbndryReg (Orientation a_face, IndexType a_typ,
195  int a_in_rad, int a_out_rad, int a_extent_rad)
196  : m_face(a_face), m_typ(a_typ), m_crse_ratio(1)
197  {
198  m_loshft = IntVect(-a_extent_rad);
199  m_hishft = IntVect( a_extent_rad);
200  IntVect nodal = a_typ.ixType();
201  m_hishft += nodal;
202  const int d = a_face.coordDir();
203  if (nodal[d]) {
204  // InterpFaceRegister & SyncRegister in IAMR
205  if (m_face.isLow()) {
206  m_loshft[d] = 0;
207  m_hishft[d] = 0;
208  } else {
209  m_loshft[d] = 1;
210  m_hishft[d] = 1;
211  }
212  m_doilo = IntVect(0);
213  m_doihi = nodal;
214  } else {
215  // BndryRegister
216  if (m_face.isLow()) {
217  m_loshft[d] = nodal[d] - a_out_rad;
218  m_hishft[d] = nodal[d] + a_in_rad - 1;
219  } else {
220  m_loshft[d] = 1 - a_in_rad;
221  m_hishft[d] = a_out_rad;
222  }
223  m_doilo = IntVect(a_extent_rad);
224  m_doihi = IntVect(a_extent_rad);
225  m_doihi += nodal;
226  if (m_face.isLow()) { // domain of influence in index space
227  m_doilo[d] = a_out_rad;
228  m_doihi[d] = 0;
229  } else {
230  m_doilo[d] = 0;
231  m_doihi[d] = a_out_rad;
232  }
233  }
234  }

236  [[nodiscard]] Box operator() (const Box& a_bx) const noexcept {
237  IntVect lo = amrex::coarsen(a_bx.smallEnd(), m_crse_ratio);
238  IntVect hi = amrex::coarsen(a_bx.bigEnd(), m_crse_ratio);
239  const int d = m_face.coordDir();
240  if (m_face.isLow()) {
241  hi[d] = lo[d];
242  } else {
243  lo[d] = hi[d];
244  }
245  lo += m_loshft;
246  hi += m_hishft;
247  return Box(lo,hi,m_typ);
248  }
