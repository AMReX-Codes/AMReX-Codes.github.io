87 MLCGSolverT<MF>::solve_bicgstab (MF& sol, const MF& rhs, RT eps_rel, RT eps_abs)
88 {
89  BL_PROFILE("MLCGSolver::bicgstab");
90 
91  const int ncomp = sol.nComp();
92 
93  const BoxArray& ba = sol.boxArray();
94  const DistributionMapping& dm = sol.DistributionMap();
95  const auto& factory = sol.Factory();
96 
97  MF ph(ba, dm, ncomp, sol.nGrowVect(), MFInfo(), factory);
98  MF sh(ba, dm, ncomp, sol.nGrowVect(), MFInfo(), factory);
99  ph.setVal(RT(0.0));
100  sh.setVal(RT(0.0));
101 
102  MF sorig(ba, dm, ncomp, nghost, MFInfo(), factory);
103  MF p (ba, dm, ncomp, nghost, MFInfo(), factory);
104  MF r (ba, dm, ncomp, nghost, MFInfo(), factory);
105  MF s (ba, dm, ncomp, nghost, MFInfo(), factory);
106  MF rh (ba, dm, ncomp, nghost, MFInfo(), factory);
107  MF v (ba, dm, ncomp, nghost, MFInfo(), factory);
108  MF t (ba, dm, ncomp, nghost, MFInfo(), factory);
109 
110  Lp.correctionResidual(amrlev, mglev, r, sol, rhs, MLLinOpT<MF>::BCMode::Homogeneous);
111 
112  // Then normalize
113  Lp.normalize(amrlev, mglev, r);
114 
115  sorig.LocalCopy(sol,0,0,ncomp,nghost);
116  rh.LocalCopy (r ,0,0,ncomp,nghost);
117 
118  sol.setVal(RT(0.0));
119 
120  RT rnorm = norm_inf(r);
121  const RT rnorm0 = rnorm;
122 
123  if ( verbose > 0 )
124  {
125  amrex::Print() << "MLCGSolver_BiCGStab: Initial error (error0) = " << rnorm0 << '\n';
126  }
127  int ret = 0;
128  iter = 1;
129  RT rho_1 = 0, alpha = 0, omega = 0;
130 
131  if ( rnorm0 == 0 || rnorm0 < eps_abs )
132  {
133  if ( verbose > 0 )
134  {
135  amrex::Print() << "MLCGSolver_BiCGStab: niter = 0,"
136  << ", rnorm = " << rnorm
137  << ", eps_abs = " << eps_abs << std::endl;
138  }
139  return ret;
140  }
141 
142  for (; iter <= maxiter; ++iter)
143  {
144  const RT rho = dotxy(rh,r);
145  if ( rho == 0 )
146  {
147  ret = 1; break;
148  }
149  if ( iter == 1 )
150  {
151  p.LocalCopy(r,0,0,ncomp,nghost);
152  }
153  else
154  {
155  const RT beta = (rho/rho_1)*(alpha/omega);
156  MF::Saxpy(p, -omega, v, 0, 0, ncomp, nghost); // p += -omega*v
157  MF::Xpay(p, beta, r, 0, 0, ncomp, nghost); // p = r + beta*p
158  }
159  ph.LocalCopy(p,0,0,ncomp,nghost);
160  Lp.apply(amrlev, mglev, v, ph, MLLinOpT<MF>::BCMode::Homogeneous, MLLinOpT<MF>::StateMode::Correction);
161  Lp.normalize(amrlev, mglev, v);
162 
163  RT rhTv = dotxy(rh,v);
164  if ( rhTv != RT(0.0) )
165  {
166  alpha = rho/rhTv;
167  }
168  else
169  {
170  ret = 2; break;
171  }
172  MF::Saxpy(sol, alpha, ph, 0, 0, ncomp, nghost); // sol += alpha * ph
173  MF::LinComb(s, RT(1.0), r, 0, -alpha, v, 0, 0, ncomp, nghost); // s = r - alpha * v
174 
175  rnorm = norm_inf(s);
176 
177  if ( verbose > 2 && ParallelDescriptor::IOProcessor() )
178  {
179  amrex::Print() << "MLCGSolver_BiCGStab: Half Iter "
180  << std::setw(11) << iter
181  << " rel. err. "
182  << rnorm/(rnorm0) << '\n';
183  }
184 
185  if ( rnorm < eps_rel*rnorm0 || rnorm < eps_abs ) { break; }
186 
187  sh.LocalCopy(s,0,0,ncomp,nghost);
188  Lp.apply(amrlev, mglev, t, sh, MLLinOpT<MF>::BCMode::Homogeneous, MLLinOpT<MF>::StateMode::Correction);
189  Lp.normalize(amrlev, mglev, t);
190  //
191  // This is a little funky. I want to elide one of the reductions
192  // in the following two dotxy()s. We do that by calculating the "local"
193  // values and then reducing the two local values at the same time.
194  //
195  RT tvals[2] = { dotxy(t,t,true), dotxy(t,s,true) };
196 
197  BL_PROFILE_VAR("MLCGSolver::ParallelAllReduce", blp_par);
198  ParallelAllReduce::Sum(tvals,2,Lp.BottomCommunicator());
199  BL_PROFILE_VAR_STOP(blp_par);
200 
201  if ( tvals[0] != RT(0.0) )
202  {
203  omega = tvals[1]/tvals[0];
204  }
205  else
206  {
207  ret = 3; break;
208  }
209  MF::Saxpy(sol, omega, sh, 0, 0, ncomp, nghost); // sol += omega * sh
210  MF::LinComb(r, RT(1.0), s, 0, -omega, t, 0, 0, ncomp, nghost); // r = s - omega * t
211 
212  rnorm = norm_inf(r);
213 
214  if ( verbose > 2 )
215  {
216  amrex::Print() << "MLCGSolver_BiCGStab: Iteration "
217  << std::setw(11) << iter
218  << " rel. err. "
219  << rnorm/(rnorm0) << '\n';
220  }
221 
222  if ( rnorm < eps_rel*rnorm0 || rnorm < eps_abs ) { break; }
223 
224  if ( omega == 0 )
225  {
226  ret = 4; break;
227  }
228  rho_1 = rho;
229  }
230 
231  if ( verbose > 0 )
232  {
233  amrex::Print() << "MLCGSolver_BiCGStab: Final: Iteration "
234  << std::setw(4) << iter
235  << " rel. err. "
236  << rnorm/(rnorm0) << '\n';
237  }
238 
239  if ( ret == 0 && rnorm > eps_rel*rnorm0 && rnorm > eps_abs)
240  {
241  if ( verbose > 0 && ParallelDescriptor::IOProcessor() ) {
242  amrex::Warning("MLCGSolver_BiCGStab:: failed to converge!");
243  }
244  ret = 8;
245  }
246 
247  if ( ( ret == 0 || ret == 8 ) && (rnorm < rnorm0) )
248  {
249  sol.LocalAdd(sorig, 0, 0, ncomp, nghost);
250  }
251  else
252  {
253  sol.setVal(RT(0.0));
254  sol.LocalAdd(sorig, 0, 0, ncomp, nghost);
255  }
256 
257  return ret;
258 }

263 {
264  BL_PROFILE("MLCGSolver::cg");
265

267

269  const DistributionMapping& dm = sol.DistributionMap();
270  const auto& factory = sol.Factory();
271 
272  MF p(ba, dm, ncomp, sol.nGrowVect(), MFInfo(), factory);
273  p.setVal(RT(0.0));
274 
275  MF sorig(ba, dm, ncomp, nghost, MFInfo(), factory);
276  MF r (ba, dm, ncomp, nghost, MFInfo(), factory);
277  MF z (ba, dm, ncomp, nghost, MFInfo(), factory);
278  MF q (ba, dm, ncomp, nghost, MFInfo(), factory);
279 
280  sorig.LocalCopy(sol,0,0,ncomp,nghost);
281 
282  Lp.correctionResidual(amrlev, mglev, r, sol, rhs, MLLinOpT<MF>::BCMode::Homogeneous);
283 
284  sol.setVal(RT(0.0));
285 
286  RT rnorm = norm_inf(r);
287  const RT rnorm0 = rnorm;
288 
289  if ( verbose > 0 )
290  {
291  amrex::Print() << "MLCGSolver_CG: Initial error (error0) : " << rnorm0 << '\n';
292  }
293 
294  RT rho_1 = 0;
295  int ret = 0;
296  iter = 1;
297 
298  if ( rnorm0 == 0 || rnorm0 < eps_abs )
299  {
300  if ( verbose > 0 ) {
301  amrex::Print() << "MLCGSolver_CG: niter = 0,"
302  << ", rnorm = " << rnorm
303  << ", eps_abs = " << eps_abs << std::endl;
304  }
305  return ret;
306  }
307 
308  for (; iter <= maxiter; ++iter)
309  {
310  z.LocalCopy(r,0,0,ncomp,nghost);
311 
312  RT rho = dotxy(z,r);
313 
314  if ( rho == 0 )
315  {
316  ret = 1; break;
317  }
318  if (iter == 1)
319  {
320  p.LocalCopy(z,0,0,ncomp,nghost);
321  }
322  else
323  {
324  RT beta = rho/rho_1;
325  MF::Xpay(p, beta, z, 0, 0, ncomp, nghost); // p = z + beta * p
326  }
327  Lp.apply(amrlev, mglev, q, p, MLLinOpT<MF>::BCMode::Homogeneous, MLLinOpT<MF>::StateMode::Correction);
328 
329  RT alpha;
330  RT pw = dotxy(p,q);
331  if ( pw != RT(0.0))
332  {
333  alpha = rho/pw;
334  }
335  else
336  {
337  ret = 1; break;
338  }
339 
340  if ( verbose > 2 )
341  {
342  amrex::Print() << "MLCGSolver_cg:"
343  << " iter " << iter
344  << " rho " << rho
345  << " alpha " << alpha << '\n';
346  }
347  MF::Saxpy(sol, alpha, p, 0, 0, ncomp, nghost); // sol += alpha * p
348  MF::Saxpy(r, -alpha, q, 0, 0, ncomp, nghost); // r += -alpha * q
349  rnorm = norm_inf(r);
350 
351  if ( verbose > 2 )
352  {
353  amrex::Print() << "MLCGSolver_cg: Iteration"
354  << std::setw(4) << iter
355  << " rel. err. "
356  << rnorm/(rnorm0) << '\n';
357  }
358 
359  if ( rnorm < eps_rel*rnorm0 || rnorm < eps_abs ) { break; }
360 
361  rho_1 = rho;
362  }
363 
364  if ( verbose > 0 )
365  {
366  amrex::Print() << "MLCGSolver_cg: Final Iteration"
367  << std::setw(4) << iter
368  << " rel. err. "
369  << rnorm/(rnorm0) << '\n';
370  }
371 
372  if ( ret == 0 && rnorm > eps_rel*rnorm0 && rnorm > eps_abs )
373  {
374  if ( verbose > 0 && ParallelDescriptor::IOProcessor() ) {
375  amrex::Warning("MLCGSolver_cg: failed to converge!");
376  }
377  ret = 8;
378  }
379 
380  if ( ( ret == 0 || ret == 8 ) && (rnorm < rnorm0) )
381  {
382  sol.LocalAdd(sorig, 0, 0, ncomp, nghost);
383  }
384  else
385  {
386  sol.setVal(RT(0.0));

388  }
