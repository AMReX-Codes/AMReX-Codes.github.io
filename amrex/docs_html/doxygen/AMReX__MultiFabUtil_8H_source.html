397

400  //

518 template <typename F>
519 Real
520 NormHelper (const MultiFab& x, int xcomp,
521  const MultiFab& y, int ycomp,
522  F && f,
523  int numcomp, IntVect nghost, bool local)
524 {
525  BL_ASSERT(x.boxArray() == y.boxArray());
526  BL_ASSERT(x.DistributionMap() == y.DistributionMap());
527  BL_ASSERT(x.nGrowVect().allGE(nghost) && y.nGrowVect().allGE(nghost));
528 
529  Real sm = Real(0.0);
530 #ifdef AMREX_USE_GPU
531  if (Gpu::inLaunchRegion()) {
532  auto const& xma = x.const_arrays();
533  auto const& yma = y.const_arrays();
534  sm = ParReduce(TypeList<ReduceOpSum>{}, TypeList<Real>{}, x, nghost,
535  [=] AMREX_GPU_DEVICE (int box_no, int i, int j, int k) noexcept -> GpuTuple<Real>
536  {
537  Real t = Real(0.0);
538  auto const& xfab = xma[box_no];
539  auto const& yfab = yma[box_no];
540  for (int n = 0; n < numcomp; ++n) {
541  t += f(xfab(i,j,k,xcomp+n) , yfab(i,j,k,ycomp+n));
542  }
543  return t;
544  });
545  } else
546 #endif
547  {
548 #ifdef AMREX_USE_OMP
549 #pragma omp parallel if (!system::regtest_reduction) reduction(+:sm)
550 #endif
551  for (MFIter mfi(x,true); mfi.isValid(); ++mfi)
552  {
553  Box const& bx = mfi.growntilebox(nghost);
554  Array4<Real const> const& xfab = x.const_array(mfi);
555  Array4<Real const> const& yfab = y.const_array(mfi);
556  AMREX_LOOP_4D(bx, numcomp, i, j, k, n,
557  {
558  sm += f(xfab(i,j,k,xcomp+n) , yfab(i,j,k,ycomp+n));
559  });
560  }
561  }
562 
563  if (!local) {
564  ParallelAllReduce::Sum(sm, ParallelContext::CommunicatorSub());
565  }
566 
567  return sm;
568 }
