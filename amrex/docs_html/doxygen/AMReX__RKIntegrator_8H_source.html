171  amrex::Real advance (T& S_old, T& S_new, amrex::Real time, const amrex::Real time_step) override
172  {
173  BaseT::timestep = time_step;
174  // Assume before advance() that S_old is valid data at the current time ("time" argument)
175  // And that if data is a MultiFab, both S_old and S_new contain ghost cells for evaluating a stencil based RHS
176  // We need this from S_old. This is convenient for S_new to have so we can use it
177  // as scratch space for stage values without creating a new scratch MultiFab with ghost cells.
178 
179  // Fill the RHS F_nodes at each stage
180  for (int i = 0; i < number_nodes; ++i)
181  {
182  // Get current stage time, t = t_old + h * Ci
183  amrex::Real stage_time = time + BaseT::timestep * nodes[i];
184 
185  // Fill S_new with the solution value for evaluating F at the current stage
186  // Copy S_new = S_old
187  IntegratorOps<T>::Copy(S_new, S_old);
188  if (i > 0) {
189  // Saxpy across the tableau row:
190  // S_new += h * Aij * Fj
191  // We should fuse these kernels ...
192  for (int j = 0; j < i; ++j)
193  {
194  IntegratorOps<T>::Saxpy(S_new, BaseT::timestep * tableau[i][j], *F_nodes[j]);
195  }
196 
197  // Call the post-update hook for the stage state value
198  BaseT::post_update(S_new, stage_time);
199  }
200 
201  // Fill F[i], the RHS at the current stage
202  // F[i] = RHS(y, t) at y = stage_value, t = stage_time
203  BaseT::rhs(*F_nodes[i], S_new, stage_time);
204  }
205 
206  // Fill new State, starting with S_new = S_old.
207  // Then Saxpy S_new += h * Wi * Fi for integration weights Wi
208  // We should fuse these kernels ...
209  IntegratorOps<T>::Copy(S_new, S_old);
210  for (int i = 0; i < number_nodes; ++i)
211  {
212  IntegratorOps<T>::Saxpy(S_new, BaseT::timestep * weights[i], *F_nodes[i]);
213  }
214 
215  // Call the post-update hook for S_new
216  BaseT::post_update(S_new, time + BaseT::timestep);
217 
218  // If we are working with an extended Butcher tableau, we can estimate the error here,
219  // and then calculate an adaptive timestep.
220

222  return BaseT::timestep;
