224template <int Power, typename T,
225 typename = std::enable_if_t<!std::is_integral<T>() || Power>=0>>
226AMREX_FORCE_INLINE
227constexpr T powi (T x) noexcept
228{
229 if constexpr (Power < 0) {
230 return T(1)/powi<-Power>(x);
231 } else if constexpr (Power == 0) {
232 //note: 0^0 is implementation-defined, but most compilers return 1
233 return T(1);
234 } else if constexpr (Power == 1) {
235 return x;
236 } else if constexpr (Power == 2) {
237 return x*x;
238 } else if constexpr (Power%2 == 0) {
239 return powi<2>(powi<Power/2>(x));
240 } else {
241 return x*powi<Power-1>(x);
242 }
243}

261template <typename T>
262AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
263T comp_ellint_1 (T k)
264{
265 // Computing K based on DLMF
266 // https://dlmf.nist.gov/19.8
267 T tol = std::numeric_limits<T>::epsilon();
268
269 T a0 = 1.0;
270 T g0 = std::sqrt(1.0 - k*k);
271 T a = a0;
272 T g = g0;
273
274 // Find Arithmetic Geometric mean
275 while(std::abs(a0 - g0) > tol) {
276 a = 0.5*(a0 + g0);
277 g = std::sqrt(a0 * g0);
278
279 a0 = a;
280 g0 = g;
281 }
282
283 return 0.5*pi<T>()/a;
284}

286template <typename T>
287AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
288T comp_ellint_2 (T k)
289{
290 // Computing E based on DLMF
291 // https://dlmf.nist.gov/19.8
292 T Kcomp = amrex::Math::comp_ellint_1<T>(k);
293 T tol = std::numeric_limits<T>::epsilon();
294
295 // Step Zero
296 T a0 = 1.0;
297 T g0 = std::sqrt(1.0 - k*k);
298 T cn = std::sqrt(a0*a0 - g0*g0);
299
300 // Step 1
301 int n = 1;
302 T a = 0.5 * (a0 + g0);
303 T g = std::sqrt(a0*g0);
304 cn = 0.25*cn*cn/a;
305
306 T sum_val = a*a;
307 a0 = a;
308 g0 = g;
309
310 while(std::abs(cn*cn) > tol) {
311 // Compute coefficients for this iteration
312 a = 0.5 * (a0 + g0);
313 g = std::sqrt(a0*g0);
314 cn = 0.25*cn*cn/a;
315
316 n++;
317 sum_val -= std::pow(2,n-1)*cn*cn;
318
319 // Save a and g for next iteration
320 a0 = a;
321 g0 = g;
322 }
323
324 return Kcomp*sum_val;
325}

374struct FastDivmodU64
375{
376 std::uint64_t divisor = 0;
377
378#ifdef AMREX_INT128_SUPPORTED
379 std::uint64_t multiplier = 1U;
380 unsigned int shift_right = 0;
381 unsigned int round_up = 0;
382
383 //
384 // Static methods
385 //
386
388 static std::uint32_t integer_log2 (std::uint64_t x)
389 {
390 std::uint32_t n = 0;
391 while (x >>= 1) {
392 ++n;
393 }
394 return n;
395 }
396
400 FastDivmodU64 (std::uint64_t divisor_)
401 : divisor(divisor_)
402 {
403 if (divisor) {
404 shift_right = integer_log2(divisor);
405
406 if ((divisor & (divisor - 1)) == 0) {
407 multiplier = 0;
408 }
409 else {
410 std::uint64_t power_of_two = (std::uint64_t(1) << shift_right);
411 auto n = amrex::UInt128_t(power_of_two) << 64;
412 std::uint64_t multiplier_lo = n / divisor;
413 n += power_of_two;
414 multiplier = n / divisor;
415 round_up = (multiplier_lo == multiplier ? 1 : 0);
416 }
417 }
418 }
419
420#else
421
422 FastDivmodU64 (std::uint64_t divisor_) : divisor(divisor_) {}
423
424#endif
425
427 FastDivmodU64 () = default;

430 [[nodiscard]] AMREX_GPU_HOST_DEVICE
431 std::uint64_t divide (std::uint64_t dividend) const
432 {
433#if defined(AMREX_INT128_SUPPORTED)
434 auto x = dividend;
435 if (multiplier) {
436 x = amrex::Math::umulhi(dividend + round_up, multiplier);
437 }
438 return (x >> shift_right);
439#else
440 return dividend / divisor;
441#endif
442 }
443
445 [[nodiscard]] AMREX_GPU_HOST_DEVICE
446 std::uint64_t modulus (std::uint64_t quotient, std::uint64_t dividend) const
447 {
448 return dividend - quotient * divisor;
449 }
450
452 [[nodiscard]] AMREX_GPU_HOST_DEVICE
453 std::uint64_t divmod (std::uint64_t &remainder, std::uint64_t dividend) const
454 {
455 auto quotient = divide(dividend);
456 remainder = modulus(quotient, dividend);
457 return quotient;
458 }
459
462 AMREX_GPU_HOST_DEVICE
463 void operator() (std::uint64_t &quotient, std::uint64_t &remainder, std::uint64_t dividend) const
464 {
465 quotient = divmod(remainder, dividend);
466 }
467};
