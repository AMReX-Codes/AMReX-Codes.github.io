321  inline std::pair<int,int>
322  team_range (int begin, int end, int rit = -1, int nworkers = 0) noexcept
323  {
324  int rb, re;
325  {
326  if (rit < 0) rit = ParallelDescriptor::MyRankInTeam();
327  if (nworkers == 0) nworkers = ParallelDescriptor::TeamSize();
328  BL_ASSERT(rit<nworkers);
329  if (nworkers == 1) {
330  rb = begin;
331  re = end;
332  } else {
333  int ntot = end - begin;
334  int nr = ntot / nworkers;
335  int nlft = ntot - nr * nworkers;
336  if (rit < nlft) { // get nr+1 items
337  rb = begin + rit * (nr + 1);
338  re = rb + nr + 1;
339  } else { // get nr items
340  rb = begin + rit * nr + nlft;
341  re = rb + nr;
342  }
343  }
344  }
345 
346 #ifdef _OPENMP
347  int nthreads = omp_get_num_threads();
348  if (nthreads > 1) {
349  int tid = omp_get_thread_num();
350  int ntot = re - rb;
351  int nr = ntot / nthreads;
352  int nlft = ntot - nr * nthreads;
353  if (tid < nlft) { // get nr+1 items
354  rb += tid * (nr + 1);
355  re = rb + nr + 1;
356  } else { // get nr items
357  rb += tid * nr + nlft;
358  re = rb + nr;
359  }
360  }
361 #endif
362 
363  return std::make_pair(rb,re);
364  }

964  int root)
965 {
966  BL_PROFILE_T_S("ParallelDescriptor::Gatherv(Tlong)", T);
967  BL_COMM_PROFILE(BLProfiler::Gatherv, BLProfiler::BeforeCall(), root, BLProfiler::NoTag());
968 
969  int tag = ParallelDescriptor::SeqNum();
970 
971  if (MyProc() == root) {
972  int nprocs = NProcs();
973  BL_ASSERT(rc.size() == nprocs);
974  BL_ASSERT(disp.size() == nprocs);
975 
976  std::vector<MPI_Request> request;
977  for (int i = 0; i < nprocs; ++i) {
978  T* buf = recv + disp[i];
979  if (i == root) {
980  for (int k = 0; k < rc[i]; ++k) {
981  *buf++ = send[k];
982  }
983  } else {

986  tag, Communicator(), &req);

988  }

993  }
994  } else {
995  MPI_Send(const_cast<T*>(send), sc, ParallelDescriptor::Mpi_typemap<T>::type(), root, tag,
996  Communicator());
997  }
