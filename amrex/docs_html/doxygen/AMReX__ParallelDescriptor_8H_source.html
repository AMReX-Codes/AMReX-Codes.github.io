329  inline std::pair<int,int>
330  team_range (int begin, int end, int rit = -1, int nworkers = 0) noexcept
331  {
332  int rb, re;
333  {
334  if (rit < 0) rit = ParallelDescriptor::MyRankInTeam();
335  if (nworkers == 0) nworkers = ParallelDescriptor::TeamSize();
336  BL_ASSERT(rit<nworkers);
337  if (nworkers == 1) {
338  rb = begin;
339  re = end;
340  } else {
341  int ntot = end - begin;
342  int nr = ntot / nworkers;
343  int nlft = ntot - nr * nworkers;
344  if (rit < nlft) { // get nr+1 items
345  rb = begin + rit * (nr + 1);
346  re = rb + nr + 1;
347  } else { // get nr items
348  rb = begin + rit * nr + nlft;
349  re = rb + nr;
350  }
351  }
352  }
353 
354 #ifdef _OPENMP
355  int nthreads = omp_get_num_threads();
356  if (nthreads > 1) {
357  int tid = omp_get_thread_num();
358  int ntot = re - rb;
359  int nr = ntot / nthreads;
360  int nlft = ntot - nr * nthreads;
361  if (tid < nlft) { // get nr+1 items
362  rb += tid * (nr + 1);
363  re = rb + nr + 1;
364  } else { // get nr items
365  rb += tid * nr + nlft;
366  re = rb + nr;
367  }
368  }
369 #endif
370 
371  return std::make_pair(rb,re);
372  }

888 template <class T>
889 void
890 ParallelDescriptor::GatherLayoutDataToVector (const LayoutData<T>& sendbuf,
891  Vector<T>& recvbuf, int root)
892 {
893  BL_PROFILE_T_S("ParallelDescriptor::GatherLayoutData(Ti)", T);
894 
895  // Gather prelims
896  Vector<int> index_to_send = sendbuf.IndexArray();
897  Vector<T> T_to_send;
898  T_to_send.reserve(sendbuf.local_size());
899 
900  for (int i : sendbuf.IndexArray())
901  {
902  T_to_send.push_back(sendbuf[i]);
903  }
904 
905  int nprocs = ParallelContext::NProcsSub();
906  Vector<int> recvcount(nprocs, 0);
907  const Vector<int>& old_pmap = sendbuf.DistributionMap().ProcessorMap();
908  for (int i=0; i<old_pmap.size(); ++i)
909  {
910  ++recvcount[old_pmap[i]];
911  }
912 
913  // Make a map from post-gather to pre-gather index
914  Vector<Vector<int>> new_ind_to_old_ind(nprocs);
915  for (int i=0; i<nprocs; ++i)
916  {
917  new_ind_to_old_ind[i].reserve(recvcount[i]);
918  }
919  for (int i=0; i<old_pmap.size(); ++i)
920  {
921  new_ind_to_old_ind[old_pmap[i]].push_back(i);
922  }
923 
924  // Flatten
925  Vector<int> new_index_to_old_index;
926  new_index_to_old_index.reserve(old_pmap.size());
927  for (const Vector<int>& v : new_ind_to_old_ind)
928  {
929  if (v.size()>0)
930  {
931  for (int el : v)
932  {
933  new_index_to_old_index.push_back(el);
934  }
935  }
936  }
937 
938  Vector<int> disp(nprocs);
939  if (!disp.empty()) disp[0] = 0;
940  std::partial_sum(recvcount.begin(), recvcount.end()-1, disp.begin()+1);
941  Vector<T> new_index_to_T(sendbuf.size());
942 
943  MPI_Gatherv(T_to_send.data(), T_to_send.size(),
944  ParallelDescriptor::Mpi_typemap<T>::type(),
945  new_index_to_T.data(), recvcount.data(), disp.data(),
946  ParallelDescriptor::Mpi_typemap<T>::type(),
947  root, ParallelContext::CommunicatorSub());
948 
949  // Now work just on the root, which now has global information on the collected;
950  // LayoutData information; with new_index_to_old_index and new_index_to_T,
951  // sort the gathered vector in pre-gather index space
952  if (ParallelContext::MyProcSub() == root)
953  {
954  // Invert the map (new_index) --> (old_index)
955  Vector<int> old_index_to_new_index(sendbuf.size());
956 
957  for (int i=0; i<old_index_to_new_index.size(); ++i)
958  {
959  old_index_to_new_index[new_index_to_old_index[i]] = i;
960  }
961 
962  for (int i=0; i<recvbuf.size(); ++i)
963  {
964  recvbuf[i] = new_index_to_T[old_index_to_new_index[i]];
965  }
966  }
967 }
