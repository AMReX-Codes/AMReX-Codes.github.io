329  inline std::pair<int,int>
330  team_range (int begin, int end, int rit = -1, int nworkers = 0) noexcept
331  {
332  int rb, re;
333  {
334  if (rit < 0) rit = ParallelDescriptor::MyRankInTeam();
335  if (nworkers == 0) nworkers = ParallelDescriptor::TeamSize();
336  BL_ASSERT(rit<nworkers);
337  if (nworkers == 1) {
338  rb = begin;
339  re = end;
340  } else {
341  int ntot = end - begin;
342  int nr = ntot / nworkers;
343  int nlft = ntot - nr * nworkers;
344  if (rit < nlft) { // get nr+1 items
345  rb = begin + rit * (nr + 1);
346  re = rb + nr + 1;
347  } else { // get nr items
348  rb = begin + rit * nr + nlft;
349  re = rb + nr;
350  }
351  }
352  }
353 
354 #ifdef _OPENMP
355  int nthreads = omp_get_num_threads();
356  if (nthreads > 1) {
357  int tid = omp_get_thread_num();
358  int ntot = re - rb;
359  int nr = ntot / nthreads;
360  int nlft = ntot - nr * nthreads;
361  if (tid < nlft) { // get nr+1 items
362  rb += tid * (nr + 1);
363  re = rb + nr + 1;
364  } else { // get nr items
365  rb += tid * nr + nlft;
366  re = rb + nr;
367  }
368  }
369 #endif
370 
371  return std::make_pair(rb,re);
372  }

889 template <class T>
890 void
891 ParallelDescriptor::GatherLayoutDataToVector (const LayoutData<T>& sendbuf,
892  Vector<T>& recvbuf, int root)
893 {
894  BL_PROFILE_T_S("ParallelDescriptor::GatherLayoutData(Ti)", T);
895 
896  // Gather prelims
897  Vector<int> index_to_send = sendbuf.IndexArray();
898  Vector<T> T_to_send;
899  T_to_send.reserve(sendbuf.local_size());
900 
901  for (int i : sendbuf.IndexArray())
902  {
903  T_to_send.push_back(sendbuf[i]);
904  }
905 
906  int nprocs = ParallelContext::NProcsSub();
907  Vector<int> recvcount(nprocs, 0);
908  const Vector<int>& old_pmap = sendbuf.DistributionMap().ProcessorMap();
909  for (int i=0; i<old_pmap.size(); ++i)
910  {
911  ++recvcount[old_pmap[i]];
912  }
913 
914  // Make a map from post-gather to pre-gather index
915  Vector<Vector<int>> new_ind_to_old_ind(nprocs);
916  for (int i=0; i<nprocs; ++i)
917  {
918  new_ind_to_old_ind[i].reserve(recvcount[i]);
919  }
920  for (int i=0; i<old_pmap.size(); ++i)
921  {
922  new_ind_to_old_ind[old_pmap[i]].push_back(i);
923  }
924 
925  // Flatten
926  Vector<int> new_index_to_old_index;
927  new_index_to_old_index.reserve(old_pmap.size());
928  for (const Vector<int>& v : new_ind_to_old_ind)
929  {
930  if (v.size()>0)
931  {
932  for (int el : v)
933  {
934  new_index_to_old_index.push_back(el);
935  }
936  }
937  }
938 
939  Vector<int> disp(nprocs);
940  if (!disp.empty()) disp[0] = 0;
941  std::partial_sum(recvcount.begin(), recvcount.end()-1, disp.begin()+1);
942  Vector<T> new_index_to_T(sendbuf.size());
943 
944  MPI_Gatherv(T_to_send.data(), T_to_send.size(),
945  ParallelDescriptor::Mpi_typemap<T>::type(),
946  new_index_to_T.data(), recvcount.data(), disp.data(),
947  ParallelDescriptor::Mpi_typemap<T>::type(),
948  root, ParallelContext::CommunicatorSub());
949 
950  // Now work just on the root, which now has global information on the collected;
951  // LayoutData information; with new_index_to_old_index and new_index_to_T,
952  // sort the gathered vector in pre-gather index space
953  if (ParallelContext::MyProcSub() == root)
954  {
955  // Invert the map (new_index) --> (old_index)
956  Vector<int> old_index_to_new_index(sendbuf.size());
957 
958  for (int i=0; i<old_index_to_new_index.size(); ++i)
959  {
960  old_index_to_new_index[new_index_to_old_index[i]] = i;
961  }
962 
963  for (int i=0; i<recvbuf.size(); ++i)
964  {
965  recvbuf[i] = new_index_to_T[old_index_to_new_index[i]];
966  }
967  }
968 }
