221 BndryRegisterT<MF>::operator+= (const BndryRegisterT<MF>& rhs)
222 {
223  BL_ASSERT(grids == rhs.grids);
224  for (OrientationIter face; face; ++face) {
225  const auto f = face();
226  const int ncomp = bndry[f].nComp();
227 #ifdef AMREX_USE_OMP
228 #pragma omp parallel if (Gpu::notInLaunchRegion())
229 #endif
230  for (FabSetIter bfsi(rhs[f]); bfsi.isValid(); ++bfsi) {
231  const Box& bx = bfsi.validbox();
232  auto const sfab = rhs[f].array(bfsi);
233  auto dfab = bndry[f].array(bfsi);
234  AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
235  {
236  dfab(i,j,k,n) += sfab(i,j,k,n);
237  });
238  }
239  }
240  return *this;
241 }
