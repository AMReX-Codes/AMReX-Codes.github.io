702 void
703 ParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt, Allocator>
704 ::CreateVirtualParticles (int level, ParticleTileType& virts) const
705 {
706  BL_PROFILE("ParticleContainer::CreateVirtualParticles()");
707  AMREX_ASSERT(level > 0);
708  AMREX_ASSERT(virts.empty());
709 
710  if (level >= static_cast<int>(m_particles.size()))
711  return;
712 
713  std::string aggregation_type = AggregationType();
714  int aggregation_buffer = AggregationBuffer();
715 
716  if (aggregation_type == "None")
717  {
718  for(ParConstIterType pti(*this, level); pti.isValid(); ++pti)
719  {
720  const auto& ptile = ParticlesAt(level, pti);
721  const auto src = ptile.getConstParticleTileData();
722 
723  for (int pindex = 0; pindex < ptile.numParticles(); ++pindex)
724  {
725  SuperParticleType p = src.getSuperParticle(pindex);
726  p.id() = VirtualParticleID;
727  virts.push_back(p);
728  }
729  }
730  }
731 
732  if (aggregation_type == "Cell")
733  {
734  BoxList bl_buffer;
735  bl_buffer.complementIn(Geom(level).Domain(), ParticleBoxArray(level));
736  BoxArray buffer(std::move(bl_buffer));
737  buffer.grow(aggregation_buffer);
738 
739  for(ParConstIterType pti(*this, level); pti.isValid(); ++pti)
740  {
741  const auto& ptile = ParticlesAt(level, pti);
742  const auto& aos = pti.GetArrayOfStructs();
743  const auto src = ptile.getConstParticleTileData();
744 
745  std::map<IntVect,SuperParticleType> agg_map;
746 
747  for (int pindex = 0; pindex < ptile.numParticles(); ++pindex)
748  {
749  IntVect cell = Index(aos[pindex], level);
750  if (buffer.contains(cell))
751  {
752  // It's in the no-aggregation buffer.
753  // Set its id to indicate that it's a virt.
754  SuperParticleType p = src.getSuperParticle(pindex);
755  p.id() = VirtualParticleID;
756  virts.push_back(p);
757  }
758  else
759  {
760  //
761  // Note that Cell aggregation assumes that p.rdata(0) is mass and
762  // that all other components should be combined in a mass-weighted
763  // average.
764  //
765  auto agg_map_it = agg_map.find(cell);
766 
767  if (agg_map_it == agg_map.end())
768  {
769  //
770  // Add the particle.
771  //
772  SuperParticleType p = src.getSuperParticle(pindex);
773  p.id() = VirtualParticleID;
774  agg_map[cell] = p;
775  }
776  else
777  {
778  AMREX_ASSERT(agg_map_it != agg_map.end());
779  const SuperParticleType& pnew = src.getSuperParticle(pindex);
780  SuperParticleType& pold = agg_map_it->second;
781  const Real old_mass = pold.rdata(0);
782  const Real new_mass = pnew.rdata(0);
783  const Real total_mass = old_mass + new_mass;
784  //
785  // Set the position to the center of mass.
786  //
787  for (int i = 0; i < AMREX_SPACEDIM; i++)
788  {
789  pold.pos(i) = static_cast<ParticleReal>((old_mass*pold.pos(i) + new_mass*pnew.pos(i))/total_mass);
790  }
791  AMREX_ASSERT(this->Index(pold, level) == cell);
792  //
793  // Set the metadata (presumably velocity) to the mass-weighted average.
794  //
795  for (int i = 1; i < NStructReal; i++)
796  {
797  pold.rdata(i) = static_cast<ParticleReal>((old_mass*pold.rdata(i) + new_mass*pnew.rdata(i))/total_mass);
798  }
799  pold.rdata(0) = static_cast<ParticleReal>(total_mass);
800  }
801  }
802  }
803 
804  //
805  // Add the aggregated particles to the virtuals.
806  //
807  for (const auto& agg_particle : agg_map)
808  {
809  virts.push_back(agg_particle.second);
810  }
811  }
812  }
813 }

2016  {

2018  delete mf_pointer;
