656 template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
657 void
658 ParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>
659 ::CreateVirtualParticles (int level, AoS& virts) const
660 {
661  BL_PROFILE("ParticleContainer::CreateVirtualParticles()");
662  BL_ASSERT(level > 0);
663  BL_ASSERT(virts.empty());
664 
665  if (level >= static_cast<int>(m_particles.size()))
666  return;
667 
668  if (aggregation_type == "")
669  {
670  ParmParse pp("particles");
671  aggregation_type = "None";
672  pp.query("aggregation_type", aggregation_type);
673  aggregation_buffer = 2;
674  pp.query("aggregation_buffer", aggregation_buffer);
675  }
676 
677  if (aggregation_type == "None");
678  else if (aggregation_type == "Cell");
679  else if (aggregation_type == "Flow") amrex::Abort("Flow aggregation not implemented");
680  else amrex::Abort("Unknown Particle Aggregation mode");
681 
682  if (aggregation_type == "None")
683  {
684  const auto& pmap = m_particles[level];
685  for (const auto& kv : pmap)
686  {
687  const auto& pbox = kv.second.GetArrayOfStructs();
688  for (auto it = pbox.cbegin(); it != pbox.cend(); ++it)
689  {
690  ParticleType p = *it;
691  p.m_idata.id = VirtualParticleID;
692  virts.push_back(p);
693  }
694  }
695  return;
696  }
697 
698 
699  if (aggregation_type == "Cell")
700  {
701  BoxList bl_buffer;
702  bl_buffer.complementIn(Geom(level).Domain(), ParticleBoxArray(level));
703  BoxArray buffer(std::move(bl_buffer));
704  buffer.grow(aggregation_buffer);
705 
706  const auto& pmap = m_particles[level];
707  for (const auto& kv : pmap)
708  {
709  const auto& pbox = kv.second.GetArrayOfStructs();
710 
711  std::map<IntVect,ParticleType> agg_map;
712 
713  for (auto it = pbox.cbegin(); it != pbox.cend(); ++it)
714  {
715  IntVect cell = Index(*it, level);
716  if (buffer.contains(cell))
717  {
718  // It's in the no-aggregation buffer.
719  // Set its id to indicate that it's a virt.
720  ParticleType p = *it;
721  p.m_idata.id = VirtualParticleID;
722  virts.push_back(p);
723  }
724  else
725  {
726  //
727  // Note that Cell aggregation assumes that p.m_rdata.arr[AMREX_SPACEDIM] is mass and
728  // that all other components should be combined in a mass-weighted
729  // average.
730  //
731  auto agg_map_it = agg_map.find(cell);
732 
733  if (agg_map_it == agg_map.end())
734  {
735  //
736  // Add the particle.
737  //
738  ParticleType p = *it;
739  //
740  // Set its id to indicate that it's a virt.
741  //
742  p.m_idata.id = VirtualParticleID;
743  agg_map[cell] = p;
744  }
745  else
746  {
747  BL_ASSERT(agg_map_it != agg_map.end());
748  const ParticleType& pnew = *it;
749  ParticleType& pold = agg_map_it->second;
750  const Real old_mass = pold.m_rdata.arr[AMREX_SPACEDIM];
751  const Real new_mass = pnew.m_rdata.arr[AMREX_SPACEDIM];
752  const Real total_mass = old_mass + new_mass;
753  //
754  // Set the position to the center of mass.
755  //
756  for (int i = 0; i < AMREX_SPACEDIM; i++)
757  {
758  pold.m_rdata.pos[i] = (old_mass*pold.m_rdata.pos[i] + new_mass*pnew.m_rdata.pos[i])/total_mass;
759  }
760  BL_ASSERT(this->Index(pold, level) == cell);
761  //
762  // Set the metadata (presumably velocity) to the mass-weighted average.
763  //
764  for (int i = AMREX_SPACEDIM + 1; i < AMREX_SPACEDIM + NStructReal; i++)
765  {
766  pold.m_rdata.arr[i] = (old_mass*pold.m_rdata.arr[i] + new_mass*pnew.m_rdata.arr[i])/total_mass;
767  }
768  pold.m_rdata.arr[AMREX_SPACEDIM] = total_mass;
769  }
770  }
771  }
772 
773  //
774  // Add the aggregated particles to the virtuals.
775  //
776  for (const auto& agg_particle : agg_map)
777  {
778  virts.push_back(agg_particle.second);
779  }
780  }
781  }
782 }

1158

1160

1162  {

1177  }

1184  //

1186  functors::assignParticleGrid(geom.data(),
1187  (*redistribute_mask_ptr)[mfi].box(),
1188  redistribute_mask_ptr->array(mfi)));
1189

1192  // those that move go to the end.

1195  m_index_sequence_tmp.begin(),

1199

1202

1204

1206  new_size = old_size + num_moved;
1207  }
