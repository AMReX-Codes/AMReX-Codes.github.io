216  void integrate (T& S_old, T& S_new, amrex::Real start_time, const amrex::Real start_timestep,
217  const amrex::Real end_time, const int start_step, const int max_steps)
218  {
219  time = start_time;
220  timestep = start_timestep;
221  bool stop_advance = false;
222  for (step_number = start_step; step_number < max_steps && !stop_advance; ++step_number)
223  {
224  if (end_time - time < timestep) {
225  timestep = end_time - time;
226  stop_advance = true;
227  }
228 
229  if (step_number > 0) {
230  std::swap(S_old, S_new);
231  }
232 
233  // Call the time integrator advance
234  integrator_ptr->advance(S_old, S_new, time, timestep);
235 
236  // Update our time variable
237  time += timestep;
238 
239  // Call the post-timestep hook
240  post_timestep();
241  }
