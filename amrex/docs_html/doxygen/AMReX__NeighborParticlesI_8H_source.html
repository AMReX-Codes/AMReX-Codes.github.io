586  const std::vector<IntVect>& pshifts = periodicity.shiftIntVect();
587  const BoxArray& ba = this->ParticleBoxArray(lev);
588  const IntVect& iv = this->Index(p, lev);
589  for (auto pit=pshifts.cbegin(); pit!=pshifts.cend(); ++pit)
590  {
591  Box pbox = amrex::grow(Box(iv, iv), ref_fac*nGrow) + (*pit);
592  bool first_only = false;
593  ba.intersections(pbox, isects, first_only, 0);
594  for (const auto& isec : isects)
595  {
596  const Box& grid_box = ba[isec.first];
597  for (IntVect cell = pbox.smallEnd(); cell <= pbox.bigEnd(); pbox.next(cell)) {
598  if ( !grid_box.contains(cell) ) continue;
599  int tile = getTileIndex(cell, grid_box,
600  this->do_tiling, this->tile_size, tbx);
601  auto nbor = NeighborCopyTag(lev, isec.first, tile);
602  nbor.periodic_shift = -1*(*pit);
603  if (src_tag != nbor) tags.push_back(nbor);
604  }
605  }
606  }
607  }
608 
609  RemoveDuplicates(tags);
610  return;
611  }
612 }
