796 template <class MF>
797 amrex::EnableIf_t<IsFabArray<MF>::value>
798 FillBoundary (Vector<MF*> const& mf, Vector<int> const& scomp,
799  Vector<int> const& ncomp, Vector<IntVect> const& nghost,
800  Vector<Periodicity> const& period, Vector<int> const& cross = {})
801 {
802  BL_PROFILE("FillBoundary(Vector)");
803 #if 1
804  const int N = mf.size();
805  for (int i = 0; i < N; ++i) {
806  mf[i]->FillBoundary_nowait(scomp[i], ncomp[i], nghost[i], period[i],
807  cross.empty() ? 0 : cross[i]);
808  }
809  for (int i = 0; i < N; ++i) {
810  mf[i]->FillBoundary_finish();
811  }
812 
813 #else
814  using FAB = typename MF::FABType::value_type;
815  using T = typename FAB::value_type;
816 
817  const int nmfs = mf.size();
818  Vector<FabArrayBase::CommMetaData const*> cmds;
819  int N_locs = 0;
820  int N_rcvs = 0;
821  int N_snds = 0;
822  for (int imf = 0; imf < nmfs; ++imf) {
823  if (nghost[imf].max() > 0) {
824  auto const& TheFB = mf[imf]->getFB(nghost[imf], period[imf],
825  cross.empty() ? 0 : cross[imf]);
826  // The FB is cached. Therefore it's safe take its address for later use.
827  cmds.push_back(static_cast<FabArrayBase::CommMetaData const*>(&TheFB));
828  N_locs += TheFB.m_LocTags->size();
829  N_rcvs += TheFB.m_RcvTags->size();
830  N_snds += TheFB.m_SndTags->size();
831  } else {
832  cmds.push_back(nullptr);
833  }
834  }
835 
836  using TagT = Array4CopyTag<T>;
837  Vector<TagT> local_tags;
838  local_tags.reserve(N_locs);
839  static_assert(amrex::IsStoreAtomic<T>::value, "FillBoundary(Vector): storing T is not atomic");
840  for (int imf = 0; imf < nmfs; ++imf) {
841  if (cmds[imf]) {
842  auto const& tags = *(cmds[imf]->m_LocTags);
843  for (auto const& tag : tags) {
844  local_tags.push_back({(*mf[imf])[tag.dstIndex].array (scomp[imf],ncomp[imf]),
845  (*mf[imf])[tag.srcIndex].const_array(scomp[imf],ncomp[imf]),
846  tag.dbox,
847  (tag.sbox.smallEnd()-tag.dbox.smallEnd()).dim3()});
848  }
849  }
850  }
851 
852  if (ParallelContext::NProcsSub() == 1) {
853  detail::fbv_copy(local_tags);
854  return;
855  }
856 
857 #ifdef AMREX_USE_MPI
858  //
859  // Do this before prematurely exiting if running in parallel.
860  // Otherwise sequence numbers will not match across MPI processes.
861  //
862  int SeqNum = ParallelDescriptor::SeqNum();
863  MPI_Comm comm = ParallelContext::CommunicatorSub();
864 
865  if (N_locs == 0 && N_rcvs == 0 && N_snds == 0) return; // No work to do
866 
867  char* the_recv_data = nullptr;
868  Vector<int> recv_from;
869  Vector<std::size_t> recv_size;
870  Vector<MPI_Request> recv_reqs;
871  Vector<MPI_Status> recv_stat;
872  Vector<TagT> recv_tags;
873 
874  if (N_rcvs > 0) {
875 
876  for (int imf = 0; imf < nmfs; ++imf) {
877  if (cmds[imf]) {
878  auto const& tags = *(cmds[imf]->m_RcvTags);
879  for (const auto& kv : tags) {
880  recv_from.push_back(kv.first);
881  }
882  }
883  }
884  amrex::RemoveDuplicates(recv_from);
885  const int nrecv = recv_from.size();
886 
887  recv_reqs.resize(nrecv, MPI_REQUEST_NULL);
888  recv_stat.resize(nrecv);
889 
890  recv_tags.reserve(N_rcvs);
891 
892  Vector<Vector<std::size_t> > recv_offset(nrecv);
893  Vector<std::size_t> offset;
894  recv_size.reserve(nrecv);
895  offset.reserve(nrecv);
896  std::size_t TotalRcvsVolume = 0;
897  for (int i = 0; i < nrecv; ++i) {
898  std::size_t nbytes = 0;
899  for (int imf = 0; imf < nmfs; ++imf) {
900  if (cmds[imf]) {
901  auto const& tags = *(cmds[imf]->m_RcvTags);
902  auto it = tags.find(recv_from[i]);
903  if (it != tags.end()) {
904  for (auto const& cct : it->second) {
905  auto& dfab = (*mf[imf])[cct.dstIndex];
906  recv_offset[i].push_back(nbytes);
907  recv_tags.push_back({dfab.array(scomp[imf],ncomp[imf]),
908  makeArray4<T const>(nullptr,cct.dbox,ncomp[imf]),
909  cct.dbox, Dim3{0,0,0}});
910  nbytes += dfab.nBytes(cct.dbox,ncomp[imf]);
911  }
912  }
913  }
914  }
915 
916  std::size_t acd = ParallelDescriptor::alignof_comm_data(nbytes);
917  nbytes = amrex::aligned_size(acd, nbytes); // so that nbytes are aligned
918 
919  // Also need to align the offset properly
920  TotalRcvsVolume = amrex::aligned_size(std::max(alignof(T),acd), TotalRcvsVolume);
921 
922  offset.push_back(TotalRcvsVolume);
923  TotalRcvsVolume += nbytes;
924 
925  recv_size.push_back(nbytes);
926  }
927 
928  the_recv_data = static_cast<char*>(amrex::The_FA_Arena()->alloc(TotalRcvsVolume));
929 
930  int k = 0;
931  for (int i = 0; i < nrecv; ++i) {
932  char* p = the_recv_data + offset[i];
933  const int rank = ParallelContext::global_to_local_rank(recv_from[i]);
934  recv_reqs[i] = ParallelDescriptor::Arecv
935  (p, recv_size[i], rank, SeqNum, comm).req();
936  for (int j = 0, nj = recv_offset[i].size(); j < nj; ++j) {
937  recv_tags[k++].sfab.p = (T const*)(p + recv_offset[i][j]);
938  }
939  }
940  }
941 
942  char* the_send_data = nullptr;
943  Vector<int> send_rank;
944  Vector<char*> send_data;
945  Vector<std::size_t> send_size;
946  Vector<MPI_Request> send_reqs;
947  if (N_snds > 0) {
948  for (int imf = 0; imf < nmfs; ++imf) {
949  if (cmds[imf]) {
950  auto const& tags = *(cmds[imf]->m_SndTags);
951  for (auto const& kv : tags) {
952  send_rank.push_back(kv.first);
953  }
954  }
955  }
956  amrex::RemoveDuplicates(send_rank);
957  const int nsend = send_rank.size();
958 
959  send_data.resize(nsend, nullptr);
960  send_reqs.resize(nsend, MPI_REQUEST_NULL);
961 
962  Vector<TagT> send_tags;
963  send_tags.reserve(N_snds);
964 
965  Vector<Vector<std::size_t> > send_offset(nsend);
966  Vector<std::size_t> offset;
967  send_size.reserve(nsend);
968  offset.reserve(nsend);
969  std::size_t TotalSndsVolume = 0;
970  for (int i = 0; i < nsend; ++i) {
971  std::size_t nbytes = 0;
972  for (int imf = 0; imf < nmfs; ++imf) {
973  if (cmds[imf]) {
974  auto const& tags = *(cmds[imf]->m_SndTags);
975  auto it = tags.find(send_rank[i]);
976  if (it != tags.end()) {
977  for (auto const& cct : it->second) {
978  auto const& sfab = (*mf[imf])[cct.srcIndex];
979  send_offset[i].push_back(nbytes);
980  send_tags.push_back({amrex::makeArray4<T>(nullptr,cct.sbox,ncomp[imf]),
981  sfab.const_array(scomp[imf],ncomp[imf]),
982  cct.sbox, Dim3{0,0,0}});
983  nbytes += sfab.nBytes(cct.sbox,ncomp[imf]);
984  }
985  }
986  }
987  }
988 
989  std::size_t acd = ParallelDescriptor::alignof_comm_data(nbytes);
990  nbytes = amrex::aligned_size(acd, nbytes); // so that bytes are aligned
991 
992  // Also need to align the offset properly
993  TotalSndsVolume = amrex::aligned_size(std::max(alignof(T),acd), TotalSndsVolume);
994 
995  offset.push_back(TotalSndsVolume);
996  TotalSndsVolume += nbytes;
997 
998  send_size.push_back(nbytes);
999  }
1000 
1001  the_send_data = static_cast<char*>(amrex::The_FA_Arena()->alloc(TotalSndsVolume));
1002  int k = 0;
1003  for (int i = 0; i < nsend; ++i) {
1004  send_data[i] = the_send_data + offset[i];
1005  for (int j = 0, nj = send_offset[i].size(); j < nj; ++j) {
1006  send_tags[k++].dfab.p = (T*)(send_data[i] + send_offset[i][j]);
1007  }
1008  }
1009 
1010  detail::fbv_copy(send_tags);
1011 
1012  FabArray<FAB>::PostSnds(send_data, send_size, send_rank, send_reqs, SeqNum);
1013  }
1014 
1015 #if !defined(AMREX_DEBUG)
1016  int recv_flag;
1017  ParallelDescriptor::Test(recv_reqs, recv_flag, recv_stat);
1018 #endif
1019 
1020  if (N_locs > 0) {
1021  detail::fbv_copy(local_tags);
1022 #if !defined(AMREX_DEBUG)
1023  ParallelDescriptor::Test(recv_reqs, recv_flag, recv_stat);
1024 #endif
1025  }
1026 
1027  if (N_rcvs > 0) {
1028  ParallelDescriptor::Waitall(recv_reqs, recv_stat);
1029 #ifdef AMREX_DEBUG
1030  if (!FabArrayBase::CheckRcvStats(recv_stat, recv_size, SeqNum)) {
1031  amrex::Abort("FillBoundary(vector) failed with wrong message size");
1032  }
1033 #endif
1034 
1035  detail::fbv_copy(recv_tags);
1036 
1037  amrex::The_FA_Arena()->free(the_recv_data);
1038  }
1039 
1040  if (N_snds > 0) {
1041  Vector<MPI_Status> stats(send_reqs.size());
1042  ParallelDescriptor::Waitall(send_reqs, stats);
1043  amrex::The_FA_Arena()->free(the_send_data);
1044  }
1045 
1046 #endif // #ifdef AMREX_USE_MPI
1047 #endif // #if 1 #else
