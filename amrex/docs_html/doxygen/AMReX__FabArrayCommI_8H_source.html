686  }

688 #endif

699  const int ncomp = fb_ncomp;

754 void
755 FabArray<FAB>::ParallelCopy (const FabArray<FAB>& src,
756  int scomp,
757  int dcomp,
758  int ncomp,
759  const IntVect& snghost,
760  const IntVect& dnghost,
761  const Periodicity& period,
762  CpOp op,
763  const FabArrayBase::CPC * a_cpc)
764 {
765  BL_PROFILE("FabArray::ParallelCopy()");

767  if (size() == 0 || src.size() == 0) return;
768 
769  BL_ASSERT(op == FabArrayBase::COPY || op == FabArrayBase::ADD);
770  BL_ASSERT(boxArray().ixType() == src.boxArray().ixType());
771 
772  BL_ASSERT(src.nGrowVect().allGE(snghost));
773  BL_ASSERT( nGrowVect().allGE(dnghost));
774 
775  if ((src.boxArray().ixType().cellCentered() || op == FabArrayBase::COPY) &&
776  (boxarray == src.boxarray && distributionMap == src.distributionMap)
777  && snghost == IntVect::TheZeroVector() && dnghost == IntVect::TheZeroVector()
778  && !period.isAnyPeriodic())
779  {
780  //
781  // Short-circuit full intersection code if we're doing copy()s or if
782  // we're doing plus()s on cell-centered data. Don't do plus()s on
783  // non-cell-centered data this simplistic way.
784  //
785 #ifdef _OPENMP
786 #pragma omp parallel if (FAB::isCopyOMPSafe() && Gpu::notInLaunchRegion())
787 #endif
788  for (MFIter fai(*this,TilingIfNotGPU()); fai.isValid(); ++fai)
789  {
790  const Box& bx = fai.tilebox();
791 
792  // avoid self copy or plus
793  if (this != &src) {
794  auto const sfab = src.array(fai);
795  auto dfab = this->array(fai);
796  if (op == FabArrayBase::COPY) {
797  AMREX_HOST_DEVICE_FOR_4D ( bx, ncomp, i, j, k, n,
798  {
799  dfab(i,j,k,dcomp+n) = sfab(i,j,k,scomp+n);
800  });
801  } else {
802  AMREX_HOST_DEVICE_FOR_4D ( bx, ncomp, i, j, k, n,
803  {
804  dfab(i,j,k,dcomp+n) += sfab(i,j,k,scomp+n);
805  });
806  }
807  }
808  }
809 
810  return;
811  }
812 
813  const CPC& thecpc = (a_cpc) ? *a_cpc : getCPC(dnghost, src, snghost, period);
814 
815  if (ParallelContext::NProcsSub() == 1)
816  {
817  //
818  // There can only be local work to do.
819  //

822  if (Gpu::inLaunchRegion() || !is_thread_safe)
823  {

825  for (int i = 0; i < N_loc; ++i)
826  {
827  const CopyComTag& tag = (*thecpc.m_LocTags)[i];
828  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {
829  loc_copy_tags[tag.dstIndex].push_back
830  ({src.fabPtr(tag.srcIndex), tag.dbox, tag.sbox.smallEnd()-tag.dbox.smallEnd()});
831  }
832  }
833 #ifdef _OPENMP
834 #pragma omp parallel if (Gpu::notInLaunchRegion())
835 #endif
836  for (MFIter mfi(*this); mfi.isValid(); ++mfi)
837  {
838  const auto& tags = loc_copy_tags[mfi];
839  auto dfab = this->array(mfi);
840  if (op == FabArrayBase::COPY)
841  {
842  for (auto const & tag : tags)
843  {
844  auto const sfab = tag.sfab->array();
845  Dim3 offset = tag.offset.dim3();
846  AMREX_HOST_DEVICE_FOR_4D ( tag.dbox, ncomp, i, j, k, n,
847  {
848  dfab(i,j,k,dcomp+n) = sfab(i+offset.x,j+offset.y,k+offset.z,scomp+n);
849  });
850  }
851  }
852  else
853  {
854  for (auto const & tag : tags)
855  {
856  auto const sfab = tag.sfab->array();
857  Dim3 offset = tag.offset.dim3();
858  AMREX_HOST_DEVICE_FOR_4D ( tag.dbox, ncomp, i, j, k, n,
859  {
860  dfab(i,j,k,dcomp+n) += sfab(i+offset.x,j+offset.y,k+offset.z,scomp+n);
861  });
862  }
863  }
864  }
865  }
866  else
867  {
868 #ifdef _OPENMP
869 #pragma omp parallel for
870 #endif
871  for (int i = 0; i < N_loc; ++i)
872  {
873  const CopyComTag& tag = (*thecpc.m_LocTags)[i];
874  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {
875  // avoid self copy or plus
876  const FAB* sfab = &(src[tag.srcIndex]);
877  FAB* dfab = &(get(tag.dstIndex));
878  if (op == FabArrayBase::COPY)
879  {
880  dfab->copy(*sfab, tag.sbox, scomp, tag.dbox, dcomp, ncomp);
881  }
882  else
883  {
884  dfab->plus(*sfab, tag.sbox, tag.dbox, scomp, dcomp, ncomp);
885  }
886  }
887  }
888  }
889 
890  return;
891  }

893 #ifdef BL_USE_MPI
894 
895  //
896  // Do this before prematurely exiting if running in parallel.
897  // Otherwise sequence numbers will not match across MPI processes.
898  //
899  int preSeqNum;
900  if (!FAB::preAllocatable()) {
901  preSeqNum = ParallelDescriptor::SeqNum();
902  }
903  int SeqNum = ParallelDescriptor::SeqNum();
904

906  const int N_rcvs = thecpc.m_RcvTags->size();
907  const int N_locs = thecpc.m_LocTags->size();
908 
909  if (N_locs == 0 && N_rcvs == 0 && N_snds == 0) {
910  //
911  // No work to do.
912  //
913  return;
914  }
915

917  // Send/Recv at most MaxComp components at a time to cut down memory usage.
918  //

922  {

924

927  //

930  Vector<int> send_size;
931  Vector<int> send_rank;
932  Vector<MPI_Request> send_reqs;

935  Vector<MPI_Request> pre_reqs;

940  send_size.reserve(N_snds);

942  send_reqs.reserve(N_snds);

951

953  if (FAB::preAllocatable())
954  {

956  {
957  nbytes += src[cct.srcIndex].nBytes(cct.sbox,SC,NC);
958  }
959  }
960  else
961  {
962  for (auto const& tag : cctc)
963  {
964  std::size_t b = src[tag.srcIndex].nBytes(tag.sbox,SC,NC);
965  nbytes += b;
966  iss.push_back(static_cast<int>(b));
967  }

973

977  send_reqs.push_back(MPI_REQUEST_NULL);
978  send_cctc.push_back(&cctc);

988  send_data[i] = p;
989  p += send_size[i];
990  }
991  }
992  }
993  }
994

996  {

998  for (int j=0; j<N_snds; ++j)
999  {
1000  pre_reqs[j] = ParallelDescriptor::Asend
1001  (indv_send_size[j].data(), indv_send_size[j].size(),
1002  ParallelContext::global_to_local_rank(send_rank[j]),
1003  preSeqNum,
1004  ParallelContext::CommunicatorSub()).req();
1005  }

1009  Vector<char*> recv_data;
1010  Vector<int> recv_size;
1011  Vector<MPI_Request> recv_reqs;
1012  //
1013  // Post rcvs. Allocate one chunk of space to hold'm all.
1014  //

1016

1018  if (N_rcvs > 0) {
1019  PostRcvs(*thecpc.m_RcvTags, the_recv_data,
1020  recv_data, recv_size, recv_from, recv_reqs, SC, NC, SeqNum, preSeqNum);
1021  actual_n_rcvs = N_rcvs - std::count(recv_size.begin(), recv_size.end(), 0);
1022  }
1023 
1024  //
1025  // Post send's
1026  //
1027  if (N_snds > 0)
1028  {
1029  bool is_thread_safe = FAB::isCopyOMPSafe();
1030 #ifdef _OPENMP
1031 #pragma omp parallel if (is_thread_safe && Gpu::notInLaunchRegion())
1032 #endif
1033  for (Gpu::StreamIter sit(N_snds,is_thread_safe); sit.isValid(); ++sit)
1034  {
1035  const int j = sit();
1036  char* dptr = send_data[j];
1037  if (dptr != nullptr)
1038  {
1039  auto const& cctc = *send_cctc[j];
1040  for (auto const& tag : cctc)
1041  {
1042  const Box& bx = tag.sbox;
1043  auto const sfab = src.array(tag.srcIndex);
1044  auto pfab = amrex::makeArray4((value_type*)(dptr),bx,NC);
1045  AMREX_HOST_DEVICE_FOR_4D ( bx, NC, ii, jj, kk, n,
1046  {
1047  pfab(ii,jj,kk,n) = sfab(ii,jj,kk,SC+n);
1048  });
1049 
1050  dptr += (bx.numPts() * NC * sizeof(value_type));
1051  }
1052  BL_ASSERT(dptr == send_data[j] + send_size[j]);
1053  }
1054  }
1055

1057  while (send_counter < N_snds)
1058  {
1059  int j;
1060 
1061  if (FAB::preAllocatable())
1062  {
1063  j = send_counter;
1064  }
1065  else
1066  {
1067  MPI_Status status;
1068  ParallelDescriptor::Waitany(pre_reqs, j, status);
1069  }
1070

1072  {
1073  send_reqs[j] = ParallelDescriptor::Asend
1074  (send_data[j], send_size[j],
1075  ParallelContext::global_to_local_rank(send_rank[j]),
1076  SeqNum,
1077  ParallelContext::CommunicatorSub()).req();
1078  }
1079 
1080  ++send_counter;
1081  }
1082  }

1085  // Do the local work. Hope for a bit of communication/computation overlap.
1086  //
1087  if (N_locs > 0)
1088  {
1089  bool is_thread_safe = FAB::isCopyOMPSafe() && thecpc.m_threadsafe_loc;
1090  if (Gpu::inLaunchRegion() || !is_thread_safe)
1091  {
1092  // gpu version or omp over dest fabs
1093

1095  for (int j = 0; j < N_locs; ++j) {
1096  const CopyComTag& tag = (*thecpc.m_LocTags)[j];
1097  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {
1098  copy_tags[tag.dstIndex].push_back
1099  ({src.fabPtr(tag.srcIndex), tag.dbox, tag.sbox.smallEnd()-tag.dbox.smallEnd()});
1100  }
1101  }
1102 
1103 #ifdef _OPENMP
1104 #pragma omp parallel if (Gpu::notInLaunchRegion())
1105 #endif
1106  for (MFIter mfi(*this); mfi.isValid(); ++mfi) {
1107  auto dfab = this->array(mfi);
1108  const auto& fab_copy_tags = copy_tags[mfi];
1109  if (op == FabArrayBase::COPY) {
1110  for (auto const& fab_tag : fab_copy_tags) {
1111  Dim3 offset = fab_tag.offset.dim3();
1112  auto const sfab = fab_tag.sfab->array();
1113  AMREX_HOST_DEVICE_FOR_4D ( fab_tag.dbox, NC, i, j, k, n,
1114  {
1115  dfab(i,j,k,DC+n) = sfab(i+offset.x,j+offset.y,k+offset.z,SC+n);
1116  });
1117  }
1118  } else {
1119  for (auto const& fab_tag : fab_copy_tags) {
1120  Dim3 offset = fab_tag.offset.dim3();
1121  auto const sfab = fab_tag.sfab->array();
1122  AMREX_HOST_DEVICE_FOR_4D ( fab_tag.dbox, NC, i, j, k, n,
1123  {
1124  dfab(i,j,k,DC+n) += sfab(i+offset.x,j+offset.y,k+offset.z,SC+n);
1125  });
1126  }
1127  }
1128  }
1129  }

1131  {

1133 #ifdef _OPENMP
1134 #pragma omp parallel for
1135 #endif
1136  for (int j = 0; j < N_locs; ++j)
1137  {
1138  const CopyComTag& tag = (*thecpc.m_LocTags)[j];
1139  // avoid self copy or plus
1140  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {

1142  FAB* dfab = &(get(tag.dstIndex));
1143  if (op == FabArrayBase::COPY) {
1144  dfab->copy(*sfab, tag.sbox, SC, tag.dbox, DC, NC);

1146  dfab->plus(*sfab, tag.sbox, tag.dbox, SC, DC, NC);
1147  }
1148  }
1149  }
1150  }
1151  }

1154  Vector<const CopyComTagsContainer*> recv_cctc;
1155  LayoutData<Vector<VoidCopyTag> > recv_copy_tags;

1160  {

1165  }
1166  }

1169  if (Gpu::inLaunchRegion() || !is_thread_safe)

1172  for (int k = 0; k < N_rcvs; ++k) {
1173  const char* dptr = recv_data[k];
1174  if (dptr != nullptr)
1175  {
1176  auto const& cctc = *recv_cctc[k];
1177  for (auto const& tag : cctc)
1178  {
1179  recv_copy_tags[tag.dstIndex].push_back({dptr,tag.dbox});
1180  dptr += tag.dbox.numPts() * NC * sizeof(value_type);
1181  }

1183  }
1184  }
1185  }
1186  }

1188  //
1189  // wait and unpack
1190  //
1191  if (actual_n_rcvs > 0) {

1196  amrex::Abort("ParallelCopy failed with wrong message size");
1197  }
1198  }
1199 
1200  if (N_rcvs > 0)

1202  if (!recv_copy_tags.empty())
1203  {
1204  // gpu version or omp over dest fabs
1205 #ifdef _OPENMP
1206 #pragma omp parallel if (Gpu::notInLaunchRegion())
1207 #endif
1208  for (MFIter mfi(*this); mfi.isValid(); ++mfi) {
1209  auto dfab = this->array(mfi);
1210  const auto& void_copy_tags = recv_copy_tags[mfi];
1211  if (op == FabArrayBase::COPY)
1212  {
1213  for (auto const& void_tag : void_copy_tags) {
1214  auto pfab = amrex::makeArray4((value_type*)(void_tag.p), void_tag.dbox, NC);
1215  AMREX_HOST_DEVICE_FOR_4D ( void_tag.dbox, NC, i, j, k, n,
1216  {
1217  dfab(i,j,k,DC+n) = pfab(i,j,k,n);
1218  });
1219  }
1220  } else {
1221  for (auto const& void_tag : void_copy_tags) {

1223  AMREX_HOST_DEVICE_FOR_4D ( void_tag.dbox, NC, i, j, k, n,

1226  });
1227  }
1228  }
1229  }
1230  }

1235 #endif

1237  {
1238  const char* dptr = recv_data[k];
1239  if (dptr != nullptr)
1240  {
1241  auto const& cctc = *recv_cctc[k];
1242  for (auto const& tag : cctc)
1243  {

1245  if (op == FabArrayBase::COPY)
1246  {
1247  dfab->copyFromMem(tag.dbox,DC,NC,dptr);
1248  }

1252  }

1255  }

1257  }
1258  }
1259  }
