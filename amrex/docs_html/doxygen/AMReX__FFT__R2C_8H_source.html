728R2C<T,D,C>::make_layout_from_local_domain (std::array<int,AMREX_SPACEDIM> const& local_start,
729 std::array<int,AMREX_SPACEDIM> const& local_size)
730{
731 IntVect lo(local_start);
732 IntVect len(local_size);
733 Box bx(lo, lo+len-1);
734#ifdef AMREX_USE_MPI
735 Vector<Box> allboxes(ParallelDescriptor::NProcs());
736 MPI_Allgather(&bx, 1, ParallelDescriptor::Mpi_typemap<Box>::type(),
737 allboxes.data(), 1, ParallelDescriptor::Mpi_typemap<Box>::type(),
738 ParallelDescriptor::Communicator());
739 Vector<int> pmap;
740 pmap.reserve(allboxes.size());
741 for (int i = 0; i < allboxes.size(); ++i) {
742 if (allboxes[i].ok()) {
743 pmap.push_back(i);
744 }
745 }
746 allboxes.erase(std::remove_if(allboxes.begin(), allboxes.end(),
747 [=] (Box const& b) { return b.isEmpty(); }),
748 allboxes.end());
749 BoxList bl(std::move(allboxes));
750 return std::make_pair(BoxArray(std::move(bl)), DistributionMapping(std::move(pmap)));
751#else
752 return std::make_pair(BoxArray(bx), DistributionMapping(Vector<int>({0})));
753#endif
754}
