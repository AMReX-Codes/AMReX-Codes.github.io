26AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
27int coarsen (int i, int ratio) noexcept
28{
29 switch (ratio) {
30 case 1: return i;
31 case 2: return (i<0) ? -std::abs(i+1)/2 -1 : i/2;
32 case 4: return (i<0) ? -std::abs(i+1)/4 -1 : i/4;
33 default: return (i<0) ? -std::abs(i+1)/ratio-1 : i/ratio;
34 }
35}

59 struct shift_hasher {
60 std::size_t operator()(const IntVectND<dim>& vec) const noexcept
61 {
62 static constexpr unsigned shift1 = sizeof(size_t)>=8 ? 20 : 10;
63 static constexpr unsigned shift2 = sizeof(size_t)>=8 ? 40 : 20;
64 if constexpr (dim == 1) {
65 amrex::ignore_unused(shift1);
66 amrex::ignore_unused(shift2);
67 return static_cast<std::size_t>(vec[0]);
68 } else if constexpr (dim == 2) {
69 amrex::ignore_unused(shift2);
70 return static_cast<std::size_t>(vec[0]) ^
71 (static_cast<std::size_t>(vec[1]) << shift1);
72 } else if constexpr (dim == 3) {
73 return static_cast<std::size_t>(vec[0]) ^
74 (static_cast<std::size_t>(vec[1]) << shift1) ^
75 (static_cast<std::size_t>(vec[2]) << shift2);
76 } else {
77 std::size_t seed = dim;
78 // hash function from
79 // https://stackoverflow.com/questions/20511347/a-good-hash-function-for-a-vector
80 for (int i=0; i<dim; ++i) {
81 auto x = static_cast<unsigned int>(vec[i]);
82 x = ((x >> 16) ^ x) * 0x45d9f3b;
83 x = ((x >> 16) ^ x) * 0x45d9f3b;
84 x = (x >> 16) ^ x;
85 seed ^= x + 0x9e3779b9 + (seed << 6) + (seed >> 2);
86 }
87 return seed;
88 }
89 }
90 };

821IntVectND<dim>::coarsen (int s) noexcept
822{
823 BL_ASSERT(s > 0);
824 switch (s) {
825 case 1:
826 break;
827 case 2:
828 for (int i=0; i<dim; ++i) {
829 vect[i] = (vect[i]<0) ? -std::abs(vect[i]+1)/2-1 : vect[i]/2;

831 break;
832 case 4:
833 for (int i=0; i<dim; ++i) {
834 vect[i] = (vect[i]<0) ? -std::abs(vect[i]+1)/4-1 : vect[i]/4;

836 break;
837 default:
838 for (int i=0; i<dim; ++i) {
839 vect[i] = (vect[i]<0) ? -std::abs(vect[i]+1)/s-1 : vect[i]/s;
840 }
