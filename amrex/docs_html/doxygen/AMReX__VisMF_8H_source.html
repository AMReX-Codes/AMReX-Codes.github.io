383 std::enable_if_t<std::is_same<FAB,IArrayBox>::value>
384 Write (const FabArray<FAB>& fa, const std::string& name)
385 {
386  BL_PROFILE("Write(FabArray)");
387  AMREX_ASSERT(name.back() != '/');
388 
389  auto data_descriptor = FAB::getDataDescriptor();
390  int data_bytes = data_descriptor->numBytes();
391 
392  bool useSparseFPP = false;
393  const Vector<int> &pmap = fa.DistributionMap().ProcessorMap();
394  std::set<int> procsWithData;
395  Vector<int> procsWithDataVector;
396  for(int i = 0; i < pmap.size(); ++i) {
397  procsWithData.insert(pmap[i]);
398  }
399  const int nOutFiles = VisMF::GetNOutFiles();
400  if (static_cast<int>(procsWithData.size()) < nOutFiles) {
401  useSparseFPP = true;
402  for (auto i : procsWithData) {
403  procsWithDataVector.push_back(i);
404  }
405  }
406 
407  std::string filePrefix = name + "_D_";
408 
409  NFilesIter nfi(nOutFiles, filePrefix, VisMF::GetGroupSets(), VisMF::GetSetBuf());
410 
411  if (useSparseFPP) {
412  nfi.SetSparseFPP(procsWithDataVector);
413  } else {
414  nfi.SetDynamic();
415  }
416 
417  const auto &fio = FAB::getFABio();
418 
419  for ( ; nfi.ReadyToWrite(); ++nfi) {
420  for(MFIter mfi(fa); mfi.isValid(); ++mfi) {
421  const FAB &fab = fa[mfi];
422  {
423  std::stringstream hss;
424  fio.write_header(hss, fab, fab.nComp());
425  auto hLength = static_cast<std::streamoff>(hss.tellp());
426  auto tstr = hss.str();
427  nfi.Stream().write(tstr.c_str(), hLength);
428  nfi.Stream().flush();
429  }
430  auto const* fabdata = fab.dataPtr();
431 #ifdef AMREX_USE_GPU
432  std::unique_ptr<FAB> hostfab;
433  if (fab.arena()->isManaged() || fab.arena()->isDevice()) {
434  hostfab = std::make_unique<FAB>(fab.box(), fab.nComp(), The_Pinned_Arena());
435  Gpu::dtoh_memcpy_async(hostfab->dataPtr(), fab.dataPtr(),
436  fab.size()*sizeof(typename FAB::value_type));
437  Gpu::streamSynchronize();
438  fabdata = hostfab->dataPtr();
439  }
440 #endif
441  Long writeDataItems = fab.box().numPts() * fa.nComp();
442  Long writeDataSize = writeDataItems * data_bytes;
443  nfi.Stream().write((char *) fabdata, writeDataSize);
444  nfi.Stream().flush();
445  }
446  }
447 
448  int coordinatorProc = ParallelDescriptor::IOProcessorNumber();
449  if (nfi.GetDynamic()) {
450  coordinatorProc = nfi.CoordinatorProc();
451  }
452 
453  if (ParallelDescriptor::MyProc() == coordinatorProc) {
454  std::string header_file_name = name + "_H";
455  VisMFBuffer::IO_Buffer io_buffer(VisMFBuffer::GetIOBufferSize());
456  std::ofstream ofs;
457  ofs.rdbuf()->pubsetbuf(io_buffer.dataPtr(), io_buffer.size());
458  ofs.open(header_file_name.c_str(), std::ios::out | std::ios::trunc);
459  if (!ofs.good()) {
460  amrex::FileOpenFailed(header_file_name);
461  }
462 
463  ofs << "amrex::FabArray<" << FAB::getClassName() << "> v1.0\n";
464  ofs << fa.nComp() << '\n';
465  ofs << fa.nGrowVect() << '\n';
466  fa.boxArray().writeOn(ofs);
467  ofs << '\n';
468 
469  const DistributionMapping& dm = fa.DistributionMap();
470  int nfabs = fa.boxArray().size();
471  int nFiles = NFilesIter::ActualNFiles(nOutFiles);
472  int nprocs = ParallelDescriptor::NProcs();
473 
474  Vector<Long> fabBytes(nfabs, 0);
475  std::map<int, Vector<int> > rankBoxOrder;
476  for (int i = 0; i < nfabs; ++i) {
477  std::stringstream hss;
478  FAB tmp(fa.fabbox(i), fa.nComp(), false);
479  fio.write_header(hss, tmp, tmp.nComp());
480  // Size includes header and data
481  fabBytes[i] = static_cast<std::streamoff>(hss.tellp()) + tmp.size() * data_bytes;
482  rankBoxOrder[dm[i]].push_back(i);
483  }
484 
485  Vector<int> fileNumbers;
486  if (nfi.GetDynamic()) {
487  fileNumbers = nfi.FileNumbersWritten();
488  } else if (nfi.GetSparseFPP()) {
489  // if sparse, write to (file number = rank)
490  fileNumbers.resize(nprocs);
491  std::iota(fileNumbers.begin(), fileNumbers.end(), 0);
492  } else {
493  fileNumbers.resize(nprocs);
494  for (int i = 0; i < nprocs; ++i) {
495  fileNumbers[i] = NFilesIter::FileNumber(nFiles, i, VisMF::GetGroupSets());
496  }
497  }
498 
499  Vector<VisMF::FabOnDisk> fod(nfabs);
500 
501  const Vector< Vector<int> > &fileNumbersWriteOrder = nfi.FileNumbersWriteOrder();
502  for (auto const& rv : fileNumbersWriteOrder) {
503  Long currentOffset = 0;
504  for (auto rank : rv) {
505  auto rbo_it = rankBoxOrder.find(rank);
506  if (rbo_it != rankBoxOrder.end()) {
507  Vector<int> const& index = rbo_it->second;
508  int whichFileNumber = fileNumbers[rank];
509  std::string const& whichFileName =
510  VisMF::BaseName(NFilesIter::FileName(whichFileNumber, filePrefix));
511  for (int i : index) {
512  fod[i].m_name = whichFileName;
513  fod[i].m_head = currentOffset;
514  currentOffset += fabBytes[i];
515  }
516  }
517  }
518  }
519  ofs << fod;
520  }

558 template <typename FAB>
559 // This function does work for MultiFab, but let's disable it to avoid confusion.
560 std::enable_if_t<std::is_same<FAB,IArrayBox>::value>
561 Read (FabArray<FAB>& fa, const std::string& name)
562 {
563  BL_PROFILE("Read(FabArray)");
564  AMREX_ASSERT(name.back() != '/');
565 
566  BoxArray ba;
567  int ncomp;
568  IntVect ngrow;
569  Vector<VisMF::FabOnDisk> fod;
570  {
571  std::string header_file_name = name + "_H";
572  Vector<char> header_file_chars;
573  ParallelDescriptor::ReadAndBcastFile(header_file_name, header_file_chars);
574  std::string header_file_string(header_file_chars.data());
575  std::stringstream ifs(header_file_string, std::istringstream::in);
576 
577  std::string type, version;
578  ifs >> type >> version;
579  AMREX_ASSERT(type == "amrex::FabArray<amrex::IArrayBox>" ||
580  type == "amrex::FabArray<amrex::FArrayBox>");
581  ifs >> ncomp;
582  ifs >> ngrow;
583  ba.readFrom(ifs);
584  ifs >> fod;
585  }
586 
587  if (fa.empty()) {
588  fa.define(ba, DistributionMapping{ba}, ncomp, ngrow);
589  } else {
590  AMREX_ASSERT(amrex::match(ba, fa.boxArray()));
591  }
592 
593 #ifdef AMREX_USE_MPI
594  const int nopensperfile = VisMF::GetMFFileInStreams(); // # of concurrent readers per file
595  const int myproc = ParallelDescriptor::MyProc();
596  const int coordproc = ParallelDescriptor::IOProcessorNumber();
597 
598  int nreqs = 0;
599  int allreadsindex = 0;
600  std::map<std::string, int> filenames; // <filename, allreadsindex>
601 
602  const int nboxes = fa.size();
603  const auto& dm = fa.DistributionMap();
604  for (int i = 0; i < nboxes; ++i) {
605  if (myproc == dm[i]) {
606  ++nreqs;
607  }
608  if (myproc == coordproc) {
609  std::string const& fname = fod[i].m_name;
610  auto r =filenames.insert(std::make_pair(fname, allreadsindex));
611  if (r.second) {
612  ++allreadsindex;
613  }
614  }
615  }
616 
617  const int readtag = ParallelDescriptor::SeqNum();
618  const int donetag = ParallelDescriptor::SeqNum();
619 
620  if (myproc == coordproc) {
621  std::multiset<int> availablefiles; // [whichFile] supports multiple reads/file
622  Vector<std::map<int,std::map<Long,int> > > allreads; // [file]<proc,<seek,index>>
623 
624  const auto nfiles = static_cast<int>(filenames.size());
625  for (int i = 0; i < nfiles; ++i) {
626  for (int j = 0; j < nopensperfile; ++j) {
627  availablefiles.insert(i);
628  }
629  }
630  allreads.resize(nfiles);
631  for (int i = 0; i < nboxes; ++i) {
632  const auto whichproc = dm[i];
633  const auto iseekpos = fod[i].m_head;
634  std::string const& fname = fod[i].m_name;
635  auto filenamesiter = filenames.find(fname);
636  if (filenamesiter != filenames.end()) {
637  const int fi = filenamesiter->second;
638  allreads[fi][whichproc].insert(std::make_pair(iseekpos,i));
639  } else {
640  amrex::Error("Error in amrex::Read: filename not found "+fname);
641  }
642  }
643 
644  int totalioreqs = nboxes;
645  int reqspending = 0;
646  int iopfileindex;
647  std::deque<int> iopreads;
648  std::set<int> busyprocs;
649  while (totalioreqs > 0) {
650  auto afilesiter = availablefiles.begin();
651  while (afilesiter != availablefiles.end()) {
652  const int fi = *afilesiter;
653  if (allreads[fi].empty()) {
654  availablefiles.erase(fi);
655  afilesiter = availablefiles.begin();
656  continue;
657  }
658  auto whichread = allreads[fi].begin();
659  for ( ; whichread != allreads[fi].end(); ++whichread) {
660  const int tryproc = whichread->first;
661  if (busyprocs.find(tryproc) == busyprocs.end()) { // not busy
662  busyprocs.insert(tryproc);
663  Vector<int> vreads;
664  vreads.reserve(whichread->second.size());
665  for (auto const& kv : whichread->second) {
666  vreads.push_back(kv.second);
667  }
668  if (tryproc == coordproc) {
669  iopfileindex = fi;
670  for (auto x : vreads) {
671  iopreads.push_back(x);
672  }
673  } else {
674  ParallelDescriptor::Send(vreads, tryproc, readtag);
675  ++reqspending;
676  }
677  availablefiles.erase(afilesiter);
678  afilesiter = availablefiles.begin();
679  break;
680  }
681  }
682  if (whichread == allreads[fi].end()) {
683  ++afilesiter;
684  } else {
685  allreads[fi].erase(whichread);
686  }
687  }
688 
689  while (!iopreads.empty()) {
690  int i = iopreads.front();
691  detail::read_fab(fa[i], fod[i], name);
692  --totalioreqs;
693  iopreads.pop_front();
694  if (iopreads.empty()) {
695  availablefiles.insert(iopfileindex);
696  busyprocs.erase(coordproc);
697  }
698  int doneflag;
699  MPI_Status status;
700  ParallelDescriptor::IProbe(MPI_ANY_SOURCE, donetag, doneflag, status);
701  if (doneflag) {
702  break;
703  }
704  }
705 
706  if (reqspending > 0) {
707  Vector<int> idone(2);
708  ParallelDescriptor::Message rmess = ParallelDescriptor::Recv(idone, MPI_ANY_SOURCE,
709  donetag);
710  const int i = idone[0];
711  const int procdone = rmess.pid();
712  totalioreqs -= idone[1];
713  --reqspending;
714  busyprocs.erase(procdone);
715  std::string const& fname = fod[i].m_name;
716  const int fi = filenames.find(fname)->second;
717  availablefiles.insert(fi);
718  }
719  }
720  } else {
721  Vector<int> recreads(nreqs, -1);
722  Vector<int> idone(2);
723  while (nreqs > 0) {
724  ParallelDescriptor::Message rmess = ParallelDescriptor::Recv(recreads, coordproc,
725  readtag);
726  const auto nrmess = static_cast<int>(rmess.count());
727  for (int ir = 0; ir < nrmess; ++ir) {
728  int i = recreads[ir];
729  detail::read_fab(fa[i], fod[i], name);
730  }
731  nreqs -= nrmess;
732  idone[0] = recreads[0];
733  idone[1] = nrmess;
734  ParallelDescriptor::Send(idone, coordproc, donetag);
735  }
736  }
737 #else
738  for (MFIter mfi(fa); mfi.isValid(); ++mfi) {
739  detail::read_fab(fa[mfi], fod[mfi.index()], name);
740  }
741 #endif
742 }
