272 parser_call_f1 (enum parser_f1_t type, double a)
273 {
274  switch (type) {
275  case PARSER_SQRT: return std::sqrt(a);
276  case PARSER_EXP: return parser_math_exp<double>(a);
277  case PARSER_LOG: return parser_math_log<double>(a);
278  case PARSER_LOG10: return parser_math_log10<double>(a);
279  case PARSER_SIN: return parser_math_sin<double>(a);
280  case PARSER_COS: return parser_math_cos<double>(a);
281  case PARSER_TAN: return parser_math_tan<double>(a);
282  case PARSER_ASIN: return parser_math_asin<double>(a);
283  case PARSER_ACOS: return parser_math_acos<double>(a);
284  case PARSER_ATAN: return parser_math_atan<double>(a);
285  case PARSER_SINH: return parser_math_sinh<double>(a);
286  case PARSER_COSH: return parser_math_cosh<double>(a);
287  case PARSER_TANH: return parser_math_tanh<double>(a);
288  case PARSER_ABS: return amrex::Math::abs(a);
289  case PARSER_POW_M3: return 1.0/(a*a*a);
290  case PARSER_POW_M2: return 1.0/(a*a);
291  case PARSER_POW_M1: return 1.0/a;
292  case PARSER_POW_P1: return a;
293  case PARSER_POW_P2: return a*a;
294  case PARSER_POW_P3: return a*a*a;
295  default:
296  amrex::Abort("parser_call_f1: Unknown function ");
297  return 0.0;
298  }

301 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE double
302 parser_call_f2 (enum parser_f2_t type, double a, double b)
303 {
304  switch (type) {
305  case PARSER_POW:
306  return parser_math_pow<double>(a,b);
307  case PARSER_GT:
308  return (a > b) ? 1.0 : 0.0;
309  case PARSER_LT:
310  return (a < b) ? 1.0 : 0.0;
311  case PARSER_GEQ:
312  return (a >= b) ? 1.0 : 0.0;
313  case PARSER_LEQ:
314  return (a <= b) ? 1.0 : 0.0;
315  case PARSER_EQ:
316  return (a == b) ? 1.0 : 0.0;
317  case PARSER_NEQ:
318  return (a != b) ? 1.0 : 0.0;
319  case PARSER_AND:
320  return ((a != 0.0) && (b != 0.0)) ? 1.0 : 0.0;
321  case PARSER_OR:
322  return ((a != 0.0) || (b != 0.0)) ? 1.0 : 0.0;
323  case PARSER_HEAVISIDE:
324  return (a < 0.0) ? 0.0 : ((a > 0.0) ? 1.0 : b);
325  case PARSER_JN:
326  return parser_math_jn<double>(int(a),b);
327  case PARSER_MIN:
328  return (a < b) ? a : b;
329  case PARSER_MAX:
330  return (a > b) ? a : b;
331  default:
332  amrex::Abort("parser_call_f2: Unknown function");
333  return 0.0;
334  }
335 }
