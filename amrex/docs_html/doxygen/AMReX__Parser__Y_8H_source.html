405 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE double
406 parser_call_f2 (enum parser_f2_t type, double a, double b)
407 {
408  switch (type) {
409  case PARSER_POW:
410  return parser_math_pow<double>(a,b);
411  case PARSER_ATAN2:
412  return parser_math_atan2<double>(a,b);
413  case PARSER_GT:
414  return (a > b) ? 1.0 : 0.0;
415  case PARSER_LT:
416  return (a < b) ? 1.0 : 0.0;
417  case PARSER_GEQ:
418  return (a >= b) ? 1.0 : 0.0;
419  case PARSER_LEQ:
420  return (a <= b) ? 1.0 : 0.0;
421  case PARSER_EQ:
422  return (a == b) ? 1.0 : 0.0;
423  case PARSER_NEQ:
424  return (a != b) ? 1.0 : 0.0;
425  case PARSER_AND:
426  return ((a != 0.0) && (b != 0.0)) ? 1.0 : 0.0;
427  case PARSER_OR:
428  return ((a != 0.0) || (b != 0.0)) ? 1.0 : 0.0;
429  case PARSER_HEAVISIDE:
430  return (a < 0.0) ? 0.0 : ((a > 0.0) ? 1.0 : b);
431  case PARSER_JN:
432  return parser_math_jn<double>(int(a),b);
433  case PARSER_MIN:
434  return (a < b) ? a : b;
435  case PARSER_MAX:
436  return (a > b) ? a : b;
437  case PARSER_FMOD:
438  return std::fmod(a,b);
439  default:
440  amrex::Abort("parser_call_f2: Unknown function");
441  return 0.0;
442  }
443 }
