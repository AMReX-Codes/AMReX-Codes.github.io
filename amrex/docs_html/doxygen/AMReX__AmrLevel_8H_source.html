613 template <typename F, typename P>
614 void AmrLevel::RK (int order, int state_type, Real time, Real dt, int iteration,
615  int ncycle, F&& f, P&& p)
616 {
617  BL_PROFILE("AmrLevel::RK()");
618 
619  AMREX_ASSERT(AmrLevel::desc_lst[state_type].nExtra() > 0); // Need ghost cells in StateData
620 
621  MultiFab& S_old = get_old_data(state_type);
622  MultiFab& S_new = get_new_data(state_type);
623  const Real t_old = state[state_type].prevTime();
624  const Real t_new = state[state_type].curTime();
625  AMREX_ALWAYS_ASSERT(amrex::almostEqual(time,t_old) && amrex::almostEqual(time+dt,t_new));
626 
627  if (order == 2) {
628  RungeKutta::RK2(S_old, S_new, time, dt, std::forward<F>(f),
629  [&] (int /*stage*/, MultiFab& mf, Real t) {
630  FillPatcherFill(mf, 0, mf.nComp(), mf.nGrow(), t,
631  state_type, 0); },
632  std::forward<P>(p));
633  } else if (order == 3) {
634  RungeKutta::RK3(S_old, S_new, time, dt, std::forward<F>(f),
635  [&] (int stage, MultiFab& mf, Real t) {
636  FillRKPatch(state_type, mf, t, stage, iteration, ncycle);
637  },
638  [&] (Array<MultiFab,3> const& rkk) {
639  if (level < parent->finestLevel()) {
640  storeRKCoarseData(state_type, time, dt, S_old, rkk);
641  }
642  },
643  std::forward<P>(p));
644  } else if (order == 4) {
645  RungeKutta::RK4(S_old, S_new, time, dt, std::forward<F>(f),
646  [&] (int stage, MultiFab& mf, Real t) {
647  FillRKPatch(state_type, mf, t, stage, iteration, ncycle);
648  },
649  [&] (Array<MultiFab,4> const& rkk) {
650  if (level < parent->finestLevel()) {
651  storeRKCoarseData(state_type, time, dt, S_old, rkk);
652  }
653  },
654  std::forward<P>(p));
655  } else {
656  amrex::Abort("AmrLevel::RK: order = "+std::to_string(order)+" is not supported");
657  }
658 }

660 template <std::size_t order>
661 void AmrLevel::storeRKCoarseData (int state_type, Real time, Real dt,
662  MultiFab const& S_old,
663  Array<MultiFab,order> const& rkk)
664 {
665  if (level == parent->finestLevel()) { return; }
666 
667  const StateDescriptor& desc = AmrLevel::desc_lst[state_type];
668 
669  auto& fillpatcher = parent->getLevel(level+1).m_fillpatcher[state_type];
670  fillpatcher = std::make_unique<FillPatcher<MultiFab>>
671  (parent->boxArray(level+1), parent->DistributionMap(level+1),
672  parent->Geom(level+1),
673  parent->boxArray(level), parent->DistributionMap(level),
674  parent->Geom(level),
675  IntVect(desc.nExtra()), desc.nComp(), desc.interp(0));
676 
677  fillpatcher->storeRKCoarseData(time, dt, S_old, rkk);
678 }
