346 void checkOverflow () const {
347 if (ok()) {
348 for (int i = 0; i < dim; ++i) {
349 auto lo = static_cast<Long>(smallend[i]);
350 auto hi = static_cast<Long>(bigend[i]);
351 Long len = hi - lo + 1;
352 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(len>=0 && len<std::numeric_limits<int>::max(),
353 "Overflow when computing length of box");
354 }
355 auto num_pts = static_cast<Long>(length(0));
356 for (int i = 1; i < dim; ++i) {
357 auto len = static_cast<Long>(length(i));
358 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(num_pts == 0 || len == 0 ||
359 num_pts <= std::numeric_limits<Long>::max() / len,
360 "Overflow when computing numPts of box");
361 num_pts *= len;
362 }
363 }
364 }

952 for (int dir = 0; dir < dim; dir++)
953 {
954 if (btype[dir]) {
955 if (bigend[dir]%ref_ratio[dir]) {
956 off.setVal(dir,1);
957 }
958 }
959 }
960 bigend.coarsen(ref_ratio);
961 bigend += off;
962 }
963 else
964 {
965 bigend.coarsen(ref_ratio);
966 }
967 }

1087{
1088 IntVectND<dim> result = smallend;
1089
1090 if constexpr (dim > 1) {
1091 GpuArray<Long, dim-1> mult{};
1092 mult[0] = length(0);
1093 for (int i = 1 ; i < dim-1; ++i) {
1094 mult[i] = mult[i-1] * length(i);

1096 for (int i = dim-1 ; i > 0; --i) {

1099 result[i] += static_cast<int>(idx);
1100 }
1101 }
1102
1103 result[0] += static_cast<int>(offset);
1104
1105 return result;

1172// Modified BoxND is low end, returned BoxND is high end.
1173// If CELL: chop_pnt included in high end.
1174// If NODE: chop_pnt included in both Boxes.
1175//
1176
1177template<int dim>
1178AMREX_GPU_HOST_DEVICE
1179inline
1180BoxND<dim>
1181BoxND<dim>::chop (int dir, int chop_pnt) noexcept
1182{
1183 //
1184 // Define new high end BoxND including chop_pnt.
1185 //
1186 IntVectND<dim> sm(smallend);
1187 IntVectND<dim> bg(bigend);
1188 sm.setVal(dir,chop_pnt);
1189 if (btype[dir])
1190 {
1191 //
1192 // NODE centered BoxND.
1193 //
1194 BL_ASSERT(chop_pnt > smallend[dir] && chop_pnt < bigend[dir]);
1195 //
1196 // Shrink original BoxND to just contain chop_pnt.
1197 //
1198 bigend.setVal(dir,chop_pnt);
1199 }
1200 else
1201 {
1202 //
1203 // CELL centered BoxND.
1204 //
1205 BL_ASSERT(chop_pnt > smallend[dir] && chop_pnt <= bigend[dir]);
1206 //
1207 // Shrink original BoxND to one below chop_pnt.
1208 //
1209 bigend.setVal(dir,chop_pnt-1);
1210 }
1211 return BoxND<dim>(sm,bg,btype);

1215AMREX_GPU_HOST_DEVICE
1216inline
1217BoxND<dim>&
1218BoxND<dim>::shiftHalf (int dir, int num_halfs) noexcept
1219{
1220 const int nbit = (num_halfs<0 ? -num_halfs : num_halfs)%2;
1221 int nshift = num_halfs/2;
1222 //
1223 // Toggle btyp bit if num_halfs is odd.
1224 //
1225 const unsigned int bit_dir = btype[dir];
1226 if (nbit) {
1227 btype.flip(dir);
1228 }
1229 if (num_halfs < 0) {
1230 nshift -= (bit_dir ? nbit : 0);
1231 } else {
1232 nshift += (bit_dir ? 0 : nbit);
1233 }
1234 smallend.shift(dir,nshift);
1235 bigend.shift(dir,nshift);
1236 return *this;
1237}
