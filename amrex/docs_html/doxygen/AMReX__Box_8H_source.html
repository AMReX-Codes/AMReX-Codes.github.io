325 void checkOverflow () const noexcept {
326 if (ok()) {
327 for (int i = 0; i < dim; ++i) {
328 auto lo = static_cast<Long>(smallend[i]);
329 auto hi = static_cast<Long>(bigend[i]);
330 Long len = hi - lo + 1;
331 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(len>=0 && len<std::numeric_limits<int>::max(),
332 "Overflow when computing length of box");
333 }
334 auto num_pts = static_cast<Long>(length(0));
335 for (int i = 1; i < dim; ++i) {
336 auto len = static_cast<Long>(length(i));
337 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(num_pts == 0 || len == 0 ||
338 num_pts <= std::numeric_limits<Long>::max() / len,
339 "Overflow when computing numPts of box");
340 num_pts *= len;
341 }
342 }
343 }

868{
869 if (ref_ratio != 1) {

885{

890 if (btype.any())
891 {
892 IntVectND<dim> off(0);
893 for (int dir = 0; dir < dim; dir++)
894 {
895 if (btype[dir]) {
896 if (bigend[dir]%ref_ratio[dir]) {
897 off.setVal(dir,1);
898 }
899 }
900 }
901 bigend.coarsen(ref_ratio);
902 bigend += off;
903 }
904 else
905 {
906 bigend.coarsen(ref_ratio);
907 }
908 }
909
910 return *this;

1025AMREX_FORCE_INLINE
1026IntVectND<dim>
1027BoxND<dim>::atOffset (Long offset) const noexcept
1028{
1029 IntVectND<dim> result = smallend;
1030
1031 if constexpr (dim > 1) {
1032 GpuArray<Long, dim-1> mult{};
1033 mult[0] = length(0);
1034 for (int i = 1 ; i < dim-1; ++i) {
1035 mult[i] = mult[i-1] * length(i);
1036 }
1037 for (int i = dim-1 ; i > 0; --i) {
1038 Long idx = offset / mult[i-1];
1039 offset -= idx * mult[i-1];
1040 result[i] += static_cast<int>(idx);
1041 }
1042 }
1043
1044 result[0] += static_cast<int>(offset);
1045
1046 return result;
1047}

1111//
1112// Modified BoxND is low end, returned BoxND is high end.
1113// If CELL: chop_pnt included in high end.
1114// If NODE: chop_pnt included in both Boxes.
1115//
1116
1117template<int dim>
1118AMREX_GPU_HOST_DEVICE
1119inline
1120BoxND<dim>
1121BoxND<dim>::chop (int dir, int chop_pnt) noexcept
1122{
1123 //
1124 // Define new high end BoxND including chop_pnt.
1125 //
1126 IntVectND<dim> sm(smallend);
1127 IntVectND<dim> bg(bigend);
1128 sm.setVal(dir,chop_pnt);
1129 if (btype[dir])
1130 {
1131 //
1132 // NODE centered BoxND.
1133 //
1134 BL_ASSERT(chop_pnt > smallend[dir] && chop_pnt < bigend[dir]);
1135 //
1136 // Shrink original BoxND to just contain chop_pnt.
1137 //
1138 bigend.setVal(dir,chop_pnt);
1139 }
1140 else
1141 {
1142 //
1143 // CELL centered BoxND.
1144 //
1145 BL_ASSERT(chop_pnt > smallend[dir] && chop_pnt <= bigend[dir]);
1146 //
1147 // Shrink original BoxND to one below chop_pnt.
1148 //
1149 bigend.setVal(dir,chop_pnt-1);
1150 }
1151 return BoxND<dim>(sm,bg,btype);
1152}

1155AMREX_GPU_HOST_DEVICE
1156inline
1157BoxND<dim>&
1158BoxND<dim>::shiftHalf (int dir, int num_halfs) noexcept
1159{
1160 const int nbit = (num_halfs<0 ? -num_halfs : num_halfs)%2;
1161 int nshift = num_halfs/2;
1162 //
1163 // Toggle btyp bit if num_halfs is odd.
1164 //
1165 const unsigned int bit_dir = btype[dir];
1166 if (nbit) {
1167 btype.flip(dir);
1168 }
1169 if (num_halfs < 0) {
1170 nshift -= (bit_dir ? nbit : 0);
1171 } else {
1172 nshift += (bit_dir ? 0 : nbit);
1173 }
1174 smallend.shift(dir,nshift);
1175 bigend.shift(dir,nshift);
1176 return *this;
1177}
