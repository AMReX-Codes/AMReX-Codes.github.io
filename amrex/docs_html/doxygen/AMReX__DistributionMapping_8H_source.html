427template <typename T>
428void DistributionMapping::ComputeDistributionMappingEfficiency (
429 const DistributionMapping& dm, const std::vector<T>& cost, Real* efficiency)
430{
431 const int nprocs = ParallelDescriptor::NProcs();
432 Vector<T> wgts(nprocs, T(0));
433
434 const auto nboxes = int(dm.size());
435 for (int ibox = 0; ibox < nboxes; ++ibox) {
436 wgts[dm[ibox]] += cost[ibox];
437 }
438
439 T max_weight = 0;
440 T sum_weight = 0;
441 for (auto const& w : wgts) {
442 max_weight = std::max(w, max_weight);
443 sum_weight += w;
444 }
445
446 AMREX_ASSERT(nprocs > 0 && max_weight > T(0));
447
448 *efficiency = static_cast<Real>(sum_weight) /
449 (static_cast<Real>(nprocs) * static_cast<Real>(max_weight));
450}
