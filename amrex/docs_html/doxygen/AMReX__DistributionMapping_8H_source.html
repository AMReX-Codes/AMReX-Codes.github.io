416 template <typename T>
417 void DistributionMapping::ComputeDistributionMappingEfficiency (
418  const DistributionMapping& dm, const std::vector<T>& cost, Real* efficiency)
419 {
420  const int nprocs = ParallelDescriptor::NProcs();
421  Vector<T> wgts(nprocs, T(0));
422 
423  const auto nboxes = int(dm.size());
424  for (int ibox = 0; ibox < nboxes; ++ibox) {
425  wgts[dm[ibox]] += cost[ibox];
426  }
427 
428  T max_weight = 0;
429  T sum_weight = 0;
430  for (auto const& w : wgts) {
431  max_weight = std::max(w, max_weight);
432  sum_weight += w;
433  }
434 
435  *efficiency = static_cast<Real>(sum_weight) /
436  (static_cast<Real>(nprocs) * static_cast<Real>(max_weight));
437 }
