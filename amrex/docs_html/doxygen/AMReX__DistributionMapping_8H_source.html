584template <typename T>
585void DistributionMapping::ComputeDistributionMappingEfficiency (
586 const DistributionMapping& dm, const std::vector<T>& cost, Real* efficiency)
587{
588 const int nprocs = ParallelDescriptor::NProcs();
589 Vector<T> wgts(nprocs, T(0));
590
591 const auto nboxes = int(dm.size());
592 for (int ibox = 0; ibox < nboxes; ++ibox) {
593 wgts[dm[ibox]] += cost[ibox];
594 }
595
596 T max_weight = 0;
597 T sum_weight = 0;
598 for (auto const& w : wgts) {
599 max_weight = std::max(w, max_weight);
600 sum_weight += w;
601 }
602
603 AMREX_ASSERT(nprocs > 0 && max_weight > T(0));
604
605 *efficiency = static_cast<Real>(sum_weight) /
606 (static_cast<Real>(nprocs) * static_cast<Real>(max_weight));
607}
