469bool
470FArrayBox::contains_nan () const noexcept
471{
472 const Real* dp = dptr;
473 const Long n = numPts()*nvar;
474#ifdef AMREX_USE_GPU
475 if (run_on == RunOn::Device && Gpu::inLaunchRegion()) {
476 ReduceOps<ReduceOpLogicalOr> reduce_op;
477 ReduceData<int> reduce_data(reduce_op);
478 using ReduceTuple = ReduceData<int>::Type;
479 reduce_op.eval(n, reduce_data,
480 [=] AMREX_GPU_DEVICE (Long i) -> ReduceTuple
481 {
482 return { static_cast<int>(amrex::isnan(dp[i])) };
483 });
484 ReduceTuple hv = reduce_data.value(reduce_op);
485 return amrex::get<0>(hv);
486 } else
487#endif
488 {
489 for (Long i = 0; i < n; i++) {
490 if (amrex::isnan(*dp++)) {
491 return true;
492 }
493 }
494 return false;
495 }
