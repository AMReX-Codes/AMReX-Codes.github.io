593  const auto Who = RcvProc[i];
594  const auto offset = rOffset[i];
595  const auto Cnt = Rcvs[Who];

599  BL_ASSERT(Who >= 0 && Who < NProcs);

602  Cnt, Who, SeqNum).req();
603  }
604 
605  // Send.
606  for (int i = 0; i < NProcs; ++i)
607  {
608  if (i == MyProc) continue;
609  const auto Who = i;
610  const auto Cnt = Snds[i];
611 
612  BL_ASSERT(Cnt > 0);
613  BL_ASSERT(Who >= 0 && Who < NProcs);
614  BL_ASSERT(Cnt < std::numeric_limits<int>::max());
615

617  Cnt, Who, SeqNum);
618  }
619 
620  if (nrcvs > 0) {
621  ParallelDescriptor::Waitall(rreqs, stats);
622 
623  rcv_box_offsets.push_back(0);
624  for (int i = 0; i < rcv_data.size(); i +=2)
625  {

628  rcv_box_ids.push_back(rcv_data[i+1]);
629  rcv_box_offsets.push_back(rcv_box_offsets.back() + rcv_box_counts.back());

643 
644 #ifdef BL_USE_MPI

649  for (int i = 0; i < local_neighbors.size(); ++i)
650  {
651  int proc = local_neighbors[i].proc_id;
652  if (proc == MyProc) continue;
653  int grid = local_neighbors[i].grid_id;
654  boxes_rcv[proc].push_back(grid);
655  }
656

658  Vector<long> Snds(NProcs, 0), Rcvs(NProcs, 0); // bytes!
659 
660  long NumSnds = 0;
661  for (int i = 0; i < NProcs; ++i)
662  {
663  if (i == MyProc) continue;
664  long nboxes = proc_box_offsets[i+1] - proc_box_offsets[i];
665  long nbytes = 0;
666  for (int j = 0; j < nboxes; ++j)
667  {
668  nbytes += box_counts[proc_box_offsets[i]+j]*sizeof(ParticleType);
669  }
670  Snds[i] = nbytes;
671  NumSnds += nbytes;
672  }
673 
674  ParallelDescriptor::ReduceLongMax(NumSnds);
675 
676  if (NumSnds == 0) return;
677 
678  BL_COMM_PROFILE(BLProfiler::Alltoall, sizeof(long),
679  ParallelDescriptor::MyProc(), BLProfiler::BeforeCall());
680 
681  BL_MPI_REQUIRE( MPI_Alltoall(Snds.dataPtr(),
682  1,
683  ParallelDescriptor::Mpi_typemap<long>::type(),
684  Rcvs.dataPtr(),
685  1,
686  ParallelDescriptor::Mpi_typemap<long>::type(),
687  ParallelDescriptor::Communicator()) );
688 
689  BL_ASSERT(Rcvs[ParallelDescriptor::MyProc()] == 0);
690 
691  BL_COMM_PROFILE(BLProfiler::Alltoall, sizeof(long),
692  ParallelDescriptor::MyProc(), BLProfiler::AfterCall());
693 
694  Vector<int> RcvProc;
695  Vector<std::size_t> rOffset;
696 
697  std::size_t TotRcvBytes = 0;
698  for (int i = 0; i < NProcs; ++i) {
699  if (Rcvs[i] > 0) {
700  RcvProc.push_back(i);
701  rOffset.push_back(TotRcvBytes/sizeof(ParticleType));
702  TotRcvBytes += Rcvs[i];
703  }
704  }
705 
706  const int nrcvs = RcvProc.size();
707  Vector<MPI_Status> stats(nrcvs);
708  Vector<MPI_Request> rreqs(nrcvs);
709 
710  const int SeqNum = ParallelDescriptor::SeqNum();
711 
712  rcv_buffer.resize(TotRcvBytes/sizeof(ParticleType));
713 
714  // Post receives.
715  for (int i = 0; i < nrcvs; ++i) {
716  const auto Who = RcvProc[i];
717  const auto offset = rOffset[i];
718  const auto Cnt = Rcvs[Who];
719 
720  BL_ASSERT(Cnt > 0);
721  BL_ASSERT(Cnt < std::numeric_limits<int>::max());
722  BL_ASSERT(Who >= 0 && Who < NProcs);
723 
724  rreqs[i] = ParallelDescriptor::Arecv((char*) thrust::raw_pointer_cast(&rcv_buffer[offset]),
725  Cnt, Who, SeqNum).req();
726  }
727 
728  // Send.
729  for (int i = 0; i < NProcs; ++i)
730  {
731  if (i == MyProc) continue;
732  const auto Who = i;
733  const auto Cnt = Snds[i];
734 
735  BL_ASSERT(Cnt > 0);
736  BL_ASSERT(Who >= 0 && Who < NProcs);
737  BL_ASSERT(Cnt < std::numeric_limits<int>::max());
738 
739  const auto snd_offset = box_offsets[proc_box_offsets[i]];
740 
741  ParallelDescriptor::Send(
742  (char*) thrust::raw_pointer_cast(&snd_buffer[snd_offset]),
743  Cnt, Who, SeqNum);
744  }
745 
746  if (nrcvs > 0) {
747  ParallelDescriptor::Waitall(rreqs, stats);
748 
749  const int lev = 0;
750  auto& plev = this->GetParticles(lev);
751 
752  auto p_rcv_box_counts = rcv_box_counts.dataPtr();
753  auto p_rcv_box_offsets = rcv_box_offsets.dataPtr();
754  auto p_rcv_box_ids = rcv_box_ids.dataPtr();
755  auto p_rcv_buffer = rcv_buffer.dataPtr();
756 
757  for (int i = 0; i < rcv_box_counts.size(); ++i)
758  {
759  int copy_size = rcv_box_counts[i];
760  int gid = rcv_box_ids[i];
761  int tid = 0;
762  int offset = rcv_box_offsets[i];
763 
764  auto index = std::make_pair(gid, tid);
765 
766  auto& tile = plev[index];
767  auto& aos = tile.GetArrayOfStructs();
768  auto np = tile.numTotalParticles();
769 
770  AMREX_ALWAYS_ASSERT(MyProc == this->ParticleDistributionMap(0)[gid]);
771 
772  int nn = tile.getNumNeighbors();
773  tile.setNumNeighbors(nn + rcv_box_counts[i]);
774 
775  ParticleType* p_ptr = &(aos[0]);
776  AMREX_FOR_1D ( copy_size, ip,
777  {
778  int src_index = offset + ip;
779  int dst_index = np + ip;
780  p_ptr[dst_index] = p_rcv_buffer[src_index];
781  });
782  }
783  }
784 #endif // MPI
