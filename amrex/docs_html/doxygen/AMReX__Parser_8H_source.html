141template <int N>
142ParserExecutor<N>
143Parser::compileHost () const
144{
145 if (m_data && m_data->m_parser) {
146 AMREX_ALWAYS_ASSERT(N == m_data->m_nvars);
147
148 // Make sure all user functions have been registered.
149 for (auto const& [ufname, nargs] : m_ufs) {
150 if (m_uf_ptrs.find(ufname) == m_uf_ptrs.cend()) {
151 amrex::Abort("amrex::Parser: Unknown user function "+ufname
152 + " in " + m_data->m_expression);
153 }
154 amrex::ignore_unused(nargs);
155 }
156
157 if (!(m_data->m_host_executor)) {
158 int stack_size;
159 m_data->m_exe_size = static_cast<int>
160 (parser_exe_size(m_data->m_parser, m_data->m_max_stack_size,
161 stack_size));
162
163 if (m_data->m_max_stack_size > AMREX_PARSER_STACK_SIZE) {
164 amrex::Abort("amrex::Parser: AMREX_PARSER_STACK_SIZE, "
165 + std::to_string(AMREX_PARSER_STACK_SIZE) + ", is too small for "
166 + m_data->m_expression);
167 }
168 if (stack_size != 0) {
169 amrex::Abort("amrex::Parser: something went wrong with parser stack! "
170 + std::to_string(stack_size));
171 }
172
173 m_data->m_host_executor = (char*)The_Pinned_Arena()->alloc(m_data->m_exe_size);
174 if (m_data->m_host_executor == nullptr) { // Arena is not ready yet
175 m_data->m_host_executor = (char*) std::malloc(m_data->m_exe_size);
176 m_data->m_use_arena = false;
177 }
178
179 try {
180 m_data->m_locals = parser_compile(m_data->m_parser, m_uf_ptrs,
181 m_data->m_host_executor);
182 } catch (const std::runtime_error& e) {
183 throw std::runtime_error(std::string(e.what()) + " in Parser expression \""
184 + m_data->m_expression + "\"");
185 }
186 }
187
188#ifdef AMREX_USE_GPU
189 return ParserExecutor<N>{m_data->m_host_executor, m_data->m_device_executor};
190#else
191 return ParserExecutor<N>{m_data->m_host_executor};
192#endif
193 } else {
194 return ParserExecutor<N>{};
195 }
196}

198template <int N>
199ParserExecutor<N>
200Parser::compile () const
201{
202 auto exe = compileHost<N>();
203
204#ifdef AMREX_USE_GPU
205 if (m_data && m_data->m_parser && !(m_data->m_device_executor)
206 && m_data->m_use_arena)
207 {
208 m_data->m_device_executor = (char*)The_Arena()->alloc(m_data->m_exe_size);
209 Gpu::htod_memcpy_async(m_data->m_device_executor, m_data->m_host_executor,
210 m_data->m_exe_size);
211 Gpu::streamSynchronize();
212 exe.m_device_executor = m_data->m_device_executor;
213 }
214#endif
215
216 return exe;
217}
