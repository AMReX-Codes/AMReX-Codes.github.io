83  template <typename R, typename I, typename F>
84  AMREX_GPU_DEVICE AMREX_FORCE_INLINE
85  R atomic_op (R* const address, R const val, F const f) noexcept
86  {
87  static_assert(sizeof(R) == sizeof(I), "sizeof R != sizeof I");
88  I* const add_as_I = reinterpret_cast<I*>(address);
89  I old_I = *add_as_I, assumed_I;
90  do {
91  assumed_I = old_I;
92  R const new_R = f(*(reinterpret_cast<R const*>(&assumed_I)), val);
93  old_I = atomicCAS(add_as_I, assumed_I, *(reinterpret_cast<I const*>(&new_R)));
94  } while (assumed_I != old_I);
95  return *(reinterpret_cast<R const*>(&old_I));
96  }

100  bool atomic_op_if (R* const address, R const val, Op&& op, Cond&& cond) noexcept
101  {
102  static_assert(sizeof(R) == sizeof(I), "sizeof R != sizeof I");
103  I* const add_as_I = reinterpret_cast<I*>(address);
104  I old_I = *add_as_I, assumed_I;
105  bool test_success;
106  do {
107  assumed_I = old_I;
