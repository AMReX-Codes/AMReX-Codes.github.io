466template <class T>
467std::ostream& amrex::ToString(std::ostream& os, const T& t, const char* symbol_begin,
468 const char* symbol_delim, const char* symbol_end,
469 const char* symbol_str, int limit)
470{
471 if constexpr (std::is_same_v<std::string, T> ||
472 std::is_same_v<std::string_view, T> ||
473 std::is_same_v<char*, std::decay_t<T>> ||
474 std::is_same_v<const char*, T>) {
475 // String-like types
476 os << symbol_str << t << symbol_str;
477 } else if constexpr (std::is_same_v<signed char, T> ||
478 std::is_same_v<unsigned char, T>) {
479 // Buffer-like types, don't print as char
480 os << static_cast<int>(t);
481 } else if constexpr (std::is_pointer_v<T>) {
482 // Pointer types, don't print char* as string
483 os << static_cast<const void*>(t);
484 } else if constexpr (detail::HasBeginEnd<T>() || std::is_array_v<T>) {
485 // Array-like types
486 os << symbol_begin;
487 Long count = 0;
488 for (auto& value : t) {
489 if (count > 0) {
490 os << symbol_delim;
491 }
492 if (count < limit || limit < 0) {
493 ToString(os, value, symbol_begin, symbol_delim, symbol_end,
494 symbol_str, limit);
495 } else if (count == limit) {
496 os << "...";
497 break;
498 }
499 ++count;
500 }
501 os << symbol_end;
502 } else if constexpr (detail::HasTupleSize<T>()) {
503 // Tuple-like types
504 detail::ToStringTupleImp(std::make_index_sequence<std::tuple_size_v<T>>{},
505 os, t, symbol_begin, symbol_delim, symbol_end,
506 symbol_str, limit);
507 } else {
508 // Scalar types
509 os << t;
510 }
511 return os;
512}
