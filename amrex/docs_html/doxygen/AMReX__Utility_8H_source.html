461template <class T>
462std::ostream& amrex::ToString(std::ostream& os, const T& t, const char* symbol_begin,
463 const char* symbol_delim, const char* symbol_end,
464 const char* symbol_str, int limit)
465{
466 if constexpr (std::is_same_v<std::string, T> ||
467 std::is_same_v<std::string_view, T> ||
468 std::is_same_v<char*, std::decay_t<T>> ||
469 std::is_same_v<const char*, T>) {
470 // String-like types
471 os << symbol_str << t << symbol_str;
472 } else if constexpr (std::is_same_v<signed char, T> ||
473 std::is_same_v<unsigned char, T>) {
474 // Buffer-like types, don't print as char
475 os << static_cast<int>(t);
476 } else if constexpr (std::is_pointer_v<T>) {
477 // Pointer types, don't print char* as string
478 os << static_cast<const void*>(t);
479 } else if constexpr (detail::HasBeginEnd<T>() || std::is_array_v<T>) {
480 // Array-like types
481 os << symbol_begin;
482 Long count = 0;
483 for (auto& value : t) {
484 if (count > 0) {
485 os << symbol_delim;
486 }
487 if (count < limit || limit < 0) {
488 ToString(os, value, symbol_begin, symbol_delim, symbol_end,
489 symbol_str, limit);
490 } else if (count == limit) {
491 os << "...";
492 break;
493 }
494 ++count;
495 }
496 os << symbol_end;
497 } else if constexpr (detail::HasTupleSize<T>()) {
498 // Tuple-like types
499 detail::ToStringTupleImp(std::make_index_sequence<std::tuple_size_v<T>>{},
500 os, t, symbol_begin, symbol_delim, symbol_end,
501 symbol_str, limit);
502 } else {
503 // Scalar types
504 os << t;
505 }
506 return os;
507}
