830 template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
831 void
832 ParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>
833 ::WriteParticles (int lev, std::ofstream& ofs, int fnum,
834  Vector<int>& which, Vector<int>& count, Vector<Long>& where,
835  const Vector<int>& write_real_comp,
836  const Vector<int>& write_int_comp,
837  const Vector<std::map<std::pair<int, int>, Gpu::DeviceVector<int>>>& particle_io_flags) const
838 {
839  BL_PROFILE("ParticleContainer::WriteParticles()");
840 
841  // For a each grid, the tiles it contains
842  std::map<int, Vector<int> > tile_map;
843 
844  for (const auto& kv : m_particles[lev])
845  {
846  const int grid = kv.first.first;
847  const int tile = kv.first.second;
848  tile_map[grid].push_back(tile);
849  const auto& pflags = particle_io_flags[lev].at(kv.first);
850 
851  // Only write out valid particles.
852  int cnt = 0;
853  for (int k = 0; k < kv.second.GetArrayOfStructs().numParticles(); ++k)
854  {
855  if (pflags[k]) cnt++;
856  }
857 
858  count[grid] += cnt;
859  }
860 
861  MFInfo info;
862  info.SetAlloc(false);
863  MultiFab state(ParticleBoxArray(lev), ParticleDistributionMap(lev), 1,0,info);
864 
865  for (MFIter mfi(state); mfi.isValid(); ++mfi)
866  {
867  const int grid = mfi.index();
868 
869  which[grid] = fnum;
870  where[grid] = VisMF::FileOffset(ofs);
871 
872  if (count[grid] == 0) continue;
873 
874  // First write out the integer data in binary.
875  int num_output_int = 0;
876  for (int i = 0; i < NumIntComps() + NStructInt; ++i)
877  if (write_int_comp[i]) ++num_output_int;
878 
879  const int iChunkSize = 2 + num_output_int;
880  Vector<int> istuff(count[grid]*iChunkSize);
881  int* iptr = istuff.dataPtr();
882 
883  for (unsigned i = 0; i < tile_map[grid].size(); i++) {
884  auto ptile_index = std::make_pair(grid, tile_map[grid][i]);
885  const auto& pbox = m_particles[lev].at(ptile_index);
886  const auto& pflags = particle_io_flags[lev].at(ptile_index);
887  for (int pindex = 0; pindex < pbox.GetArrayOfStructs().numParticles(); ++pindex) {
888  const auto& aos = pbox.GetArrayOfStructs();
889  const auto& p = aos[pindex];
890  if (pflags[pindex])
891  {
892  // always write these
893  for (int j = 0; j < 2; j++) iptr[j] = p.m_idata.arr[j];
894  iptr += 2;
895 
896  // optionally write these
897  for (int j = 0; j < NStructInt; j++)
898  {
899  if (write_int_comp[j])
900  {
901  *iptr = p.m_idata.arr[2+j];
902  ++iptr;
903  }
904  }
905 
906  const auto& soa = pbox.GetStructOfArrays();
907  for (int j = 0; j < NumIntComps(); j++)
908  {
909  if (write_int_comp[NStructInt+j])
910  {
911  *iptr = soa.GetIntData(j)[pindex];
912  ++iptr;
913  }
914  }
915  }
916  }
917  }
918 
919  writeIntData(istuff.dataPtr(), istuff.size(), ofs);
920  ofs.flush(); // Some systems require this flush() (probably due to a bug)
921 
922  // Write the Real data in binary.
923  int num_output_real = 0;
924  for (int i = 0; i < NumRealComps() + NStructReal; ++i)
925  if (write_real_comp[i]) ++num_output_real;
926 
927  const int rChunkSize = AMREX_SPACEDIM + num_output_real;
928  Vector<typename ParticleType::RealType> rstuff(count[grid]*rChunkSize);
929  typename ParticleType::RealType* rptr = rstuff.dataPtr();
930 
931  for (unsigned i = 0; i < tile_map[grid].size(); i++) {
932  auto ptile_index = std::make_pair(grid, tile_map[grid][i]);
933  const auto& pbox = m_particles[lev].at(ptile_index);
934  const auto& pflags = particle_io_flags[lev].at(ptile_index);
935  for (int pindex = 0; pindex < pbox.GetArrayOfStructs().numParticles(); ++pindex) {
936  const auto& aos = pbox.GetArrayOfStructs();
937  const auto& p = aos[pindex];
938  if (pflags[pindex])
939  {
940  // always write these
941  for (int j = 0; j < AMREX_SPACEDIM; j++) rptr[j] = p.m_rdata.arr[j];
942  rptr += AMREX_SPACEDIM;
943 
944  // optionally write these
945  for (int j = 0; j < NStructReal; j++)
946  {
947  if (write_real_comp[j])
948  {
949  *rptr = p.m_rdata.arr[AMREX_SPACEDIM+j];
950  ++rptr;
951  }
952  }
953 
954  const auto& soa = pbox.GetStructOfArrays();
955  for (int j = 0; j < NumRealComps(); j++)
956  {
957  if (write_real_comp[NStructReal+j])
958  {
959  *rptr = (typename ParticleType::RealType) soa.GetRealData(j)[pindex];
960  ++rptr;
961  }
962  }
963  }
964  }
965  }
966 
967  WriteParticleRealData(rstuff.dataPtr(), rstuff.size(), ofs, ParticleRealDescriptor);
968  ofs.flush(); // Some systems require this flush() (probably due to a bug)
969  }
970 }

1015  // "Version_One_Dot_Zero" -- hard-wired to write out in double precision.
1016  // "Version_One_Dot_One" -- can write out either as either single or double precision.
1017  // Appended to the latter version string are either "_single" or "_double" to
1018  // indicate how the particles were written.
1019  // "Version_Two_Dot_Zero" -- this is the AMReX particle file format
1020  std::string how;
1021  if (version.find("Version_One_Dot_Zero") != std::string::npos) {
1022  how = "double";
1023  }
1024  else if (version.find("Version_One_Dot_One") != std::string::npos or
1025  version.find("Version_Two_Dot_Zero") != std::string::npos) {
1026  if (version.find("_single") != std::string::npos) {
1027  how = "single";
1028  }
1029  else if (version.find("_double") != std::string::npos) {
1030  how = "double";
1031  }

1033  std::string msg("ParticleContainer::Restart(): bad version string: ");
1034  msg += version;
1035  amrex::Error(version.c_str());
1036  }
1037  }
1038  else {
1039  std::string msg("ParticleContainer::Restart(): unknown version string: ");
1040  msg += version;
1041  amrex::Abort(msg.c_str());
1042  }
1043 
1044  int dm;
1045  HdrFile >> dm;
1046  if (dm != AMREX_SPACEDIM)
1047  amrex::Abort("ParticleContainer::Restart(): dm != AMREX_SPACEDIM");
1048 
1049  int nr;
1050  HdrFile >> nr;
1051  if (nr != NStructReal + NumRealComps())
1052  amrex::Abort("ParticleContainer::Restart(): nr != NStructReal + NumRealComps()");
1053 
1054  std::string comp_name;
1055  for (int i = 0; i < nr; ++i)
1056  HdrFile >> comp_name;
1057

1059  HdrFile >> ni;
1060  if (ni != NStructInt + NumIntComps())
1061  amrex::Abort("ParticleContainer::Restart(): ni != NStructInt");
1062 
1063  for (int i = 0; i < ni; ++i)
1064  HdrFile >> comp_name;
1065 
1066  bool checkpoint;
1067  HdrFile >> checkpoint;
1068 
1069  Long nparticles;
1070  HdrFile >> nparticles;
1071  AMREX_ASSERT(nparticles >= 0);
1072 
1073  int maxnextid;
1074  HdrFile >> maxnextid;
1075  AMREX_ASSERT(maxnextid > 0);
1076  ParticleType::NextID(maxnextid);
1077 
1078  int finest_level_in_file;
1079  HdrFile >> finest_level_in_file;
1080  AMREX_ASSERT(finest_level_in_file >= 0);
1081 
1082  // Determine whether this is a dual-grid restart or not.
1083  Vector<BoxArray> particle_box_arrays(finest_level_in_file + 1);
1084  bool dual_grid = false;

1087  for (int lev = 0; lev <= finest_level_in_file; lev++)
1088  {

1091  phdr_name += "/Particle_H";
1092

1094  have_pheaders = true;
1095  break;
1096  }
1097  }
1098 
1099  if (have_pheaders)
1100  {
1101  for (int lev = 0; lev <= finest_level_in_file; lev++)
1102  {
1103  std::string phdr_name = fullname;
1104  phdr_name = amrex::Concatenate(phdr_name + "/Level_", lev, 1);
1105  phdr_name += "/Particle_H";
1106 
1107  if (not amrex::FileExists(phdr_name)) continue;
1108

1110  ParallelDescriptor::ReadAndBcastFile(phdr_name, phdr_chars);
1111  std::string phdr_string(phdr_chars.dataPtr());
1112  std::istringstream phdr_file(phdr_string, std::istringstream::in);
1113 
1114  if (lev > finestLevel())
1115  {
1116  dual_grid = true;
1117  break;
1118  }
1119 
1120  particle_box_arrays[lev].readFrom(phdr_file);
1121  if (not particle_box_arrays[lev].CellEqual(ParticleBoxArray(lev))) dual_grid = true;
1122  }
1123  } else // if no particle box array information exists in the file, we assume a single grid restart
1124  {
1125  dual_grid = false;
1126  }
1127 
1128  if (dual_grid) {
1129  for (int lev = 0; lev <= finestLevel(); lev++) {
1130  SetParticleBoxArray(lev, particle_box_arrays[lev]);
1131  DistributionMapping pdm(particle_box_arrays[lev]);
1132  SetParticleDistributionMap(lev, pdm);
1133  }
1134  }
1135

1137  for (int lev = 0; lev <= finest_level_in_file; lev++) {
1138  HdrFile >> ngrids[lev];
1139  AMREX_ASSERT(ngrids[lev] > 0);
1140  if (lev <= finestLevel()) {
1141  AMREX_ASSERT(ngrids[lev] == int(ParticleBoxArray(lev).size()));
1142  }
1143  }
1144 
1145  resizeData();
1146 
1147  if (finest_level_in_file > finestLevel()) {
1148  m_particles.resize(finest_level_in_file+1);
1149  }

1152  Vector<int> which(ngrids[lev]);
1153  Vector<int> count(ngrids[lev]);
1154  Vector<Long> where(ngrids[lev]);

1157  }

1161  for (MFIter mfi(*m_dummy_mf[lev]); mfi.isValid(); ++mfi) {
1162  grids_to_read.push_back(mfi.index());
1163  }
1164  } else {

1167  // on finer levels, and put them in the right place via Redistribute()

1170  const int NReaders = ParticleType::MaxReaders();

1175

1177  if (rank < Nleft) {
1178  lo = rank*(Navg + 1);
1179  hi = lo + Navg + 1;
1180  }
1181  else {
1182  lo = rank * Navg + Nleft;
1183  hi = lo + Navg;
1184  }
1185

1187  grids_to_read.push_back(i);
1188  }
1189  }
1190

1192  const int grid = grids_to_read[igrid];

1198

1200  name += '/';

1204  name += '/';

1206  name += amrex::Concatenate("", which[grid], DATA_Digits_Read);

1218  ReadParticles<float>(count[grid], grid, lev, ParticleFile, finest_level_in_file);
1219  }
1220  else if (how == "double") {
1221  ReadParticles<double>(count[grid], grid, lev, ParticleFile, finest_level_in_file);
1222  }
1223  else {
1224  std::string msg("ParticleContainer::Restart(): bad parameter: ");
1225  msg += how;
1226  amrex::Error(msg.c_str());
1227  }
1228 
1229  ParticleFile.close();
1230 
1231  if (!ParticleFile.good())
1232  amrex::Abort("ParticleContainer::Restart(): problem reading particles");
1233  }
1234  }
1235

1237 
1238  AMREX_ASSERT(OK());
1239
