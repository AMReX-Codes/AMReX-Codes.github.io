624  Box box = pti.tilebox();
625  box.coarsen(ref_fac);
626  box.grow(m_num_neighbor_cells+1); // need an extra cell to account for roundoff errors.
627  head.resize(box);
628  head.setVal(-1);
629  list.resize(N, -1);
630 
631  for (int i = 0; i < N; ++i) {
632  const ParticleType& p = tmp_particles[i];
633  const IntVect& cell = this->Index(p, 0); // we always bin on level 0
634  cells[i] = cell;
635  list[i] = head(cell);
636  head(cell) = i;
637  }
638 
639  // using these lists, we build a neighbor list containing both
640  // kinds of particles.
641  int p_start_index = 0;
642  for (unsigned i = 0; i < Np; ++i) {
643  const ParticleType& p = tmp_particles[i];
644 
645  int num_neighbors = 0;
646  nl.push_back(0);
647 
648  const IntVect& cell = cells[i];
649  Box bx(cell, cell);
650  bx.grow(m_num_neighbor_cells);
651 
652  for (IntVect iv = bx.smallEnd(); iv <= bx.bigEnd(); bx.next(iv)) {
653  int j = head(iv);
654  while (j >= 0) {
655  if (i == j) {
656  j = list[j];
657  continue;
658  }
659  if ( check_pair(p, tmp_particles[j]) ) {
660  nl.push_back(j+1);
661  num_neighbors += 1;
662  }
663  j = list[j];
664  }
665  }
666 
667  nl[p_start_index] = num_neighbors;
668  p_start_index += num_neighbors + 1;
669  }
670 
671  if (sort) {
672  for (unsigned i = 0; i < nl.size(); i += nl[i] +1) {
673 #ifdef AMREX_USE_CUDA
674  thrust::sort(nl.begin() + i + 1,
675  nl.begin() + nl[i] + i + 1);
676 #else
677  std::sort(nl.begin() + i + 1,
678  nl.begin() + nl[i] + i + 1);
679 #endif
680  }
681  }
682 #ifdef AMREX_USE_CUDA
683  neighbor_list[lev][index].resize(nl.size());
684  thrust::copy(nl.begin(), nl.end(), neighbor_list[lev][index].begin());
685 #endif
686  }
687  }
688  }
689 }
