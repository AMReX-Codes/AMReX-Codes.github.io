85 MLMGBndryT<MF>::setBoxBC (RealTuple& bloc, BCTuple& bctag,

92  const Array<Real,AMREX_SPACEDIM>& domain_bloc_hi,
93  const GpuArray<int,AMREX_SPACEDIM>& is_periodic)
94 {

96 
97  for (OrientationIter fi; fi; ++fi)
98  {
99  const Orientation face = fi();
100  const int dir = face.coordDir();
101 
102  if (domain[face] == bx[face] && !is_periodic[dir])
103  {
104  // All physical bc values are located on face.
105  bloc[face] = static_cast<T>
106  (face.isLow() ? domain_bloc_lo[dir] : domain_bloc_hi[dir]);
107  const auto linop_bc = face.isLow() ? lo[dir] : hi[dir];
108  if (linop_bc == LinOpBCType::Dirichlet) {
109  bctag[face] = AMREX_LO_DIRICHLET;
110  } else if (linop_bc == LinOpBCType::Neumann) {
111  bctag[face] = AMREX_LO_NEUMANN;
112  } else if (linop_bc == LinOpBCType::reflect_odd) {
113  bctag[face] = AMREX_LO_REFLECT_ODD;
114  } else {
115  amrex::Abort("MLMGBndry::setBoxBC: Unknown LinOpBCType");
116  }
117  }
118  else
119  {
120  // Internal bndry.
121  bctag[face] = AMREX_LO_DIRICHLET;
122  bloc[face] = static_cast<T>(ratio > 0 ? Real(0.5)*ratio*dx[dir]
123  : interior_bloc[dir]);
124  // If this is next to another same level box, bloc is
125  // wrong. But it doesn't matter, because we also have
126  // mask. It is used only if mask says it is next to

128  }
