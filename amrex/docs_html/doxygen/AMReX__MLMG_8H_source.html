328  RT a_tol_rel, RT a_tol_abs, const char* checkpoint_file) -> RT
329 {
330  BL_PROFILE("MLMG::solve()");
331 
332  if constexpr (std::is_same<AMF,MultiFab>()) {
333  if (checkpoint_file != nullptr) {
334  checkPoint(a_sol, a_rhs, a_tol_rel, a_tol_abs, checkpoint_file);
335  }
336  }
337 
338  if (bottom_solver == BottomSolver::Default) {
339  bottom_solver = linop.getDefaultBottomSolver();
340  }
341 
342 #if (defined(AMREX_USE_HYPRE) || defined(AMREX_USE_PETSC)) && (AMREX_SPACEDIM > 1)
343  if (bottom_solver == BottomSolver::hypre || bottom_solver == BottomSolver::petsc) {
344  int mo = linop.getMaxOrder();
345  if (a_sol[0]->hasEBFabFactory()) {
346  linop.setMaxOrder(2);
347  } else {
348  linop.setMaxOrder(std::min(3,mo)); // maxorder = 4 not supported
349  }
350  }
351 #endif
352 
353  bool is_nsolve = linop.m_parent;
354 
355  auto solve_start_time = amrex::second();
356 
357  RT& composite_norminf = m_final_resnorm0;
358 
359  m_niters_cg.clear();
360  m_iter_fine_resnorm0.clear();
361 
362  prepareForSolve(a_sol, a_rhs);
363 
364  computeMLResidual(finest_amr_lev);
365 
366  bool local = true;
367  RT resnorm0 = MLResNormInf(finest_amr_lev, local);
368  RT rhsnorm0 = MLRhsNormInf(local);
369  if (!is_nsolve) {
370  ParallelAllReduce::Max<RT>({resnorm0, rhsnorm0}, ParallelContext::CommunicatorSub());
371 
372  if (verbose >= 1)
373  {
374  amrex::Print() << "MLMG: Initial rhs = " << rhsnorm0 << "\n"
375  << "MLMG: Initial residual (resid0) = " << resnorm0 << "\n";
376  }
377  }
378 
379  m_init_resnorm0 = resnorm0;
380  m_rhsnorm0 = rhsnorm0;
381 
382  RT max_norm;
383  std::string norm_name;
384  if (always_use_bnorm || rhsnorm0 >= resnorm0) {
385  norm_name = "bnorm";
386  max_norm = rhsnorm0;
387  } else {
388  norm_name = "resid0";
389  max_norm = resnorm0;
390  }
391  const RT res_target = std::max(a_tol_abs, std::max(a_tol_rel,RT(1.e-16))*max_norm);
392 
393  if (!is_nsolve && resnorm0 <= res_target) {
394  composite_norminf = resnorm0;
395  if (verbose >= 1) {
396  amrex::Print() << "MLMG: No iterations needed\n";
397  }
398  } else {
399  auto iter_start_time = amrex::second();
400  bool converged = false;
401 
402  const int niters = do_fixed_number_of_iters ? do_fixed_number_of_iters : max_iters;
403  for (int iter = 0; iter < niters; ++iter)
404  {
405  oneIter(iter);
406 
407  converged = false;
408 
409  // Test convergence on the fine amr level
410  computeResidual(finest_amr_lev);
411 
412  if (is_nsolve) continue;
413 
414  RT fine_norminf = ResNormInf(finest_amr_lev);
415  m_iter_fine_resnorm0.push_back(fine_norminf);
416  composite_norminf = fine_norminf;
417  if (verbose >= 2) {
418  amrex::Print() << "MLMG: Iteration " << std::setw(3) << iter+1 << " Fine resid/"
419  << norm_name << " = " << fine_norminf/max_norm << "\n";
420  }
421  bool fine_converged = (fine_norminf <= res_target);
422 
423  if (namrlevs == 1 && fine_converged) {
424  converged = true;
425  } else if (fine_converged) {
426  // finest level is converged, but we still need to test the coarse levels
427  computeMLResidual(finest_amr_lev-1);
428  RT crse_norminf = MLResNormInf(finest_amr_lev-1);
429  if (verbose >= 2) {
430  amrex::Print() << "MLMG: Iteration " << std::setw(3) << iter+1
431  << " Crse resid/" << norm_name << " = "
432  << crse_norminf/max_norm << "\n";
433  }
434  converged = (crse_norminf <= res_target);
435  composite_norminf = std::max(fine_norminf, crse_norminf);
436  } else {
437  converged = false;
438  }
439 
440  if (converged) {
441  if (verbose >= 1) {
442  amrex::Print() << "MLMG: Final Iter. " << iter+1
443  << " resid, resid/" << norm_name << " = "
444  << composite_norminf << ", "
445  << composite_norminf/max_norm << "\n";
446  }
447  break;
448  } else {
449  if (composite_norminf > RT(1.e20)*max_norm)
450  {
451  if (verbose > 0) {
452  amrex::Print() << "MLMG: Failing to converge after " << iter+1 << " iterations."
453  << " resid, resid/" << norm_name << " = "
454  << composite_norminf << ", "
455  << composite_norminf/max_norm << "\n";
456  }
457  amrex::Abort("MLMG failing so lets stop here");
458  }
459  }
460  }
461 
462  if (!converged && do_fixed_number_of_iters == 0) {
463  if (verbose > 0) {
464  amrex::Print() << "MLMG: Failed to converge after " << max_iters << " iterations."
465  << " resid, resid/" << norm_name << " = "
466  << composite_norminf << ", "
467  << composite_norminf/max_norm << "\n";
468  }
469  amrex::Abort("MLMG failed");
470  }
471  timer[iter_time] = amrex::second() - iter_start_time;
472  }
473 
474  linop.postSolve(sol);
475 
476  IntVect ng_back = final_fill_bc ? IntVect(1) : IntVect(0);
477  if (linop.hasHiddenDimension()) {
478  ng_back[linop.hiddenDirection()] = 0;
479  }
480  for (int alev = 0; alev < namrlevs; ++alev)
481  {
482  if (!sol_is_alias[alev]) {
483  a_sol[alev]->LocalCopy(sol[alev], 0, 0, ncomp, ng_back);
484  }
485  }
486 
487  timer[solve_time] = amrex::second() - solve_start_time;
488  if (verbose >= 1) {
489  ParallelReduce::Max<double>(timer.data(), timer.size(), 0,
490  ParallelContext::CommunicatorSub());
491  if (ParallelContext::MyProcSub() == 0)
492  {
493  amrex::AllPrint() << "MLMG: Timers: Solve = " << timer[solve_time]
494  << " Iter = " << timer[iter_time]
495  << " Bottom = " << timer[bottom_time] << "\n";
496  }
497  }
498 
499  ++solve_called;
500 
501  return composite_norminf;
502 }
