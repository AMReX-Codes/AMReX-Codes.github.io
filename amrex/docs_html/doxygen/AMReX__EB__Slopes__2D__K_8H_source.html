12template <int N, typename DU>
13AMREX_GPU_DEVICE AMREX_FORCE_INLINE
14GpuArray<Real,AMREX_SPACEDIM>
15calc_slopes_eb (Real A[N][AMREX_SPACEDIM], DU const& f, int nx, int ny)
16{
17 Real AtA[AMREX_SPACEDIM][AMREX_SPACEDIM] = {};
18 Real Atb[AMREX_SPACEDIM] = {};
19
20 int lc = 0;
21 for (int jj = -ny; jj <= ny; ++jj) {
22 for (int ii = -nx; ii <= nx; ++ii) {
23 AtA[0][0] += A[lc][0] * A[lc][0];
24 AtA[0][1] += A[lc][0] * A[lc][1];
25 AtA[1][1] += A[lc][1] * A[lc][1];
26 auto du = f(ii,jj);
27 Atb[0] += A[lc][0]*du;
28 Atb[1] += A[lc][1]*du;
29 ++lc;
30 }}
31
32 // Fill in symmetric
33 AtA[1][0] = AtA[0][1];
34
35 Real detAtA =
36 (AtA[0][0]*AtA[1][1])-
37 (AtA[0][1]*AtA[1][0]);
38
39 Real detAtA_x =
40 (Atb[0] *AtA[1][1]) -
41 (AtA[0][1]*Atb[1]);
42
43 // Slope at centroid of (i,j,k)
44 Real xs = detAtA_x / detAtA;
45
46 Real detAtA_y =
47 (AtA[0][0]*Atb[1]) -
48 (Atb[0] * AtA[1][0]);
49
50 // Slope at centroid of (i,j,k)
51 Real ys = detAtA_y / detAtA;
52
53 return {xs,ys};
54}
55}

57// amrex_calc_slopes_eb calculates the slope in each coordinate direction using a
58// least squares linear fit to the 8 nearest neighbors, with the function
59// going through the centroid of cell(i,j,k). This does not assume that the cell centroids,
60// where the data is assume to live, are the same as cell centers. Note that this routine
61// is called either by amrex_calc_slopes_eb or amrex_calc_slopes_extdir_eb; in the former
62// A is defined with the cell centroids; in the latter, the A values corresponding to values
63// defined on faces use the face centroid location.
64//
65AMREX_GPU_DEVICE AMREX_FORCE_INLINE
66amrex::GpuArray<amrex::Real,AMREX_SPACEDIM>
67amrex_calc_slopes_eb_given_A (int i, int j, int /*k*/, int n,
68 amrex::Real A[9][AMREX_SPACEDIM],
69 amrex::Array4<amrex::Real const> const& state,
70 amrex::Array4<amrex::EBCellFlag const> const& flag) noexcept
71{
72 auto f = [&] (int ii, int jj) {
73 if( flag(i,j,0).isConnected(ii,jj,0) && !(ii==0 && jj==0))
74 {
75 return state(i+ii,j+jj,0,n) - state(i,j,0,n);
76 } else {
77 return amrex::Real(0.0);
78 }
79 };
80 return amrex::calc_slopes_eb<9>(A, f, 1, 1);
81}

113// amrex_overwrite_with_regular_slopes calculates the slope in each coordinate direction
114// with a standard non-EB slope calculation (that depends on max_order)

116AMREX_GPU_DEVICE AMREX_FORCE_INLINE
117void
118amrex_overwrite_with_regular_slopes(int i, int j, int k, int n,
119 amrex::Real& xslope, amrex::Real& yslope,
120 amrex::Array4<amrex::Real const> const& state,
121 amrex::Array4<amrex::Real const> const& vfrac,
122 int max_order) noexcept
123{
124 //
125 // Here we over-write -- if possible -- with a stencil not using the EB stencil
126 //
127 AMREX_ALWAYS_ASSERT( max_order == 0 || max_order == 2 || max_order == 4);
128
129 // We have enough cells in the x-direction to do 4th order slopes
130 // centered on (i,j,k) with all values at cell centers
131 if (max_order == 4 &&
132 vfrac(i,j,k) == 1. && vfrac(i-1,j,k) == 1. && vfrac(i-2,j,k) == 1. &&
133 vfrac(i+1,j,k) == 1. && vfrac(i+2,j,k) == 1.)
134 {
135 int order = 4;
136 xslope = amrex_calc_xslope(i,j,k,n,order,state);
137 }
138 // We have enough cells in the x-direction to do 2nd order slopes
139 // centered on (i,j,k) with all values at cell centers
140 else if (max_order > 0 && vfrac(i,j,k) == 1. && vfrac(i-1,j,k) == 1. && vfrac(i+1,j,k) == 1.)
141 {
142 int order = 2;
143 xslope = amrex_calc_xslope(i,j,k,n,order,state);
144 } else if (max_order == 0) {
145 xslope = 0.;
146 }
147
148 // We have enough cells in the y-direction to do 4th order slopes
149 // centered on (i,j,k) with all values at cell centers
150 if (max_order == 4 &&
151 vfrac(i,j,k) == 1. && vfrac(i,j-1,k) == 1. && vfrac(i,j-2,k) == 1. &&
152 vfrac(i,j+1,k) == 1. && vfrac(i,j+2,k) == 1.)
153 {
154 int order = 4;
155 yslope = amrex_calc_yslope(i,j,k,n,order,state);
156 }
157 // We have enough cells in the y-direction to do 2nd order slopes
158 // centered on (i,j,k) with all values at cell centers
159 else if (max_order > 0 && vfrac(i,j,k) == 1. && vfrac(i,j-1,k) == 1. && vfrac(i,j+1,k) == 1.)
160 {
161 int order = 2;
162 yslope = amrex_calc_yslope(i,j,k,n,order,state);
163 } else if (max_order == 0) {
164 yslope = 0.;
165 }

168// amrex_calc_slopes_eb calculates the slope in each coordinate direction using a
169// 1) standard limited slope if all three cells in the stencil are regular cells
170// OR
171// 2) least squares linear fit to the at-most 8 nearest neighbors, with the function
172// going through the centroid of cell(i,j,k). This does not assume that the cell centroids,
173// where the data is assume to live, are the same as cell centers. Note that calc_slopes_eb
174// defines the matrix A and passes this A to amrex_calc_slopes_eb_given_A.
175//
176// This routine assumes that there are no relevant hoextrap/extdir domain boundary conditions for this cell --
177// it does not test for them so this should not be called if such boundaries might be present
178//
179AMREX_GPU_DEVICE AMREX_FORCE_INLINE
180amrex::GpuArray<amrex::Real,AMREX_SPACEDIM>
181amrex_calc_slopes_eb (int i, int j, int k, int n,
182 amrex::Array4<amrex::Real const> const& state,
183 amrex::Array4<amrex::Real const> const& ccent,
184 amrex::Array4<amrex::Real const> const& vfrac,
185 amrex::Array4<amrex::EBCellFlag const> const& flag,
186 int max_order) noexcept
187{
188 constexpr int dim_a = 9;
189 amrex::Real A[dim_a][AMREX_SPACEDIM];
190
191 int lc=0;
192 int kk = 0;
193 {
194 for(int jj(-1); jj<=1; jj++){
195 for(int ii(-1); ii<=1; ii++){
196
197 if( flag(i,j,k).isConnected(ii,jj,kk) &&
198 ! (ii==0 && jj==0 && kk==0)) {
199
200 // Not multiplying by dx to be consistent with how the
201 // slope is stored. Also not including the global shift
202 // wrt plo or i,j,k. We only need relative distance.
203
204 A[lc][0] = amrex::Real(ii) + ccent(i+ii,j+jj,k+kk,0) - ccent(i,j,k,0);
205 A[lc][1] = amrex::Real(jj) + ccent(i+ii,j+jj,k+kk,1) - ccent(i,j,k,1);
206
207 } else {
208
209 A[lc][0] = amrex::Real(0.0);
210 A[lc][1] = amrex::Real(0.0);
211 }
212 lc++;
213 }
214 }
215 }
216
217 const auto& eb_slopes = amrex_calc_slopes_eb_given_A (i,j,k,n,A,state,flag);
218
219 amrex::Real xslope = eb_slopes[0];
220 amrex::Real yslope = eb_slopes[1];
221
222 // This will over-write the values of xslope and yslope if appropriate
223 amrex_overwrite_with_regular_slopes(i,j,k,n,xslope,yslope,state,vfrac,max_order);
224
225 return {xslope,yslope};
226}

350// 1) standard limited slope if all three cells in the stencil are regular cells

358amrex::GpuArray<amrex::Real,AMREX_SPACEDIM>
359amrex_calc_slopes_extdir_eb (int i, int j, int k, int n,

377 (edlo_y && j <= domlo_y) || (edhi_y && j >= domhi_y);

405

408

417 {
418 if (!jlo_test && !jhi_test && !klo_test && !khi_test)
419 {
420 A[lc][1] = amrex::Real(jj) + fcx(i ,j+jj,k+kk,0);

423 } else if (ihi_test) {
424
425 if (!jlo_test && !jhi_test && !klo_test && !khi_test)
426 {
427 A[lc][1] = amrex::Real(jj) + fcx(i+ii,j+jj,k+kk,0);

436 {
437 A[lc][0] = amrex::Real(ii) + fcy(i+ii,j ,k+kk,0);

442
443 if (!ilo_test && !ihi_test && !klo_test && !khi_test)
444 {
445 A[lc][0] = amrex::Real(ii) + fcy(i+ii,j+jj,k+kk,0);

451 A[lc][1] -= ccent(i,j,k,1);

472 (edlo_y && j < domlo_y) || (edhi_y && j > domhi_y) )

519 const auto& slopes = amrex_calc_slopes_eb_grown (i,j,k,n,nx,ny,state,ccent,vfrac,flag,max_order);
520 return slopes;
521
522 } else {
523
524 amrex::Real A[dim_a][AMREX_SPACEDIM];
525
526 int lc=0;
527 int kk = 0;
528
529 for(int jj(-ny); jj<=ny; jj++) {
530 for(int ii(-nx); ii<=nx; ii++)
531 {
532 if ( !flag(i+ii,j+jj,k).isCovered() && !(ii==0 && jj==0 && kk==0) )
533 {
534 bool ilo_test = ( edlo_x && (i == domlo_x) && ii == -1);
535 bool ihi_test = ( edhi_x && (i == domhi_x) && ii == 1);
536
537 bool jlo_test = ( edlo_y && (j == domlo_y) && jj == -1);
538 bool jhi_test = ( edhi_y && (j == domhi_y) && jj == 1);
539
540 bool klo_test = false;
541 bool khi_test = false;
542
543 // These are the default values if no physical boundary
544 A[lc][0] = amrex::Real(ii) + ccent(i+ii,j+jj,k+kk,0);
545 A[lc][1] = amrex::Real(jj) + ccent(i+ii,j+jj,k+kk,1);
546 // Do corrections for entire x-face
547 if (ilo_test)
548 {
549 if (!jlo_test && !jhi_test && !klo_test && !khi_test)
550 {
551 A[lc][1] = amrex::Real(jj) + fcx(i ,j+jj,k+kk,0);
552 }
553 A[lc][0] = amrex::Real(-0.5) ;
554 } else if (ihi_test) {
555
556 if (!jlo_test && !jhi_test && !klo_test && !khi_test)
557 {
558 A[lc][1] = amrex::Real(jj) + fcx(i+ii,j+jj,k+kk,0);
559 }
560 A[lc][0] = amrex::Real(0.5) ;
561 }
562
563 // Do corrections for entire y-face
564 if (jlo_test)
565 {
566 if (!ilo_test && !ihi_test && !klo_test && !khi_test)
567 {
568 A[lc][0] = amrex::Real(ii) + fcy(i+ii,j ,k+kk,0);
569 }
570 A[lc][1] = amrex::Real(-0.5) ;
571
572 } else if (jhi_test) {
573
574 if (!ilo_test && !ihi_test && !klo_test && !khi_test)
575 {
576 A[lc][0] = amrex::Real(ii) + fcy(i+ii,j+jj,k+kk,0);
577 }
578 A[lc][1] = amrex::Real(0.5) ;
579 }
580
581 A[lc][0] -= ccent(i,j,k,0);
582 A[lc][1] -= ccent(i,j,k,1);
583
584 } else {
585 A[lc][0] = amrex::Real(0.0);
586 A[lc][1] = amrex::Real(0.0);
587 }
588 lc++;
589 }} // i,j
590
591 const auto& slopes = amrex_calc_slopes_eb_given_A_grown (i,j,k,n,nx,ny,A,state,flag);
592 xslope = slopes[0];
593 yslope = slopes[1];
594
595 // This will over-write the values of xslope and yslope if appropriate
596 amrex_overwrite_with_regular_slopes_extdir(i,j,k,n,xslope,yslope,state,vfrac,
597 edlo_x,edlo_y,edhi_x,edhi_y,
598 domlo_x,domlo_y,domhi_x,domhi_y,max_order);
599
600 // Zero out slopes outside of an extdir (or hoextrap) boundary
601 // TODO: is this the right thing to do at a HOEXTRAP boundary??
602 if ( (edlo_x && i < domlo_x) || (edhi_x && i > domhi_x) ||
603 (edlo_y && j < domlo_y) || (edhi_y && j > domhi_y) )
604 {
605 xslope = 0.;
606 yslope = 0.;
607 }
608 return {xslope,yslope};
609 }

612AMREX_GPU_DEVICE AMREX_FORCE_INLINE
613amrex::Real
614amrex_calc_alpha_stencil(amrex::Real q_hat, amrex::Real q_max,
615 amrex::Real q_min, amrex::Real state, amrex::Real alpha) noexcept
616{
617 using namespace amrex::literals;
618
619 auto alpha_temp = 0.0_rt;
620 auto sml = 1.0e-13_rt;
621
622 if ((q_hat-state) > sml) {
623 alpha_temp = amrex::min(1.0_rt,(q_max-state)/(q_hat-state));
624 } else if ((q_hat-state) < -sml) {
625 alpha_temp = amrex::min(1.0_rt,(q_min-state)/(q_hat-state));
626 } else {
627 alpha_temp = 1.0_rt;
628 }
629 return amrex::min(alpha, alpha_temp);
630}

632AMREX_GPU_DEVICE AMREX_FORCE_INLINE
633amrex::GpuArray<amrex::Real,AMREX_SPACEDIM>
634amrex_calc_alpha_limiter(int i, int j, int k, int n,
635 amrex::Array4<amrex::Real const> const& state,
636 amrex::Array4<amrex::EBCellFlag const> const& flag,
637 const amrex::GpuArray<amrex::Real,AMREX_SPACEDIM>& slopes,
638 amrex::Array4<amrex::Real const> const& fcx,
639 amrex::Array4<amrex::Real const> const& fcy,
640 amrex::Array4<amrex::Real const> const& ccent) noexcept
641{
642 amrex::Real alpha = 2.0;
643
644 int cuts_x = 0;
645 int cuts_y = 0;
646
647 // Compute how many cut or regular faces do we have in 3x3 block
648 int kk = 0;
649 for(int jj(-1); jj<=1; jj++){
650 for(int ii(-1); ii<=1; ii++){
651 if( flag(i,j,k).isConnected(ii,jj,kk) && !(ii==0 && jj==0))
652 {
653 if ((ii==-1 || ii==1) && jj==0) { cuts_x++; }
654 if ((jj==-1 || jj==1) && ii==0) { cuts_y++; }
655 }
656 }
657 }
658
659 amrex::Real xc = ccent(i,j,k,0); // centroid of cell (i,j,k)
660 amrex::Real yc = ccent(i,j,k,1);
661
662 //Reconstruct values at the face centroids and compute the limiter
663 if(flag(i,j,k).isConnected(0,1,0)) {
664 amrex::Real xf = fcy(i,j+1,k,0); // local (x,z) of centroid of y-face we are extrapolating to
665
666 amrex::Real delta_x = xf - xc;
667 amrex::Real delta_y = amrex::Real(0.5) - yc;
668
669 amrex::Real q_hat = state(i,j,k,n) + delta_x * slopes[0] + delta_y * slopes[1];
670
671 amrex::Real q_min = state(i,j,k,n);
672 amrex::Real q_max = state(i,j,k,n);
673
674 // Compute max and min values in a 3x2 stencil
675 for(int jj(0); jj<=1; jj++){
676 for(int ii(-1); ii<=1; ii++){
677 if ( flag(i,j,k).isConnected(ii,jj,kk) && !(ii==0 && jj==0) ) {
678 if (state(i+ii,j+jj,k+kk,n) > q_max) { q_max = state(i+ii,j+jj,k+kk,n); }
679 if (state(i+ii,j+jj,k+kk,n) < q_min) { q_min = state(i+ii,j+jj,k+kk,n); }
680 }
681 }
682 }
683
684 alpha = amrex_calc_alpha_stencil(q_hat, q_max, q_min, state(i,j,k,n), alpha);
685 }
686 if (flag(i,j,k).isConnected(0,-1,0)){
687 amrex::Real xf = fcy(i,j,k,0); // local (x,z) of centroid of y-face we are extrapolating to
688
689 amrex::Real delta_x = xf - xc;
690 amrex::Real delta_y = amrex::Real(0.5) + yc;
691
692 amrex::Real q_hat = state(i,j,k,n) + delta_x * slopes[0] - delta_y * slopes[1];
693
694 amrex::Real q_min = state(i,j,k,n);
695 amrex::Real q_max = state(i,j,k,n);
696
697 // Compute max and min values in a 3x2 stencil
698 for(int jj(-1); jj<=0; jj++){
699 for(int ii(-1); ii<=1; ii++){
700 if ( flag(i,j,k).isConnected(ii,jj,kk) && !(ii==0 && jj==0) ) {
701 if (state(i+ii,j+jj,k+kk,n) > q_max) { q_max = state(i+ii,j+jj,k+kk,n); }
702 if (state(i+ii,j+jj,k+kk,n) < q_min) { q_min = state(i+ii,j+jj,k+kk,n); }
703 }
704 }
705 }
706
707 alpha = amrex_calc_alpha_stencil(q_hat, q_max, q_min, state(i,j,k,n), alpha);
708 }
709 if(flag(i,j,k).isConnected(1,0,0)) {
710 amrex::Real yf = fcx(i+1,j,k,0); // local (y,z) of centroid of x-face we are extrapolating to
711
712 amrex::Real delta_x = amrex::Real(0.5) - xc;
713 amrex::Real delta_y = yf - yc;
714
715 amrex::Real q_hat = state(i,j,k,n) + delta_x * slopes[0] + delta_y * slopes[1];
716
717 amrex::Real q_min = state(i,j,k,n);
718 amrex::Real q_max = state(i,j,k,n);
719
720 for(int jj(-1); jj<=1; jj++){
721 for(int ii(0); ii<=1; ii++){
722 if ( flag(i,j,k).isConnected(ii,jj,kk) && !(ii==0 && jj==0) ) {
723 if (state(i+ii,j+jj,k+kk,n) > q_max) { q_max = state(i+ii,j+jj,k+kk,n); }
724 if (state(i+ii,j+jj,k+kk,n) < q_min) { q_min = state(i+ii,j+jj,k+kk,n); }
725 }
726 }
727 }
728
729 alpha = amrex_calc_alpha_stencil(q_hat, q_max, q_min, state(i,j,k,n), alpha);
730 }
731 if(flag(i,j,k).isConnected(-1,0,0)) {
732 amrex::Real yf = fcx(i,j,k,0); // local (y,z) of centroid of x-face we are extrapolating to
733
734 amrex::Real delta_x = amrex::Real(0.5) + xc;
735 amrex::Real delta_y = yf - yc;
736
737 amrex::Real q_hat = state(i,j,k,n) - delta_x * slopes[0] + delta_y * slopes[1];
738
739 amrex::Real q_min = state(i,j,k,n);
740 amrex::Real q_max = state(i,j,k,n);
741
742 for(int jj(-1); jj<=1; jj++){
743 for(int ii(-1); ii<=0; ii++){
744 if( flag(i,j,k).isConnected(ii,jj,kk) && !(ii==0 && jj==0)) {
745 if (state(i+ii,j+jj,k+kk,n) > q_max) { q_max = state(i+ii,j+jj,k+kk,n); }
746 if (state(i+ii,j+jj,k+kk,n) < q_min) { q_min = state(i+ii,j+jj,k+kk,n); }
747 }
748 }
749 }
750 alpha = amrex_calc_alpha_stencil(q_hat, q_max, q_min, state(i,j,k,n), alpha);
751 }
752
753 amrex::Real xalpha = alpha;
754 amrex::Real yalpha = alpha;
755
756 //Zeroing out the slopes in the direction where a covered face exists.
757 if (cuts_x<2) { xalpha = 0; }
758 if (cuts_y<2) { yalpha = 0; }
759
760 return {xalpha,yalpha};
761}

764//is multiplied by a limiter based on the work of Barth-Jespersen.

766amrex::GpuArray<amrex::Real,AMREX_SPACEDIM>
767amrex_lim_slopes_eb (int i, int j, int k, int n,

792 }
793
794 return {alpha_lim[0]*slopes[0],alpha_lim[1]*slopes[1]};
