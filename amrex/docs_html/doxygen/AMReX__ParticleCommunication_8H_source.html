214  AMREX_FOR_1D ( num_copies, i,

217  if (dst_box < 0) continue;
218  int box_offset = p_box_offsets[p_box_perm[dst_box]];
219  int dst_index = box_offset + p_dst_indices[i];
220  int src_index = p_src_indices[i];
221  p_snd_buffer[dst_index] = ptd.getSuperParticle(src_index);
