33struct RedistributeUnpackPolicy
34{
35 template <class PTile>
36 void resizeTiles (std::vector<PTile*>& tiles, const std::vector<int>& sizes, std::vector<int>& offsets) const
37 {
38 int N = static_cast<int>(sizes.size());
39
40 std::map<PTile*, int> tile_sizes;
41 for(int i = 0; i < N; ++i) {
42 tile_sizes[tiles[i]] = tiles[i]->numParticles();
43 }
44
45 for(int i = 0; i < N; ++i)
46 {
47 offsets.push_back(tile_sizes[tiles[i]]);
48 tile_sizes[tiles[i]] += sizes[i];
49 }
50
51 for (auto& kv : tile_sizes) {
52 kv.first->resize(kv.second);
53 }
54 }

232 m_snd_pad_correction_d.begin());

234 d_int_comp_mask.resize(int_comp_mask.size());
235 Gpu::copyAsync(Gpu::hostToDevice, int_comp_mask.begin(), int_comp_mask.end(),
236 d_int_comp_mask.begin());
237 d_real_comp_mask.resize(real_comp_mask.size());

241 Gpu::streamSynchronize();
242
243 int NStructReal = PC::ParticleContainerType::NStructReal;
244 int NStructInt = PC::ParticleContainerType::NStructInt;
245
246 int num_real_comm_comp = 0;
247 for (int i = AMREX_SPACEDIM + NStructReal; i < real_comp_mask.size(); ++i) {
248 if (real_comp_mask[i]) {++num_real_comm_comp;}
249 }
250

328template <class PC, class Buffer,
329 std::enable_if_t<IsParticleContainer<PC>::value &&
330 std::is_base_of_v<PolymorphicArenaAllocator<typename Buffer::value_type>,
331 Buffer>, int> foo = 0>
332void packBuffer (const PC& pc, const ParticleCopyOp& op, const ParticleCopyPlan& plan,
333 Buffer& snd_buffer)
334{
335 BL_PROFILE("amrex::packBuffer");
336
337 Long psize = plan.superParticleSize();
338
339 int num_levels = op.numLevels();
340 int num_buckets = pc.BufferMap().numBuckets();
341
342 std::size_t total_buffer_size = 0;
343 if (plan.m_snd_offsets.empty())
344 {
345 unsigned int np = 0;
346 Gpu::copy(Gpu::deviceToHost, plan.m_box_offsets.begin() + num_buckets,
347 plan.m_box_offsets.begin() + num_buckets + 1, &np);
348 total_buffer_size = np*psize;
349 }
350 else
351 {
352 total_buffer_size = plan.m_snd_offsets.back();
353 }
354
355 if (! snd_buffer.arena()->hasFreeDeviceMemory(total_buffer_size)) {
356 snd_buffer.clear();
357 snd_buffer.setArena(The_Pinned_Arena());
358 }
359 snd_buffer.resize(total_buffer_size);
360
361 const auto* p_comm_real = plan.d_real_comp_mask.dataPtr();
362 const auto* p_comm_int = plan.d_int_comp_mask.dataPtr();
363
364 const auto plo = pc.Geom(0).ProbLoArray();
365 const auto phi = pc.Geom(0).ProbHiArray();
366 const auto is_per = pc.Geom(0).isPeriodicArray();
367 for (int lev = 0; lev < num_levels; ++lev)
368 {
369 auto& plev = pc.GetParticles(lev);
370 for (auto& kv : plev)
371 {
372 int gid = kv.first.first;
373 int tid = kv.first.second;
374 auto index = std::make_pair(gid, tid);
375
376 auto& src_tile = plev.at(index);
377 const auto& ptd = src_tile.getConstParticleTileData();
378
379 int num_copies = op.numCopies(gid, lev);
380 if (num_copies == 0) { continue; }
381
382 const auto* p_boxes = op.m_boxes[lev].at(gid).dataPtr();
383 const auto* p_levels = op.m_levels[lev].at(gid).dataPtr();
384 const auto* p_src_indices = op.m_src_indices[lev].at(gid).dataPtr();
385 const auto* p_periodic_shift = op.m_periodic_shift[lev].at(gid).dataPtr();
386 const auto* p_dst_indices = plan.m_dst_indices[lev].at(gid).dataPtr();
387 auto* p_snd_buffer = snd_buffer.dataPtr();
388 GetSendBufferOffset get_offset(plan, pc.BufferMap());
389
390 AMREX_FOR_1D ( num_copies, i,
391 {
392 int dst_box = p_boxes[i];
393 if (dst_box >= 0)
394 {
395 int dst_lev = p_levels[i];
396 auto dst_offset = get_offset(dst_box, dst_lev, psize, p_dst_indices[i]);
397 int src_index = p_src_indices[i];
398 ptd.packParticleData(p_snd_buffer, src_index, dst_offset, p_comm_real, p_comm_int);
399
400 const IntVect& pshift = p_periodic_shift[i];
401 bool do_periodic_shift =
402 AMREX_D_TERM( (is_per[0] && pshift[0] != 0),
403 || (is_per[1] && pshift[1] != 0),
404 || (is_per[2] && pshift[2] != 0) );
405
406 if (do_periodic_shift)
407 {
408 ParticleReal pos[AMREX_SPACEDIM];
409 amrex::Gpu::memcpy(&pos[0], &p_snd_buffer[dst_offset],
410 AMREX_SPACEDIM*sizeof(ParticleReal));
411 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
412 {
413 if (! is_per[idim]) { continue; }
414 if (pshift[idim] > 0) {
415 pos[idim] += phi[idim] - plo[idim];
416 } else if (pshift[idim] < 0) {
417 pos[idim] -= phi[idim] - plo[idim];
418 }
419 }
420 amrex::Gpu::memcpy(&p_snd_buffer[dst_offset], &pos[0],
421 AMREX_SPACEDIM*sizeof(ParticleReal));
422 }
423 }
424 });
425 }
426 }
427}

429template <class PC, class Buffer, class UnpackPolicy,
430 std::enable_if_t<IsParticleContainer<PC>::value, int> foo = 0>
431void unpackBuffer (PC& pc, const ParticleCopyPlan& plan, const Buffer& snd_buffer, UnpackPolicy const& policy)
432{
433 BL_PROFILE("amrex::unpackBuffer");
434
435 using PTile = typename PC::ParticleTileType;
436
437 int num_levels = pc.BufferMap().numLevels();
438 Long psize = plan.superParticleSize();
439
440 // count how many particles we have to add to each tile
441 std::vector<int> sizes;
442 std::vector<PTile*> tiles;
443 for (int lev = 0; lev < num_levels; ++lev)
444 {
445 for(MFIter mfi = pc.MakeMFIter(lev); mfi.isValid(); ++mfi)
446 {
447 int gid = mfi.index();
448 int tid = mfi.LocalTileIndex();
449 auto& tile = pc.DefineAndReturnParticleTile(lev, gid, tid);
450 int num_copies = plan.m_box_counts_h[pc.BufferMap().gridAndLevToBucket(gid, lev)];
451 sizes.push_back(num_copies);
452 tiles.push_back(&tile);
453 }
454 }
455
456 // resize the tiles and compute offsets
457 std::vector<int> offsets;
458 policy.resizeTiles(tiles, sizes, offsets);
459
460 const auto* p_comm_real = plan.d_real_comp_mask.dataPtr();
461 const auto* p_comm_int = plan.d_int_comp_mask.dataPtr();
462
463 // local unpack
464 int uindex = 0;
465 for (int lev = 0; lev < num_levels; ++lev)
466 {
467 auto& plev = pc.GetParticles(lev);
468 for(MFIter mfi = pc.MakeMFIter(lev); mfi.isValid(); ++mfi)
469 {
470 int gid = mfi.index();
471 int tid = mfi.LocalTileIndex();
472 auto index = std::make_pair(gid, tid);
473
474 auto& tile = plev[index];
475
476 GetSendBufferOffset get_offset(plan, pc.BufferMap());
477 auto p_snd_buffer = snd_buffer.dataPtr();
478
479 int offset = offsets[uindex];
480 int size = sizes[uindex];
481 ++uindex;
482
483 auto ptd = tile.getParticleTileData();
484 AMREX_FOR_1D ( size, i,
485 {
486 auto src_offset = get_offset(gid, lev, psize, i);
487 int dst_index = offset + i;
488 ptd.unpackParticleData(p_snd_buffer, src_offset, dst_index, p_comm_real, p_comm_int);
489 });
490 }
491 }
492}

494template <class PC, class SndBuffer, class RcvBuffer,
495 std::enable_if_t<IsParticleContainer<PC>::value, int> foo = 0>
496void communicateParticlesStart (const PC& pc, ParticleCopyPlan& plan, const SndBuffer& snd_buffer, RcvBuffer& rcv_buffer)
497{
498 BL_PROFILE("amrex::communicateParticlesStart");
499
500#ifdef AMREX_USE_MPI
501 Long psize = plan.superParticleSize();
502 const int NProcs = ParallelContext::NProcsSub();
503 const int MyProc = ParallelContext::MyProcSub();
504
505 if (NProcs == 1) { return; }
506
507 Vector<int> RcvProc;
508 Vector<Long> rOffset;
509
510 plan.m_rcv_pad_correction_h.resize(0);
511 plan.m_rcv_pad_correction_h.push_back(0);
512
513 Long TotRcvBytes = 0;
514 for (int i = 0; i < NProcs; ++i) {
515 if (plan.m_rcv_num_particles[i] > 0) {
516 RcvProc.push_back(i);
517 Long nbytes = plan.m_rcv_num_particles[i]*psize;
518 std::size_t acd = ParallelDescriptor::sizeof_selected_comm_data_type(nbytes);
519 TotRcvBytes = Long(amrex::aligned_size(acd, TotRcvBytes));
520 rOffset.push_back(TotRcvBytes);
521 TotRcvBytes += Long(amrex::aligned_size(acd, nbytes));
522 plan.m_rcv_pad_correction_h.push_back(plan.m_rcv_pad_correction_h.back() + nbytes);
523 }
524 }
525
526 for (int i = 0; i < plan.m_nrcvs; ++i)
527 {
528 plan.m_rcv_pad_correction_h[i] = rOffset[i] - plan.m_rcv_pad_correction_h[i];
529 }
530
531 plan.m_rcv_pad_correction_d.resize(plan.m_rcv_pad_correction_h.size());
532 Gpu::copy(Gpu::hostToDevice, plan.m_rcv_pad_correction_h.begin(), plan.m_rcv_pad_correction_h.end(),
533 plan.m_rcv_pad_correction_d.begin());
534
535 rcv_buffer.resize(TotRcvBytes);
536
537 plan.m_nrcvs = int(RcvProc.size());
538
539 plan.m_particle_rstats.resize(0);
540 plan.m_particle_rstats.resize(plan.m_nrcvs);
541
542 plan.m_particle_rreqs.resize(0);
543 plan.m_particle_rreqs.resize(plan.m_nrcvs);
544
545 plan.m_particle_sstats.resize(0);
546 plan.m_particle_sreqs.resize(0);
547
548 const int SeqNum = ParallelDescriptor::SeqNum();
549
550 // Post receives.
551 for (int i = 0; i < plan.m_nrcvs; ++i) {
552 const auto Who = RcvProc[i];
553 const auto offset = rOffset[i];
554 Long nbytes = plan.m_rcv_num_particles[Who]*psize;
555 std::size_t acd = ParallelDescriptor::sizeof_selected_comm_data_type(nbytes);
556 const auto Cnt = amrex::aligned_size(acd, nbytes);
557
558 AMREX_ASSERT(Cnt > 0);
559 AMREX_ASSERT(Who >= 0 && Who < NProcs);
560 AMREX_ASSERT(amrex::aligned_size(acd, nbytes) % acd == 0);
561
562 plan.m_particle_rreqs[i] =
563 ParallelDescriptor::Arecv((char*) (rcv_buffer.dataPtr() + offset), Cnt, Who, SeqNum, ParallelContext::CommunicatorSub()).req();
564 }
565
566 if (plan.m_NumSnds == 0) { return; }
567
568 // Send.
569 for (int i = 0; i < NProcs; ++i)
570 {
571 if (i == MyProc) { continue; }
572 const auto Who = i;
573 const auto Cnt = plan.m_snd_counts[i];
574 if (Cnt == 0) { continue; }
575
576 auto snd_offset = plan.m_snd_offsets[i];
577 AMREX_ASSERT(plan.m_snd_counts[i] % ParallelDescriptor::sizeof_selected_comm_data_type(plan.m_snd_num_particles[i]*psize) == 0);
578 AMREX_ASSERT(Who >= 0 && Who < NProcs);
579
580 plan.m_particle_sreqs.push_back(ParallelDescriptor::Asend((char const*)(snd_buffer.dataPtr()+snd_offset), Cnt, Who, SeqNum,
581 ParallelContext::CommunicatorSub()).req());
582 }
583
584 plan.m_particle_sstats.resize(plan.m_particle_sreqs.size());
585
586 amrex::ignore_unused(pc);
587#else
588 amrex::ignore_unused(pc,plan,snd_buffer,rcv_buffer);
589#endif // MPI
590}

594template <class PC, class Buffer, class UnpackPolicy,

596void unpackRemotes (PC& pc, const ParticleCopyPlan& plan, Buffer& rcv_buffer, UnpackPolicy const& policy)
597{
598 BL_PROFILE("amrex::unpackRemotes");
599
600#ifdef AMREX_USE_MPI
601 const int NProcs = ParallelContext::NProcsSub();
602 if (NProcs == 1) { return; }
603
604 const int MyProc = ParallelContext::MyProcSub();
605 amrex::ignore_unused(MyProc);
606 using PTile = typename PC::ParticleTileType;
607
608 if (plan.m_nrcvs > 0)
609 {
610 const auto* p_comm_real = plan.d_real_comp_mask.dataPtr();
611 const auto* p_comm_int = plan.d_int_comp_mask.dataPtr();
612 auto* p_rcv_buffer = rcv_buffer.dataPtr();
613
614 std::vector<int> sizes;
615 std::vector<PTile*> tiles;
616 for (int i = 0, N = int(plan.m_rcv_box_counts.size()); i < N; ++i)
617 {
618 int copy_size = plan.m_rcv_box_counts[i];
619 int lev = plan.m_rcv_box_levs[i];
620 int gid = plan.m_rcv_box_ids[i];
621 int tid = 0;
622 auto& tile = pc.DefineAndReturnParticleTile(lev, gid, tid);
623 sizes.push_back(copy_size);
624 tiles.push_back(&tile);
625 }
626
627 Vector<int> offsets;
628 policy.resizeTiles(tiles, sizes, offsets);
629 Gpu::streamSynchronize();
630 int uindex = 0;
631 int procindex = 0, rproc = plan.m_rcv_box_pids[0];
632 for (int i = 0, N = int(plan.m_rcv_box_counts.size()); i < N; ++i)
633 {
634 int lev = plan.m_rcv_box_levs[i];
635 int gid = plan.m_rcv_box_ids[i];
636 int tid = 0;
637 auto offset = plan.m_rcv_box_offsets[i];
638 procindex = (rproc == plan.m_rcv_box_pids[i]) ? procindex : procindex+1;
639 rproc = plan.m_rcv_box_pids[i];
640
641 auto& tile = pc.DefineAndReturnParticleTile(lev, gid, tid);
642 auto ptd = tile.getParticleTileData();
643
644 AMREX_ASSERT(MyProc ==
645 ParallelContext::global_to_local_rank(pc.ParticleDistributionMap(lev)[gid]));
646
647 int dst_offset = offsets[uindex];
648 int size = sizes[uindex];
649 ++uindex;
650
651 Long psize = plan.superParticleSize();
652 const auto* p_pad_adjust = plan.m_rcv_pad_correction_d.dataPtr();
653
654 AMREX_FOR_1D ( size, ip, {
655 Long src_offset = psize*(offset + ip) + p_pad_adjust[procindex];
656 int dst_index = dst_offset + ip;
657 ptd.unpackParticleData(p_rcv_buffer, src_offset, dst_index,
658 p_comm_real, p_comm_int);
659 });
660
661 Gpu::streamSynchronize();
662 }
663 }
664#else
665 amrex::ignore_unused(pc,plan,rcv_buffer,policy);
666#endif // MPI
667}
