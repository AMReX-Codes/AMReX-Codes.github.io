329  const auto* p_comm_int = plan.d_int_comp_mask.dataPtr();

332  const auto phi = pc.Geom(0).ProbHiArray();

334  for (int lev = 0; lev < num_levels; ++lev)

345

347  if (num_copies == 0) { continue; }
