481 template <typename T>
482 T R2X<T>::scalingFactor () const
483 {
484  auto r = m_dom_0.numPts();
485  for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
486  if (m_bc[idim].first != Boundary::periodic

501  // forward
502 
503  m_rx.ParallelCopy(inmf, 0, 0, 1);
504  if (m_bc[0].first == Boundary::periodic) {
505  m_fft_fwd_x.template compute_r2c<Direction::forward>();

515  }
