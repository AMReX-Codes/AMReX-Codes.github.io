155  {

157  const auto hi = ubound(bx);
158  build(BinPolicy::GPU, nitems, v, bx.numPts(),
159  [=] AMREX_GPU_DEVICE (const T& t) noexcept
160  {
161  auto iv = f(t);
162  auto iv3 = iv.dim3();
163  int nx = hi.x-lo.x+1;
164  int ny = hi.y-lo.y+1;

300  void build (BinPolicy::OpenMPBinPolicy, N nitems, T const* v, int nbins, F&& f)
301  {

303 
304  m_items = v;
305 
306  m_bins.resize(nitems);
307  m_perm.resize(nitems);
308 
309  int nchunks = OpenMP::get_max_threads();
310  int chunksize = nitems / nchunks;
311  auto counts = (index_type*)(The_Arena()->alloc(nchunks*nbins*sizeof(index_type)));
312  for (int i = 0; i < nbins*nchunks; ++i) { counts[i] = 0;}
313 
314  m_counts.resize(0);
315  m_counts.resize(nbins+1, 0);
316 
317  m_offsets.resize(0);
318  m_offsets.resize(nbins+1);
319 
320 #ifdef AMREX_USE_OMP
321 #pragma omp parallel for
322 #endif
323  for (int j = 0; j < nchunks; ++j) {
324  int istart = j*chunksize;
325  int istop = (j == nchunks-1) ? nitems : (j+1)*chunksize;
326  for (int i = istart; i < istop; ++i) {
327  m_bins[i] = f(v[i]);
328  ++counts[nbins*j+m_bins[i]];
329  }
330  }
331 
332 #ifdef AMREX_USE_OMP
333 #pragma omp parallel for
334 #endif
335  for (int i = 0; i < nbins; ++i) {
336  index_type total = 0;
337  for (int j = 0; j < nchunks; ++j) {
338  auto tmp = counts[nbins*j+i];
339  counts[nbins*j+i] = total;
340  total += tmp;
341  }
342  m_counts[i] = total;
343  }
344 
345  // note - this part has to be serial
346  m_offsets[0] = 0;
347  for (int i = 0; i < nbins; ++i) {m_offsets[i+1] = m_offsets[i] + m_counts[i];}
348 
349 #ifdef AMREX_USE_OMP
350 #pragma omp parallel for
351 #endif
352  for (int i = 0; i < nbins; ++i) {
353  for (int j = 0; j < nchunks; ++j) {
354  counts[nbins*j+i] += m_offsets[i];
355  }
356  }
357 
358 #ifdef AMREX_USE_OMP
359 #pragma omp parallel for
360 #endif
361  for (int j = 0; j < nchunks; ++j) {
362  int istart = j*chunksize;
363  int istop = (j == nchunks-1) ? nitems : (j+1)*chunksize;
364  for (int i = istart; i < istop; ++i) {
365  auto bid = m_bins[i];
366  m_perm[counts[nbins*j+bid]++] = i;
367  }
368  }
