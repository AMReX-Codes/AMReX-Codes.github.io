741  m_amr_ref_ratio.resize(m_num_amr_levels);
742  m_num_mg_levels.resize(m_num_amr_levels);
743 
744  m_geom.resize(m_num_amr_levels);
745  m_grids.resize(m_num_amr_levels);
746  m_dmap.resize(m_num_amr_levels);
747  m_factory.resize(m_num_amr_levels);
748 
749  m_default_comm = ParallelContext::CommunicatorSub();
750 
751  const RealBox& rb = a_geom[0].ProbDomain();
752  const int coord = a_geom[0].Coord();
753  const Array<int,AMREX_SPACEDIM>& is_per = a_geom[0].isPeriodic();
754 
755  IntVect mg_coarsen_ratio_v(mg_coarsen_ratio);
756  IntVect mg_box_min_width_v(mg_box_min_width);
757  IntVect mg_domain_min_width_v(mg_domain_min_width);
758  if (hasHiddenDimension()) {

760  "Hidden direction only supported for 3d level solve");

762  mg_box_min_width_v[info.hidden_direction] = 0;
763  mg_domain_min_width_v[info.hidden_direction] = 0;
764  }
765 
766  // fine amr levels
767  for (int amrlev = m_num_amr_levels-1; amrlev > 0; --amrlev)
768  {
769  m_num_mg_levels[amrlev] = 1;
770  m_geom[amrlev].push_back(a_geom[amrlev]);
771  m_grids[amrlev].push_back(a_grids[amrlev]);
772  m_dmap[amrlev].push_back(a_dmap[amrlev]);
773  if (amrlev < a_factory.size()) {
774  m_factory[amrlev].emplace_back(a_factory[amrlev]->clone());
775  } else {
776  m_factory[amrlev].push_back(std::make_unique<DefaultFabFactory<FAB>>());
777  }
778 
779  IntVect rr = mg_coarsen_ratio_v;
780  const Box& dom = a_geom[amrlev].Domain();
781  for (int i = 0; i < 2; ++i)
782  {
783  if (!dom.coarsenable(rr)) amrex::Abort("MLLinOp: Uncoarsenable domain");
784 
785  const Box& cdom = amrex::coarsen(dom,rr);
786  if (cdom == a_geom[amrlev-1].Domain()) break;
787 
788  ++(m_num_mg_levels[amrlev]);
789 
790  m_geom[amrlev].emplace_back(cdom, rb, coord, is_per);
791 
792  m_grids[amrlev].push_back(a_grids[amrlev]);
793  AMREX_ASSERT(m_grids[amrlev].back().coarsenable(rr));
794  m_grids[amrlev].back().coarsen(rr);
795 
796  m_dmap[amrlev].push_back(a_dmap[amrlev]);
797 
798  rr *= mg_coarsen_ratio_v;
799  }
800 
801 #if (AMREX_SPACEDIM > 1)
802  if (hasHiddenDimension()) {
803  m_amr_ref_ratio[amrlev-1] = rr[AMREX_SPACEDIM-info.hidden_direction];
804  } else
805 #endif
806  {
807  m_amr_ref_ratio[amrlev-1] = rr[0];
808  }
809  }
810 
811  // coarsest amr level
812  m_num_mg_levels[0] = 1;
813  m_geom[0].push_back(a_geom[0]);
814  m_grids[0].push_back(a_grids[0]);
815  m_dmap[0].push_back(a_dmap[0]);
816  if (!a_factory.empty()) {
817  m_factory[0].emplace_back(a_factory[0]->clone());
818  } else {
819  m_factory[0].push_back(std::make_unique<DefaultFabFactory<FAB>>());
820  }
821

823  auto npts0 = m_grids[0][0].numPts();
824  m_domain_covered[0] = (npts0 == compactify(m_geom[0][0].Domain()).numPts());
825  for (int amrlev = 1; amrlev < m_num_amr_levels; ++amrlev)
826  {
827  if (!m_domain_covered[amrlev-1]) break;
828  m_domain_covered[amrlev] = (m_grids[amrlev][0].numPts() ==
829  compactify(m_geom[amrlev][0].Domain()).numPts());
830  }

832  Box aggbox;
833  bool aggable = false;
834

836  {
837  if (m_domain_covered[0])
838  {
839  aggbox = m_geom[0][0].Domain();
840  if (hasHiddenDimension()) {
841  aggbox.makeSlab(hiddenDirection(), m_grids[0][0][0].smallEnd(hiddenDirection()));

844  }
845  else
846  {

849  }
850  }
851 
852  bool agged = false;
853  bool coned = false;
854  int agg_lev = 0, con_lev = 0;
855 
856  AMREX_ALWAYS_ASSERT( ! (info.do_semicoarsening && info.hasHiddenDimension())
857  && info.semicoarsening_direction >= -1
858  && info.semicoarsening_direction < AMREX_SPACEDIM );
859

861  {
862  Box dbx = m_geom[0][0].Domain();
863  Box bbx = aggbox;
864  Real const nbxs = static_cast<Real>(m_grids[0][0].size());
865  Real const threshold_npts = static_cast<Real>(AMREX_D_TERM(info.agg_grid_size,
866  *info.agg_grid_size,
867  *info.agg_grid_size));
868  Vector<Box> domainboxes{dbx};
869  Vector<Box> boundboxes{bbx};
870  Vector<int> agg_flag{false};
871  Vector<IntVect> accum_coarsen_ratio{IntVect(1)};
872  int numsclevs = 0;
873 
874  for (int lev = 0; lev < info.max_coarsening_level; ++lev)
875  {
876  IntVect rr_level = mg_coarsen_ratio_v;
877  bool const do_semicoarsening_level = info.do_semicoarsening
878  && numsclevs < info.max_semicoarsening_level;
879  if (do_semicoarsening_level
880  && info.semicoarsening_direction != -1)
881  {
882  rr_level[info.semicoarsening_direction] = 1;
883  }
884  IntVect is_coarsenable;
885  for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
886  IntVect rr_dir(1);
887  rr_dir[idim] = rr_level[idim];
888  is_coarsenable[idim] = dbx.coarsenable(rr_dir, mg_domain_min_width_v)
889  && bbx.coarsenable(rr_dir, mg_box_min_width_v);
890  if (!is_coarsenable[idim] && do_semicoarsening_level
891  && info.semicoarsening_direction == -1)
892  {
893  is_coarsenable[idim] = true;
894  rr_level[idim] = 1;
895  }
896  }
897  if (is_coarsenable != IntVect(1) || rr_level == IntVect(1)) {
898  break;
899  }
900  if (do_semicoarsening_level && info.semicoarsening_direction == -1) {
901  // make sure there is at most one direction that is not coarsened
902  int n_ones = AMREX_D_TERM( static_cast<int>(rr_level[0] == 1),
903  + static_cast<int>(rr_level[1] == 1),
904  + static_cast<int>(rr_level[2] == 1));
905  if (n_ones > 1) { break; }
906  }
907  if (rr_level != mg_coarsen_ratio_v) {
908  ++numsclevs;
909  }
910

912  domainboxes.push_back(dbx.coarsen(rr_level));
913  boundboxes.push_back(bbx.coarsen(rr_level));

915  agg_flag.push_back(to_agg);
916  }
917 
918  for (int lev = 1, nlevs = static_cast<int>(domainboxes.size()); lev < nlevs; ++lev) {
919  if (!agged && !agg_flag[lev] &&
920  a_grids[0].coarsenable(accum_coarsen_ratio[lev], mg_box_min_width_v))
921  {
922  m_grids[0].push_back(amrex::coarsen(a_grids[0], accum_coarsen_ratio[lev]));
923  m_dmap[0].push_back(a_dmap[0]);
924  } else {
925  IntVect cr = domainboxes[lev-1].length() / domainboxes[lev].length();
926  if (!m_grids[0].back().coarsenable(cr)) {
927  break; // average_down would fail if fine boxarray is not coarsenable.
928  }
929  m_grids[0].emplace_back(boundboxes[lev]);
930  IntVect max_grid_size(info.agg_grid_size);
931  if (info.do_semicoarsening && info.max_semicoarsening_level >= lev
932  && info.semicoarsening_direction != -1)
933  {
934  IntVect blen = amrex::enclosedCells(boundboxes[lev]).size();
935  AMREX_D_TERM(int mgs_0 = (max_grid_size[0]+blen[0]-1) / blen[0];,
936  int mgs_1 = (max_grid_size[1]+blen[1]-1) / blen[1];,
937  int mgs_2 = (max_grid_size[2]+blen[2]-1) / blen[2]);
938  max_grid_size[info.semicoarsening_direction]
939  *= AMREX_D_TERM(mgs_0, *mgs_1, *mgs_2);
940  }
941  m_grids[0].back().maxSize(max_grid_size);
942  m_dmap[0].push_back(DistributionMapping());
943  if (!agged) {
944  agged = true;
945  agg_lev = lev;
946  }
947  }
948  m_geom[0].emplace_back(domainboxes[lev],rb,coord,is_per);
949  }
950  }
951  else
952  {
953  Long consolidation_threshold = 0;
954  Real avg_npts = 0.0;
955  if (info.do_consolidation) {
956  avg_npts = static_cast<Real>(a_grids[0].d_numPts()) / static_cast<Real>(ParallelContext::NProcsSub());
957  consolidation_threshold = AMREX_D_TERM(Long(info.con_grid_size),
958  *info.con_grid_size,
959  *info.con_grid_size);
960  }
961 
962  Box const& dom0 = a_geom[0].Domain();
963  IntVect rr_vec(1);
964  int numsclevs = 0;
965  for (int lev = 0; lev < info.max_coarsening_level; ++lev)
966  {
967  IntVect rr_level = mg_coarsen_ratio_v;
968  bool do_semicoarsening_level = info.do_semicoarsening
969  && numsclevs < info.max_semicoarsening_level;
970  if (do_semicoarsening_level
971  && info.semicoarsening_direction != -1)
972  {
973  rr_level[info.semicoarsening_direction] = 1;
974  }
975  IntVect is_coarsenable;
976  for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
977  IntVect rr_dir(1);
978  rr_dir[idim] = rr_vec[idim] * rr_level[idim];
979  is_coarsenable[idim] = dom0.coarsenable(rr_dir, mg_domain_min_width_v)
980  && a_grids[0].coarsenable(rr_dir, mg_box_min_width_v);
981  if (!is_coarsenable[idim] && do_semicoarsening_level
982  && info.semicoarsening_direction == -1)
983  {
984  is_coarsenable[idim] = true;
985  rr_level[idim] = 1;
986  }
987  }
988  if (is_coarsenable != IntVect(1) || rr_level == IntVect(1)) {
989  break;
990  }
991  if (do_semicoarsening_level && info.semicoarsening_direction == -1) {
992  // make sure there is at most one direction that is not coarsened
993  int n_ones = AMREX_D_TERM( static_cast<int>(rr_level[0] == 1),
994  + static_cast<int>(rr_level[1] == 1),
995  + static_cast<int>(rr_level[2] == 1));
996  if (n_ones > 1) { break; }
997  }
998  if (rr_level != mg_coarsen_ratio_v) {
999  ++numsclevs;
1000  }
1001  rr_vec *= rr_level;
1002 
1003  m_geom[0].emplace_back(amrex::coarsen(dom0, rr_vec), rb, coord, is_per);
1004  m_grids[0].push_back(amrex::coarsen(a_grids[0], rr_vec));
1005 
1006  if (info.do_consolidation)
1007  {
1008  if (avg_npts/static_cast<Real>(AMREX_D_TERM(rr_vec[0], *rr_vec[1], *rr_vec[2]))
1009  < Real(0.999)*static_cast<Real>(consolidation_threshold))
1010  {
1011  coned = true;
1012  con_lev = m_dmap[0].size();
1013  m_dmap[0].push_back(DistributionMapping());
1014  }
1015  else
1016  {
1017  m_dmap[0].push_back(m_dmap[0].back());
1018  }
1019  }
1020  else
1021  {
1022  m_dmap[0].push_back(a_dmap[0]);
1023  }
1024  }
1025  }
1026 
1027  m_num_mg_levels[0] = m_grids[0].size();
1028 
1029  for (int mglev = 0; mglev < m_num_mg_levels[0] - 1; mglev++){
1030  const Box& fine_domain = m_geom[0][mglev].Domain();
1031  const Box& crse_domain = m_geom[0][mglev+1].Domain();
1032  mg_coarsen_ratio_vec.push_back(fine_domain.length()/crse_domain.length());
1033  }
1034 
1035  for (int amrlev = 0; amrlev < m_num_amr_levels; ++amrlev) {
1036  if (AMRRefRatio(amrlev) == 4 && mg_coarsen_ratio_vec.empty()) {
1037  mg_coarsen_ratio_vec.push_back(IntVect(2));
1038  }
1039  }
1040 
1041  if (agged)
1042  {
1043  makeAgglomeratedDMap(m_grids[0], m_dmap[0]);
1044  }
1045  else if (coned)
1046  {
1047  makeConsolidatedDMap(m_grids[0], m_dmap[0], info.con_ratio, info.con_strategy);
1048  }
1049 
1050  if (agged || coned)
1051  {
1052  m_bottom_comm = makeSubCommunicator(m_dmap[0].back());
1053  }
1054  else
1055  {
1056  m_bottom_comm = m_default_comm;
1057  }
1058 
1059  m_do_agglomeration = agged;
1060  m_do_consolidation = coned;
1061 
1062  if (verbose > 1) {
1063  if (agged) {
1064  Print() << "MLLinOp::defineGrids(): agglomerated AMR level 0 starting at MG level "
1065  << agg_lev << " of " << m_num_mg_levels[0] << "\n";
1066  } else if (coned) {
1067  Print() << "MLLinOp::defineGrids(): consolidated AMR level 0 starting at MG level "
1068  << con_lev << " of " << m_num_mg_levels[0]
1069  << " (ratio = " << info.con_ratio << ")" << "\n";
1070  } else {
1071  Print() << "MLLinOp::defineGrids(): no agglomeration or consolidation of AMR level 0\n";
1072  }
1073  }
1074 
1075  for (int amrlev = 0; amrlev < m_num_amr_levels; ++amrlev)
1076  {
1077  for (int mglev = 1; mglev < m_num_mg_levels[amrlev]; ++mglev)
1078  {
1079  m_factory[amrlev].emplace_back(makeFactory(amrlev,mglev));
1080  }
1081  }
1082 
1083  for (int amrlev = 1; amrlev < m_num_amr_levels; ++amrlev)
1084  {
1085  AMREX_ASSERT_WITH_MESSAGE(m_grids[amrlev][0].coarsenable(m_amr_ref_ratio[amrlev-1]),
1086  "MLLinOp: grids not coarsenable between AMR levels");
1087  }
1088 }
