315template <typename T, typename F>
316AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
317std::enable_if_t<std::is_same_v<std::decay_t<decltype(std::declval<T>().box())>, Box>, void>
318tagparfor_call_f (int icell, T const& tag, F&& f) noexcept
319{
320 int ncells = tag.box().numPts();
321 const auto len = amrex::length(tag.box());
322 const auto lo = amrex::lbound(tag.box());
323 int k = icell / (len.x*len.y);
324 int j = (icell - k*(len.x*len.y)) / len.x;
325 int i = (icell - k*(len.x*len.y)) - j*len.x;
326 i += lo.x;
327 j += lo.y;
328 k += lo.z;
329 f(icell, ncells, i, j, k, tag);
330}

343template <class TagType, class F>
344void
345ParallelFor_doit (TagVector<TagType> const& tv, F const& f)
346{
347 AMREX_ALWAYS_ASSERT(tv.is_defined());
348
349 if (tv.ntags == 0) { return; }
350
351 const auto d_tags = tv.d_tags;
352 const auto d_nwarps = tv.d_nwarps;
353 const auto ntags = tv.ntags;
354 const auto ntotwarps = tv.ntotwarps;
355 constexpr auto nthreads = TagVector<TagType>::nthreads;
356
357 amrex::launch<nthreads>(tv.nblocks, Gpu::gpuStream(),
358#ifdef AMREX_USE_SYCL
359 [=] AMREX_GPU_DEVICE (sycl::nd_item<1> const& item) noexcept
360 [[sycl::reqd_work_group_size(nthreads)]]
361 [[sycl::reqd_sub_group_size(Gpu::Device::warp_size)]]
362#else
363 [=] AMREX_GPU_DEVICE () noexcept
364#endif
365 {
366#ifdef AMREX_USE_SYCL
367 std::size_t g_tid = item.get_global_id(0);
368#else
369 auto g_tid = std::size_t(blockDim.x)*blockIdx.x + threadIdx.x;
370#endif
371 auto g_wid = int(g_tid / Gpu::Device::warp_size);
372 if (g_wid >= ntotwarps) { return; }
373
374 int tag_id = amrex::bisect(d_nwarps, 0, ntags, g_wid);
375
376 int b_wid = g_wid - d_nwarps[tag_id]; // b_wid'th warp on this box
377#ifdef AMREX_USE_SYCL
378 int lane = item.get_local_id(0) % Gpu::Device::warp_size;
379#else
380 int lane = threadIdx.x % Gpu::Device::warp_size;
381#endif
382 int icell = b_wid*Gpu::Device::warp_size + lane;
383
384 tagparfor_call_f(icell, d_tags[tag_id], f);
385 });
386}

390template <class TagType, class F>
391void
392ParallelFor_doit (TagVector<TagType> const& tv, F const& f)
393{
394 // Note: this CPU version may not have optimal performance:
395 // The loop over ncomp is the innermost instead of the outermost
396 // There is no load-balancing or splitting of tags
397 AMREX_ALWAYS_ASSERT(tv.is_defined());
398
399 constexpr bool tag_type = is_box_tag(TagType{});
400
401 if (tv.ntags == 0) { return; }
402
403 const auto d_tags = tv.d_tags;
404 const auto ntags = tv.ntags;
405
406#ifdef AMREX_USE_OMP
407#pragma omp parallel for
408#endif
409 for (int itag = 0; itag < ntags; ++itag) {
410
411 const auto& t = d_tags[itag];
412
413 if constexpr (tag_type) {
414 const auto lo = amrex::lbound(t.box());
415 const auto hi = amrex::ubound(t.box());
416
417 for (int k = lo.z; k <= hi.z; ++k) {
418 for (int j = lo.y; j <= hi.y; ++j) {
419 AMREX_PRAGMA_SIMD
420 for (int i = lo.x; i <= hi.x; ++i) {
421 f(0, 1, i, j, k, t);
422 }
423 }
424 }
425 } else {
426 const auto size = t.size();
427
428 AMREX_PRAGMA_SIMD
429 for (int i = 0; i < size; ++i) {
430 f(i, size, t);
431 }
432 }
433 }
434}
435
436#endif

438}
