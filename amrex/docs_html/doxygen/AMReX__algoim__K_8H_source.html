116  // Determines the sign conditions for restricting a (possibly
117  // already restricted) level set function, i.e., sgn_L and sgn_U
118  // in [R. Saye, High-Order Quadrature Methods for Implicitly
119  // Defined Surfaces and Volumes in Hyperrectangles, SIAM
120  // J. Sci. Comput., Vol. 37, No. 2, pp. A993-A1019,
121  // http://dx.doi.org/10.1137/140966290].
122  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
123  void
124  determineSigns (bool positiveAbove, int sign, int& bottomSign, int& topSign) noexcept
125  {
126  // When evaluating a volume integral:
127  // If 0 integrating over the positive part:
128  // if positive above the height function: bottom = +/-, top = +
129  // if positive below the height function: bottom = +, top = +/-
130  // If integrating over the negative part:
131  // if positive above the height function: bottom = -, top = +/-
132  // if positive below the height function: bottom = +/-, top = -
133  // If integrating over both the positive and negative part (i.e. unrestricted in sign), keep it alive
134  if (sign == 1)
135  {
136  bottomSign = positiveAbove? 0 : 1;
137  topSign = positiveAbove? 1 : 0;
138  }
139  else if (sign == -1)
140  {
141  bottomSign = positiveAbove? -1 : 0;
142  topSign = positiveAbove? 0 : -1;
143  }
144  else {
145  bottomSign = topSign = 0;
146  }
147  }
148 } // namespace detail

150 // PsiCode encodes sign information of restricted level set functions
151 // on particular sides of a hyperrectangle in a packed array of
152 // bits. The first N bits encodes side information, the N+1st bit is
153 // true iff the sign == 0, while the N+2nd bit stores the sign if sign
154 // != 0.
155 template<int N>
156 struct PsiCode
157 {
158  unsigned char bits = 0;
159 
160  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE constexpr
161  PsiCode () noexcept {}
162 
163  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
164  PsiCode (const GpuArray<int,N>& sides, int sign) noexcept
165  {
166  static_assert(N <= 3, "algoim::PsiCode: N must be <= 3");
167  for (int dim = 0; dim < N; ++dim) {
168  if (sides[dim] == 1) {
169  bits |= (1 << dim);
170  }
171  }
172  if (sign == 0) {
173  bits |= (1 << N);
174  } else {
175  bits &= ~(1 << N);
176  if (sign == 1) {
177  bits |= (1 << (N+1));
178  }
179  }
180  }
181 
182  // Modify an existing code by restriction in a particular dimension.
183  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
184  PsiCode (const PsiCode& i, int dim, int side, int sign) noexcept
185  : bits(i.bits)
186  {
187  if (side == 1) {
188  bits |= (1 << dim);
189  }
190  if (sign == 0) {
191  bits |= (1 << N);
192  } else {
193  bits &= ~(1 << N);
194  if (sign == 1) {
195  bits |= (1 << (N+1));
196  }
197  }
198  }
199 
200  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
201  int side (int dim) const noexcept
202  {
203  return bits & (1 << dim);
204  }
205 
206  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
207  int sign () const noexcept
208  {
209  return (bits & (1 << N)) ? 0 : ((bits & (1 << (N+1))) ? 1 : -1);
210  }
211 };

238 // M-dimensional integral of an N-dimensional function restricted to
239 // given implicitly defined domains
240 template<int M, int N, typename Phi, typename F>
241 struct ImplicitIntegral
242 {
243  const Phi& phi;
244  F& f;
245  const GpuArray<bool,N> free;
246  GpuArray<PsiCode<N>,1 << (N - 1)> psi;
247  int psiCount;
248  const BoundingBox<N> xrange;
249  int e0;
250  GpuArray<Interval,N> xint;
251  static constexpr int p = 4;

253  // Prune the given set of functions by checking for the existence
254  // of the interface. If a function is uniformly positive or
255  // negative and is consistent with specified sign, it can be
256  // removed. If a function is uniformly positive or negative but
257  // inconsistent with specified sign, the domain of integration is
258  // empty.
259  AMREX_GPU_HOST_DEVICE
260  bool prune () noexcept
261  {
262 #if !defined(__CUDACC__) || (__CUDACC_VER_MAJOR__ != 9) || (__CUDACC_VER_MINOR__ != 2)
263  static constexpr Real eps = std::numeric_limits<Real>::epsilon();
264 #else
265  static constexpr Real eps = DBL_EPSILON;
266 #endif
267  static constexpr Real almostone = 1.0-10.*eps;
268  for (int i = 0; i < psiCount; )
269  {
270  GpuArray<Real,N> mid = xrange.midpoint();
271  Real dphi_max = 0.0;
272  for (int dim = 0; dim < N; ++dim) {
273  if (free[dim]) {
274  dphi_max += std::abs(phi.grad(dim));
275  } else {
276  mid[dim] = xrange(psi[i].side(dim),dim);
277  }
278  }
279  dphi_max *= 0.5*almostone;
280  const Real phi_0 = phi(mid);
281  bool uniform_sign = (phi_0 > dphi_max) or (phi_0 < -dphi_max);
282  if (uniform_sign)
283  {
284  if ((phi_0 >= 0.0 and psi[i].sign() >= 0) or
285  (phi_0 <= 0.0 and psi[i].sign() <= 0) )
286  {
287  --psiCount;
288  detail::swap(psi[i], psi[psiCount]);
289  }
290  else {
291  return false;
292  }
293  }
294  else {
295  ++i;
296  }
297  }
298  return true;
299  }

301  // Gaussian quadrature for when the domain of integration is
302  // determined to be the entire M-dimensional cube.
303  AMREX_GPU_HOST_DEVICE
304  void tensorProductIntegral () noexcept
305  {
306  constexpr Real gauss_x[4]={0.069431844202973712388026755553595247452,
307  0.33000947820757186759866712044837765640,
308  0.66999052179242813240133287955162234360,
309  0.93056815579702628761197324444640475255};
310  constexpr Real gauss_w[4]={0.173927422568726928686531974610999703618,
311  0.326072577431273071313468025389000296382,
312  0.326072577431273071313468025389000296382,
313  0.173927422568726928686531974610999703618};
314 
315  int nloops = 1;
316  for (int i = 0; i < M; ++i) {
317  nloops *= p;
318  }
319  GpuArray<int,M> i{}; // i is initialized to zero.
320  for (int iloop = 0; iloop < nloops; ++iloop)
321  {
322  GpuArray<Real,N> x{};
323  Real w = 1.0;
324  for (int dim = 0, k = 0; dim < N; ++dim) {
325  if (free[dim]) {
326  x[dim] = xrange.min(dim) + xrange.extent(dim) * gauss_x[i[k]];
327  w *= xrange.extent(dim) * gauss_w[i[k]];
328  ++k;
329  }
330  }
331  f.evalIntegrand(x, w);
332 
333  for (int m = M-1; m >= 0; --m) {
334  ++(i[m]);
335  if (i[m] < p) {
336  break;
337  } else {
338  i[m] = 0;
339  }
340  }
341  }
342  }

344  // Given x, valid in all free variables barring e0, root find in
345  // the e0 direction on each of the implicit functions, and apply
346  // Gaussian quadrature to each segment. Weights are multiplied
347  // upon going back up the tree of recursive calls.
348  AMREX_GPU_HOST_DEVICE
349  void evalIntegrand (GpuArray<Real,N> x, Real w) const noexcept
350  {
351  constexpr Real gauss_x[4]={0.069431844202973712388026755553595247452,
352  0.33000947820757186759866712044837765640,
353  0.66999052179242813240133287955162234360,
354  0.93056815579702628761197324444640475255};
355  constexpr Real gauss_w[4]={0.173927422568726928686531974610999703618,
356  0.326072577431273071313468025389000296382,
357  0.326072577431273071313468025389000296382,
358  0.173927422568726928686531974610999703618};
359 
360  // Each thread has its own storage for computing roots. These
361  // are not corrupted by the recursive chain of evalIntegrand()
362  // calls since each call is in a different templated
363  // namespace. Moreover, if using OpenMP tasking, each task is
364  // assumed tied and so one thread can only execute
365  // "evalIntegrand" from start to end uninterrupted.
366  GpuArray<Real,4> roots;
367 
368  // Partition [xmin(e0), xmax(e0)] by roots found
369  Real x_min = xrange.min(e0);
370  Real x_max = xrange.max(e0);
371  roots[0] = x_min;
372  int nroots = 1;
373  for (int i = 0; i < psiCount; ++i)
374  {
375  for (int dim = 0; dim < N; ++dim) {
376  if (!free[dim]) {
377  x[dim] = xrange(psi[i].side(dim),dim);
378  }
379  }
380  // x is now valid in all variables except e0
381  x[e0] = x_min;
382  Real phi_lo = phi(x);
383  Real xroot = x[e0] - phi_lo / phi.grad(e0);
384  if (xroot > x_min and xroot < x_max) {
385  roots[nroots++] = xroot;
386  }
387  }
388  AMREX_ASSERT(nroots <= 3);
389  if (nroots == 3 and roots[1] > roots[2]) {
390  detail::swap(roots[1],roots[2]);
391  }
392  roots[nroots++] = x_max;
393 
394  // In rare cases, degenerate segments can be found, filter out with a tolerance
395 #if !defined(__CUDACC__) || (__CUDACC_VER_MAJOR__ != 9) || (__CUDACC_VER_MINOR__ != 2)
396  static constexpr Real eps = std::numeric_limits<Real>::epsilon();
397 #else
398  static constexpr Real eps = DBL_EPSILON;
399 #endif
400  constexpr Real tol = 50.0*eps;
401 
402  // Loop over segments of divided interval
403  for (int i = 0; i < nroots - 1; ++i)
404  {
405  if (roots[i+1] - roots[i] < tol) continue;
406 
407  // Evaluate sign of phi within segment and check for consistency with psi
408  bool okay = true;
409  x[e0] = (roots[i] + roots[i+1])*0.5;
410  for (int j = 0; j < psiCount && okay; ++j)
411  {
412  for (int dim = 0; dim < N; ++dim) {
413  if (!free[dim]) {
414  x[dim] = xrange(psi[j].side(dim), dim);
415  }
416  }
417  bool new_ok = (phi(x) > 0.0) ? (psi[j].sign() >= 0) : (psi[j].sign() <= 0);
418  okay = okay and new_ok;
419  }
420  if (!okay) continue;
421 
422  for (int j = 0; j < p; ++j)
423  {
424  x[e0] = roots[i] + (roots[i+1] - roots[i]) * gauss_x[j];
425  Real gw = (roots[i+1] - roots[i]) * gauss_w[j];
426  f.evalIntegrand(x, w * gw);
427  }
428  }
429  }

431  // Main calling engine; parameters with underscores are copied
432  // upon entry but modified internally in the ctor
433  AMREX_GPU_HOST_DEVICE
434  ImplicitIntegral (const Phi& phi_, F& f_, const GpuArray<bool,N>& free_,
435  const GpuArray<PsiCode<N>,1 << (N-1)>& psi_,
436  int psiCount_) noexcept
437  : phi(phi_), f(f_), free(free_), psi(psi_), psiCount(psiCount_),
438  xrange()
439  {
440  // For the one-dimensional base case, evaluate the
441  // bottom-level integral.
442  if (M == 1)
443  {
444  for (int dim = 0; dim < N; ++dim) {
445  if (free[dim]) {
446  e0 = dim;
447  }
448  }
449  evalIntegrand(GpuArray<Real,N>{}, 1.0);
450  return;
451  }
452 
453  // Establish interval bounds for prune() and remaining part of ctor.
454  for (int dim = 0; dim < N; ++dim)
455  {
456  if (free[dim])
457  {
458  xint[dim].alpha = xrange.midpoint(dim);
459  }
460  else
461  {
462  xint[dim].alpha = 0.0;
463  }
464  }
465 
466  // Prune list of psi functions: if prune procedure returns
467  // false, then the domain of integration is empty.
468  if (!prune()) {
469  return;
470  }
471 
472  // If all psi functions were pruned, then the volumetric
473  // integral domain is the entire hyperrectangle.
474  if (psiCount == 0)
475  {
476  tensorProductIntegral();
477  return;
478  }
479 
480  // Among all monotone height function directions, choose the
481  // one that makes the associated height function look as flat
482  // as possible. This is a modification to the criterion
483  // presented in [R. Saye, High-Order Quadrature Methods for
484  // Implicitly Defined Surfaces and Volumes in Hyperrectangles,
485  // SIAM J. Sci. Comput., Vol. 37, No. 2, pp. A993-A1019,
486  // http://dx.doi.org/10.1137/140966290].
487  e0 = -1;
488  {
489  Real gmax = -1.;
490  for (int dim = 0; dim < N; ++dim) {
491  if (free[dim] && std::abs(phi.grad(dim)) > gmax) {
492  gmax = std::abs(phi.grad(dim));
493  e0 = dim;
494  }
495  }
496  }
497 
498  // Check compatibility with all implicit functions whilst
499  // simultaneously constructing new implicit functions.
500  GpuArray<PsiCode<N>,1 << (N-1)> newPsi;
501  int newPsiCount = 0;
502  for (int i = 0; i < psiCount; ++i)
503  {
504  // Evaluate gradient in an interval
505  for (int dim = 0; dim < N; ++dim) {
506  if (!free[dim]) {
507  xint[dim].alpha = xrange(psi[i].side(dim), dim);
508  }
509  }
510 
511  // w.z. We assume grad does not change.
512  int bottomSign, topSign;
513  detail::determineSigns(phi.grad(e0) > 0.0, psi[i].sign(),
514  bottomSign, topSign);
515  // w.z. There are sides 0 and 1.
516  newPsi[newPsiCount++] = PsiCode<N>(psi[i], e0, 0, bottomSign);
517  newPsi[newPsiCount++] = PsiCode<N>(psi[i], e0, 1, topSign);
518  }
519 
520  // Dimension reduction call
521  GpuArray<bool,N> new_free = free;
522  new_free[e0] = false;
523  ImplicitIntegral<M-1,N,Phi,ImplicitIntegral<M,N,Phi,F> >
524  (phi, *this, new_free, newPsi, newPsiCount);
525  }
