391  std::unique_ptr<MultiFab> fill_impfunc() {
392  std::unique_ptr<MultiFab> mf_impfunc = std::unique_ptr<MultiFab>(new MultiFab);
393  mf_impfunc->define(m_ls_ba, m_ls_dm, 1, m_ls_pad);
394 
395  Box bounding_box = m_geom.Domain();
396  bounding_box.surroundingNodes();
397  bool extend_domain_face = EB2::ExtendDomainFace();
398  for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
399  if (!extend_domain_face || m_geom.isPeriodic(idim)) {
400  bounding_box.grow(m_ls_pad);

402  }

404 #ifdef _OPENMP

406 #endif
407  for(MFIter mfi(* mf_impfunc); mfi.isValid(); ++ mfi) {
408  m_gshop.fillFab((* mf_impfunc)[mfi], m_geom, RunOn::Gpu, bounding_box);
409  }
