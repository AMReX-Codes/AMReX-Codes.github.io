30 //
31 // ParmParse class implements a simple database for the storage and
32 // retrieval of command-line and input-file arguments. The entries are
33 // stored in a static table in (name,value_list) pairs.
34 //
35 // The format of the input file is a series of DEFINITIONS.
36 //
37 // A DEFINITION is of the form <name> = <value> <value> ...
38 // The equal sign is important since the list of values can span multiple
39 // lines.
40 //
41 // Comments in an input file include all text from a '#' character to the
42 // end of the line. Here is an example input file:
43 /*
44  niter = 100 # niter is an integer
45  title = "Double Wammy" # example of a string with spaces
46  cell_size = 0.5 0.75 # cell spacing in each dimension
47  plot.var = Density 1 10 # a list of values
48  plot.var = Energy 5 12 # another list of values
49  bigarray = 1 2 3 4 5 6 7 8 \
50  9 10 11 12 # continuation of bigarray
51  multi_line_string = "This is a
52  multi-line string."
53  test = apple "boy blue" 10 20 30 40
54  FILE = prob_file # insert contents of this "prob_file" here
55 */
56 // For values spanning multiple lines, one must use '\' at the end of a line
57 // for continuation, otherwise it's a runtime error. Note that there must be
58 // at least one space before the continuation character `\`. Multiple lines
59 // inside a pair of double quotes are considered a single string containing
60 // '\n's. The "FILE = <filename>" definition is special. Rather than just
61 // adding this entry to the database, it reads the contents of <filename>
62 // into the database.
63 // For CI/CD workflows and out-of-source tests, the environment variable
64 // AMREX_INPUTS_FILE_PREFIX can be set to prefix every FILE = <filename>
65 // with a custom path.
66 //
67 // ParmParse stores all entries in a static table which is built the
68 // first time a ParmParse object is constructed (usually in main()).
69 // Subsequent invocations have access to this table.
70 // A ParmParse constructor has an optional "prefix" argument that will
71 // limit the searches to only those entries of the table with this prefix
72 // in name. For example:
73 // ParmParse pp("plot");
74 // will find only those entries with name given by "plot.<string>".
75 //
76 // All values in the table are stored as strings. For example, the
77 // values of "cell_size" in the above input file are stored as the
78 // strings "0.5" and "0.75". These strings can be returned as either
79 // strings or numeric values by the query functions.
80 // Character strings with spaces must be delimited by double quotes
81 // in the input file but the quotes are stripped before they are entered
82 // into the table. For example, 'title' in the above input file has a
83 // single value, the string 'Double Wammy' (without the quotes).
84 // Each value in the list associated with a definition can be referred to
85 // by its index number. The index numbers start at 0 just like an array
86 // in the C programming language. Consider the definition of "test" in
87 // the above input file. The first value 'apple'is a string with index
88 // 0. The second value 'boy blue' is a string with index 1. The
89 // remaining four values are integers indexed 2, 3, 4, and 5.
90 //
91 // For a string value to represent an integer or float it must fit the
92 // following regular expression:
93 // Sign ::= '+' | '-'
94 // Digit ::= '0' | '1' | ... | '9'
95 // Integer ::= [Sign]Digit+
96 // Exp ::= ('e'|'E')Integer
97 // Float ::= ( Integer[.Digit*][Exp] | [Integer].Digit+[Exp] )
98 //
99 // Where '+' indicates one or more occurrences, '*' represents zero or
100 // more occurrences, '|' means one or the other and '[]' represents zero
101 // or one occurrence.
102 //
103 // Note that floats and doubles have the same string representation and
104 // that the FORTRAN "double" exponent format is not supported.
105 // That is, 1.0d+3 is not a valid representation of a floating point
106 // number but that 1.0e+3 is acceptable.
107 //
108 // There are a host of functions allowing the user to query the database
109 // and retrieve values. Here are some general rules about the names of
110 // the member functions:
111 //
112 // * Functions with the string "get" in their names attempt to get a
113 // value or an array of values from the table. They generate a
114 // run-time error if they are not successful.
115 //
116 // * Functions with the string "query" in their names attempt to get a
117 // value or an array of values from the table. They return the value 1
118 // (true) if they are successful and 0 (false) if not.
119 //
120 // * Functions with the string "kth" in their names get values from
121 // the Kth entry with the given name in the database. This is
122 // necessary since there may be multiple definitions with the same
123 // name in the database.
124 //
125 // * Functions without the string "kth" in their names get values from
126 // the last entry with the given name in the database. Note that the
127 // definitions from the command line are appended to the database table
128 // and hence will be the last entries.
129 //
130 // * Functions with the string "arr" in their names get an Array of
131 // values from the given entry in the table. The array argument is
132 // resized (if necessary) to hold all the values requested.
133 //
134 // * Functions without the string "arr" in their names get single
135 // values from the given entry in the table.
136 //
137 // The following is a code sample showing how to use ParmParse:
138 //
139 // main(int argc, char **argv)
140 // {
141 // char* in_file_name = argv[1];
142 // ParmParse::Initialize(argc-2, argv+2, in_file_name);
143 //
144 // // Query table for value of "niter". If not in table
145 // // then set to default value
146 // if (!pp.query("niter",niter)) niter = 20;
147 //
148 // // read array of cell sizes if in table
149 // Vector<float> dx;
150 // if (nx=pp.countval("cell_size")) {
151 // // get nx values starting at index 0 and store in dx.
152 // // dx is automatically resized here.
153 // pp.getarr("cell_size",dx,0,nx);
154 // }
155 // ParmParse::Finalize();
156 // }
157 //
158 // void do_graphics()
159 // {
160 // //
161 // // Will only query entries with the "plot" prefix:
162 // //
163 // ParmParse pp("plot");
164 // //
165 // // Read all variables with "plot.var" keyword.
166 // //
167 // std::string var_name;
168 // Vector<int> range;
169 // int num = pp.countname("var");
170 // for (int k = 0; k < num; k++)
171 // {
172 // //
173 // // Element 0 in list is a string.
174 // //
175 // pp.getkth("var",k,var_name,0);
176 // //
177 // // Elements 1 and 2 are integers.
178 // // Note that "range" will be resized to hold 2 elements.
179 // //
180 // pp.getktharr("var",k,range,1,2);
181 // cout << "variable = " << var_name << "lo, hi = ",
182 // << range[0] << " " << range[1] << endl;
183 // }
184 // }
185 // -----------------------------------------------------------------
186 // ----------------------- END COMMENTS ---------------------------
187 // -----------------------------------------------------------------
