19 //
20 // ParmParse class implements a simple database for the storage and
21 // retrieval of command-line and input-file arguments. The entries are
22 // stored in a static table in (name,value_list) pairs.
23 //
24 // The format of the input file is a series of DEFINITIONS.
25 //
26 // A DEFINITION is of the form <name> = <value> <value> ...
27 // The equal sign is important since the list of values can span multiple
28 // lines.
29 //
30 // Comments in an input file include all text from a '#' character to the
31 // end of the line. Here is an example input file:
32 //
33 // niter = 100 # niter is an integer
34 // title = "Double Wammy" # example of a string with spaces
35 // cell_size = 0.5 0.75 # cell spacing in each dimension
36 // plot.var = Density 1 10 # a list of values
37 // plot.var = Energy 5 12 # another list of values
38 // bigarray = 1 2 3 4 5 6 7 8 # first part of array
39 // 9 10 11 12 # continuation of bigarray
40 // test = apple "boy blue" 10 20 30 40
41 // FILE = prob_file # insert contents of this "prob_file" here
42 //
43 // The "FILE = <filename>" definition is special. Rather than just
44 // adding this entry to the database, it reads the contents of <filename>
45 // into the database.
46 //
47 // ParmParse stores all entries in a static table which is built the
48 // first time a ParmParse object is constructed (usually in main()).
49 // Subsequent invocations have access to this table.
50 // A ParmParse constructor has an optional "prefix" argument that will
51 // limit the searches to only those entries of the table with this prefix
52 // in name. For example:
53 // ParmParse pp("plot");
54 // will find only those entries with name given by "plot.<string>".
55 //
56 // All values in the table are stored as strings. For example, the
57 // values of "cell_size" in the above input file are stored as the
58 // strings "0.5" and "0.75". These strings can be returned as either
59 // strings or numeric values by the query functions.
60 // Character strings with spaces must be delimited by double quotes
61 // in the input file but the quotes are stripped before they are entered
62 // into the table. For example, 'title' in the above input file has a
63 // single value, the string 'Double Wammy' (without the quotes).
64 // Each value in the list associated with a definition can be referred to
65 // by its index number. The index numbers start at 0 just like an array
66 // in the C programming language. Consider the definition of "test" in
67 // the above input file. The first value 'apple'is a string with index
68 // 0. The second value 'boy blue' is a string with index 1. The
69 // remaining four values are integers indexed 2, 3, 4, and 5.
70 //
71 // For a string value to represent an integer or float it must fit the
72 // following regular experssion:
73 // Sign ::= '+' | '-'
74 // Digit ::= '0' | '1' | ... | '9'
75 // Integer ::= [Sign]Digit+
76 // Exp ::= ('e'|'E')Integer
77 // Float ::= ( Integer[.Digit*][Exp] | [Integer].Digit+[Exp] )
78 //
79 // Where '+' indicates one or more occurences, '*' represents zero or
80 // more occurences, '|' means one or the other and '[]' represents zero
81 // or one occurence.
82 //
83 // Note that floats and doubles have the same string representation and
84 // that the FORTRAN "double" exponent format is not supported.
85 // That is, 1.0d+3 is not a valid representation of a floating point
86 // number but that 1.0e+3 is acceptable.
87 //
88 // There are a host of functions allowing the user to query the database
89 // and retrieve values. Here are some general rules about the names of
90 // the member functions:
91 //
92 // * Functions with the string "get" in their names attempt to get a
93 // value or an array of values from the table. They generate a
94 // run-time error if they are not successful.
95 //
96 // * Functions with the string "query" in their names attempt to get a
97 // value or an array of values from the table. They return the value 1
98 // (true) if they are successful and 0 (false) if not.
99 //
100 // * Functions with the string "kth" in their names get values from
101 // the Kth entry with the given name in the database. This is
102 // necessary since there may be multiple definitions with the same
103 // name in the database.
104 //
105 // * Functions without the string "kth" in their names get values from
106 // the last entry with the given name in the database. Note that the
107 // definitions from the command line are appended to the database table
108 // and hence will be the last entries.
109 //
110 // * Functions with the string "arr" in their names get an Array of
111 // values from the given entry in the table. The array argument is
112 // resized (if necessary) to hold all the values requested.
113 //
114 // * Functions without the string "arr" in their names get single
115 // values from the given entry in the table.
116 //
117 // The following is a code sample showing how to use ParmParse:
118 //
119 // main(int argc, char **argv)
120 // {
121 // char* in_file_name = argv[1];
122 // ParmParse::Initialize(argc-2, argv+2, in_file_name);
123 //
124 // // Query table for value of "niter". If not in table
125 // // then set to default value
126 // if (!pp.query("niter",niter)) niter = 20;
127 //
128 // // read array of cell sizes if in table
129 // Vector<float> dx;
130 // if (nx=pp.countval("cell_size")) {
131 // // get nx values starting at index 0 and store in dx.
132 // // dx is automatically resized here.
133 // pp.getarr("cell_size",dx,0,nx);
134 // }
135 // ParmParse::Finalize();
136 // }
137 //
138 // void do_graphics()
139 // {
140 // //
141 // // Will only query entries with the "plot" prefix:
142 // //
143 // ParmParse pp("plot");
144 // //
145 // // Read all variables with "plot.var" keyword.
146 // //
147 // std::string var_name;
148 // Vector<int> range;
149 // int num = pp.countname("var");
150 // for (int k = 0; k < num; k++)
151 // {
152 // //
153 // // Element 0 in list is a string.
154 // //
155 // pp.getkth("var",k,var_name,0);
156 // //
157 // // Elements 1 and 2 are integers.
158 // // Note that "range" will be resized to hold 2 elements.
159 // //
160 // pp.getktharr("var",k,range,1,2);
161 // cout << "variable = " << var_name << "lo, hi = ",
162 // << range[0] << " " << range[1] << endl;
163 // }
164 // }
165 // -----------------------------------------------------------------
166 // ----------------------- END COMMENTS ---------------------------
167 // -----------------------------------------------------------------

889  void get (const char* name, std::array<T,N>& ref) const {
890  std::vector<T> v;
891  this->getarr(name, v);
892  AMREX_ALWAYS_ASSERT(v.size() >= N);
893  for (std::size_t i = 0; i < N; ++i) {
894  ref[i] = v[i];
895  }
896  }

898  template <typename T, std::size_t N>
899  int query (const char* name, std::array<T,N>& ref) const {
900  std::vector<T> v;
901  int exist = this->queryarr(name, v);
902  if (exist) {
903  AMREX_ALWAYS_ASSERT(v.size() >= N);
904  for (std::size_t i = 0; i < N; ++i) {
905  ref[i] = v[i];
906  }
907  }
908  return exist;
909  }

void getktharr(const char *name, int k, std::vector< int > &ref, int start_ix=FIRST, int num_val=ALL) const
Gets an std::vector<int> of num_val values from kth occurrence of given name. If successful, the values are converted to an int and stored in the std::vector<int> object ref. ref is resized (if necessary) to hold num_val values. The value in the list indexed by start_ix is copied into std::vector<int>[0], std::vector<int>[1] holds start_ix+1, etc. If the kth occurrence does not exist or there are fewer than start_ix + num_val values associated with the kth occurrence, or if some of the values cannot be converted to an int, an error message is reported and the program halts.
