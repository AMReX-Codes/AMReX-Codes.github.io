22 //
23 // ParmParse class implements a simple database for the storage and
24 // retrieval of command-line and input-file arguments. The entries are
25 // stored in a static table in (name,value_list) pairs.
26 //
27 // The format of the input file is a series of DEFINITIONS.
28 //
29 // A DEFINITION is of the form <name> = <value> <value> ...
30 // The equal sign is important since the list of values can span multiple
31 // lines.
32 //
33 // Comments in an input file include all text from a '#' character to the
34 // end of the line. Here is an example input file:
35 /*
36  niter = 100 # niter is an integer
37  title = "Double Wammy" # example of a string with spaces
38  cell_size = 0.5 0.75 # cell spacing in each dimension
39  plot.var = Density 1 10 # a list of values
40  plot.var = Energy 5 12 # another list of values
41  bigarray = 1 2 3 4 5 6 7 8 \
42  9 10 11 12 # continuation of bigarray
43  multi_line_string = "This is a
44  multi-line string."
45  test = apple "boy blue" 10 20 30 40
46  FILE = prob_file # insert contents of this "prob_file" here
47 */
48 // For values spanning multiple lines, one must use '\' at the end of a line
49 // for continuation, otherwise it's a runtime error. Note that there must be
50 // at least one space before the continuation character `\`. Multiple lines
51 // inside a pair of double quotes are considered a single string containing
52 // '\n's. The "FILE = <filename>" definition is special. Rather than just
53 // adding this entry to the database, it reads the contents of <filename>
54 // into the database.
55 //
56 // ParmParse stores all entries in a static table which is built the
57 // first time a ParmParse object is constructed (usually in main()).
58 // Subsequent invocations have access to this table.
59 // A ParmParse constructor has an optional "prefix" argument that will
60 // limit the searches to only those entries of the table with this prefix
61 // in name. For example:
62 // ParmParse pp("plot");
63 // will find only those entries with name given by "plot.<string>".
64 //
65 // All values in the table are stored as strings. For example, the
66 // values of "cell_size" in the above input file are stored as the
67 // strings "0.5" and "0.75". These strings can be returned as either
68 // strings or numeric values by the query functions.
69 // Character strings with spaces must be delimited by double quotes
70 // in the input file but the quotes are stripped before they are entered
71 // into the table. For example, 'title' in the above input file has a
72 // single value, the string 'Double Wammy' (without the quotes).
73 // Each value in the list associated with a definition can be referred to
74 // by its index number. The index numbers start at 0 just like an array
75 // in the C programming language. Consider the definition of "test" in
76 // the above input file. The first value 'apple'is a string with index
77 // 0. The second value 'boy blue' is a string with index 1. The
78 // remaining four values are integers indexed 2, 3, 4, and 5.
79 //
80 // For a string value to represent an integer or float it must fit the
81 // following regular expression:
82 // Sign ::= '+' | '-'
83 // Digit ::= '0' | '1' | ... | '9'
84 // Integer ::= [Sign]Digit+
85 // Exp ::= ('e'|'E')Integer
86 // Float ::= ( Integer[.Digit*][Exp] | [Integer].Digit+[Exp] )
87 //
88 // Where '+' indicates one or more occurrences, '*' represents zero or
89 // more occurrences, '|' means one or the other and '[]' represents zero
90 // or one occurrence.
91 //
92 // Note that floats and doubles have the same string representation and
93 // that the FORTRAN "double" exponent format is not supported.
94 // That is, 1.0d+3 is not a valid representation of a floating point
95 // number but that 1.0e+3 is acceptable.
96 //
97 // There are a host of functions allowing the user to query the database
98 // and retrieve values. Here are some general rules about the names of
99 // the member functions:
100 //
101 // * Functions with the string "get" in their names attempt to get a
102 // value or an array of values from the table. They generate a
103 // run-time error if they are not successful.
104 //
105 // * Functions with the string "query" in their names attempt to get a
106 // value or an array of values from the table. They return the value 1
107 // (true) if they are successful and 0 (false) if not.
108 //
109 // * Functions with the string "kth" in their names get values from
110 // the Kth entry with the given name in the database. This is
111 // necessary since there may be multiple definitions with the same
112 // name in the database.
113 //
114 // * Functions without the string "kth" in their names get values from
115 // the last entry with the given name in the database. Note that the
116 // definitions from the command line are appended to the database table
117 // and hence will be the last entries.
118 //
119 // * Functions with the string "arr" in their names get an Array of
120 // values from the given entry in the table. The array argument is
121 // resized (if necessary) to hold all the values requested.
122 //
123 // * Functions without the string "arr" in their names get single
124 // values from the given entry in the table.
125 //
126 // The following is a code sample showing how to use ParmParse:
127 //
128 // main(int argc, char **argv)
129 // {
130 // char* in_file_name = argv[1];
131 // ParmParse::Initialize(argc-2, argv+2, in_file_name);
132 //
133 // // Query table for value of "niter". If not in table
134 // // then set to default value
135 // if (!pp.query("niter",niter)) niter = 20;
136 //
137 // // read array of cell sizes if in table
138 // Vector<float> dx;
139 // if (nx=pp.countval("cell_size")) {
140 // // get nx values starting at index 0 and store in dx.
141 // // dx is automatically resized here.
142 // pp.getarr("cell_size",dx,0,nx);
143 // }
144 // ParmParse::Finalize();
145 // }
146 //
147 // void do_graphics()
148 // {
149 // //
150 // // Will only query entries with the "plot" prefix:
151 // //
152 // ParmParse pp("plot");
153 // //
154 // // Read all variables with "plot.var" keyword.
155 // //
156 // std::string var_name;
157 // Vector<int> range;
158 // int num = pp.countname("var");
159 // for (int k = 0; k < num; k++)
160 // {
161 // //
162 // // Element 0 in list is a string.
163 // //
164 // pp.getkth("var",k,var_name,0);
165 // //
166 // // Elements 1 and 2 are integers.
167 // // Note that "range" will be resized to hold 2 elements.
168 // //
169 // pp.getktharr("var",k,range,1,2);
170 // cout << "variable = " << var_name << "lo, hi = ",
171 // << range[0] << " " << range[1] << endl;
172 // }
173 // }
174 // -----------------------------------------------------------------
175 // ----------------------- END COMMENTS ---------------------------
176 // -----------------------------------------------------------------
