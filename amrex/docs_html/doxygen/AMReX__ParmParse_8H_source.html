22 //
23 // ParmParse class implements a simple database for the storage and
24 // retrieval of command-line and input-file arguments. The entries are
25 // stored in a static table in (name,value_list) pairs.
26 //
27 // The format of the input file is a series of DEFINITIONS.
28 //
29 // A DEFINITION is of the form <name> = <value> <value> ...
30 // The equal sign is important since the list of values can span multiple
31 // lines.
32 //
33 // Comments in an input file include all text from a '#' character to the
34 // end of the line. Here is an example input file:
35 //
36 // niter = 100 # niter is an integer
37 // title = "Double Wammy" # example of a string with spaces
38 // cell_size = 0.5 0.75 # cell spacing in each dimension
39 // plot.var = Density 1 10 # a list of values
40 // plot.var = Energy 5 12 # another list of values
41 // bigarray = 1 2 3 4 5 6 7 8 # first part of array
42 // 9 10 11 12 # continuation of bigarray
43 // test = apple "boy blue" 10 20 30 40
44 // FILE = prob_file # insert contents of this "prob_file" here
45 //
46 // The "FILE = <filename>" definition is special. Rather than just
47 // adding this entry to the database, it reads the contents of <filename>
48 // into the database.
49 //
50 // ParmParse stores all entries in a static table which is built the
51 // first time a ParmParse object is constructed (usually in main()).
52 // Subsequent invocations have access to this table.
53 // A ParmParse constructor has an optional "prefix" argument that will
54 // limit the searches to only those entries of the table with this prefix
55 // in name. For example:
56 // ParmParse pp("plot");
57 // will find only those entries with name given by "plot.<string>".
58 //
59 // All values in the table are stored as strings. For example, the
60 // values of "cell_size" in the above input file are stored as the
61 // strings "0.5" and "0.75". These strings can be returned as either
62 // strings or numeric values by the query functions.
63 // Character strings with spaces must be delimited by double quotes
64 // in the input file but the quotes are stripped before they are entered
65 // into the table. For example, 'title' in the above input file has a
66 // single value, the string 'Double Wammy' (without the quotes).
67 // Each value in the list associated with a definition can be referred to
68 // by its index number. The index numbers start at 0 just like an array
69 // in the C programming language. Consider the definition of "test" in
70 // the above input file. The first value 'apple'is a string with index
71 // 0. The second value 'boy blue' is a string with index 1. The
72 // remaining four values are integers indexed 2, 3, 4, and 5.
73 //
74 // For a string value to represent an integer or float it must fit the
75 // following regular expression:
76 // Sign ::= '+' | '-'
77 // Digit ::= '0' | '1' | ... | '9'
78 // Integer ::= [Sign]Digit+
79 // Exp ::= ('e'|'E')Integer
80 // Float ::= ( Integer[.Digit*][Exp] | [Integer].Digit+[Exp] )
81 //
82 // Where '+' indicates one or more occurrences, '*' represents zero or
83 // more occurrences, '|' means one or the other and '[]' represents zero
84 // or one occurrence.
85 //
86 // Note that floats and doubles have the same string representation and
87 // that the FORTRAN "double" exponent format is not supported.
88 // That is, 1.0d+3 is not a valid representation of a floating point
89 // number but that 1.0e+3 is acceptable.
90 //
91 // There are a host of functions allowing the user to query the database
92 // and retrieve values. Here are some general rules about the names of
93 // the member functions:
94 //
95 // * Functions with the string "get" in their names attempt to get a
96 // value or an array of values from the table. They generate a
97 // run-time error if they are not successful.
98 //
99 // * Functions with the string "query" in their names attempt to get a
100 // value or an array of values from the table. They return the value 1
101 // (true) if they are successful and 0 (false) if not.
102 //
103 // * Functions with the string "kth" in their names get values from
104 // the Kth entry with the given name in the database. This is
105 // necessary since there may be multiple definitions with the same
106 // name in the database.
107 //
108 // * Functions without the string "kth" in their names get values from
109 // the last entry with the given name in the database. Note that the
110 // definitions from the command line are appended to the database table
111 // and hence will be the last entries.
112 //
113 // * Functions with the string "arr" in their names get an Array of
114 // values from the given entry in the table. The array argument is
115 // resized (if necessary) to hold all the values requested.
116 //
117 // * Functions without the string "arr" in their names get single
118 // values from the given entry in the table.
119 //
120 // The following is a code sample showing how to use ParmParse:
121 //
122 // main(int argc, char **argv)
123 // {
124 // char* in_file_name = argv[1];
125 // ParmParse::Initialize(argc-2, argv+2, in_file_name);
126 //
127 // // Query table for value of "niter". If not in table
128 // // then set to default value
129 // if (!pp.query("niter",niter)) niter = 20;
130 //
131 // // read array of cell sizes if in table
132 // Vector<float> dx;
133 // if (nx=pp.countval("cell_size")) {
134 // // get nx values starting at index 0 and store in dx.
135 // // dx is automatically resized here.
136 // pp.getarr("cell_size",dx,0,nx);
137 // }
138 // ParmParse::Finalize();
139 // }
140 //
141 // void do_graphics()
142 // {
143 // //
144 // // Will only query entries with the "plot" prefix:
145 // //
146 // ParmParse pp("plot");
147 // //
148 // // Read all variables with "plot.var" keyword.
149 // //
150 // std::string var_name;
151 // Vector<int> range;
152 // int num = pp.countname("var");
153 // for (int k = 0; k < num; k++)
154 // {
155 // //
156 // // Element 0 in list is a string.
157 // //
158 // pp.getkth("var",k,var_name,0);
159 // //
160 // // Elements 1 and 2 are integers.
161 // // Note that "range" will be resized to hold 2 elements.
162 // //
163 // pp.getktharr("var",k,range,1,2);
164 // cout << "variable = " << var_name << "lo, hi = ",
165 // << range[0] << " " << range[1] << endl;
166 // }
167 // }
168 // -----------------------------------------------------------------
169 // ----------------------- END COMMENTS ---------------------------
170 // -----------------------------------------------------------------
