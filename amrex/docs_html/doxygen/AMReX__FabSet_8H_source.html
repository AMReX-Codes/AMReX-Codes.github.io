315 FabSetT<MF>&
316 FabSetT<MF>::linComb (value_type a, const MF& mfa, int a_comp,

318  int dcomp, int ncomp, int ngrow)
319 {
320  BL_PROFILE("FabSetT<MF>::linComb()");
321  BL_ASSERT(ngrow <= mfa.nGrow());
322  BL_ASSERT(ngrow <= mfb.nGrow());
323  BL_ASSERT(mfa.boxArray() == mfb.boxArray());
324  BL_ASSERT(boxArray() != mfa.boxArray());
325 
326  MF bdrya(boxArray(),DistributionMap(),ncomp,0,MFInfo());
327  MF bdryb(boxArray(),DistributionMap(),ncomp,0,MFInfo());
328 
329  const value_type huge = static_cast<value_type>(sizeof(value_type) == 8 ? 1.e200 : 1.e30);
330 
331 #ifdef AMREX_USE_OMP
332 #pragma omp parallel if (Gpu::notInLaunchRegion())
333 #endif
334  for (MFIter mfi(bdrya); mfi.isValid(); ++mfi) // tiling is not safe for this BoxArray
335  {
336  const Box& bx = mfi.validbox();
337  auto afab = bdrya.array(mfi);
338  auto bfab = bdryb.array(mfi);
339  AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
340  {
341  afab(i,j,k,n) = huge;
342  bfab(i,j,k,n) = huge;
343  });
344  }

347  bdryb.ParallelCopy(mfb,b_comp,0,ncomp,ngrow,0);

351 #endif
352  for (FabSetIter fsi(*this); fsi.isValid(); ++fsi)
353  {
354  const Box& bx = fsi.validbox();
355  auto const afab = bdrya.array(fsi);
356  auto const bfab = bdryb.array(fsi);
