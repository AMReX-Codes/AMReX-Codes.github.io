15inline Geometry shift_geom (Geometry const& geom)
16{
17 auto const& dom = geom.Domain();
18 auto const& dlo = dom.smallEnd();
19 if (dlo == 0) {
20 return geom;
21 } else {
22 return Geometry(amrex::shift(dom,-dlo), geom.ProbDomain(), geom.CoordInt(),
23 geom.isPeriodic());
24 }
25}

27template <typename MF>
28void shift_mfs (IntVect const& domain_lo, MF const& mf1, MF const& mf2,
29 MF& new_mf1, MF& new_mf2)
30{
31 AMREX_ALWAYS_ASSERT(mf1.boxArray() == mf2.boxArray() &&
32 mf1.DistributionMap() == mf2.DistributionMap());
33 BoxArray ba = mf1.boxArray();
34 ba.shift(-domain_lo);
35 new_mf1.define(ba, mf1.DistributionMap(), 1, mf1.nGrowVect(),
36 MFInfo().SetAlloc(false));
37 new_mf2.define(ba, mf2.DistributionMap(), 1, mf2.nGrowVect(),
38 MFInfo().SetAlloc(false));
39 for (MFIter mfi(mf1); mfi.isValid(); ++mfi) {
40 typename MF::fab_type fab1(mf1[mfi], amrex::make_alias, 0, 1);
41 fab1.shift(-domain_lo);
42 new_mf1.setFab(mfi, std::move(fab1));
43
44 typename MF::fab_type fab2(mf2[mfi], amrex::make_alias, 0, 1);
45 fab2.shift(-domain_lo);
46 new_mf2.setFab(mfi, std::move(fab2));
47 }
48}

219template <typename MF>
220void Poisson<MF>::solve (MF& a_soln, MF const& a_rhs)
221{
222 BL_PROFILE("FFT::Poisson::solve");
223
224 MF* soln = &a_soln;
225 MF const* rhs = &a_rhs;
226 MF solntmp, rhstmp;
227 if (m_domain_lo != 0) {
228 detail::shift_mfs(m_domain_lo, a_soln, a_rhs, solntmp, rhstmp);
229 soln = &solntmp;
230 rhs = &rhstmp;
231 }
232
233 AMREX_ASSERT(soln->is_cell_centered() && rhs->is_cell_centered());
234
235 using T = typename MF::value_type;
236
237 GpuArray<T,AMREX_SPACEDIM> fac
238 {AMREX_D_DECL(Math::pi<T>()/T(m_geom.Domain().length(0)),
239 Math::pi<T>()/T(m_geom.Domain().length(1)),
240 Math::pi<T>()/T(m_geom.Domain().length(2)))};
241 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
242 if (m_bc[idim].first == Boundary::periodic) {
243 fac[idim] *= T(2);
244 }
245 }
246 GpuArray<T,AMREX_SPACEDIM> dxfac
247 {AMREX_D_DECL(T(2)/T(m_geom.CellSize(0)*m_geom.CellSize(0)),
248 T(2)/T(m_geom.CellSize(1)*m_geom.CellSize(1)),
249 T(2)/T(m_geom.CellSize(2)*m_geom.CellSize(2)))};
250 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
251 if (m_geom.Domain().length(idim) == 1) {
252 dxfac[idim] = 0;
253 }
254 }
255 auto scale = (m_r2x) ? m_r2x->scalingFactor() : m_r2c->scalingFactor();
256
257 GpuArray<T,AMREX_SPACEDIM> offset{AMREX_D_DECL(T(0),T(0),T(0))};
258 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
259 if (m_bc[idim].first == Boundary::odd &&
260 m_bc[idim].second == Boundary::odd)
261 {
262 offset[idim] = T(1);
263 }
264 else if ((m_bc[idim].first == Boundary::odd &&
265 m_bc[idim].second == Boundary::even) ||
266 (m_bc[idim].first == Boundary::even &&
267 m_bc[idim].second == Boundary::odd))
268 {
269 offset[idim] = T(0.5);
270 }
271 }
272
273 auto f = [=] AMREX_GPU_DEVICE (int i, int j, int k, auto& spectral_data)
274 {
275 amrex::ignore_unused(j,k);
276 AMREX_D_TERM(T a = fac[0]*(i+offset[0]);,
277 T b = fac[1]*(j+offset[1]);,
278 T c = fac[2]*(k+offset[2]));
279 T k2 = AMREX_D_TERM(dxfac[0]*(std::cos(a)-T(1)),
280 +dxfac[1]*(std::cos(b)-T(1)),
281 +dxfac[2]*(std::cos(c)-T(1)));
282 if (k2 != T(0)) {
283 spectral_data /= k2;
284 }
285 spectral_data *= scale;
286 };
287
288 IntVect const& ng = amrex::elemwiseMin(soln->nGrowVect(), IntVect(1));
289
290 if (m_r2x) {
291 m_r2x->forwardThenBackward_doit_0(*rhs, *soln, f, ng, m_geom.periodicity());
292 detail::fill_physbc(*soln, m_geom, m_bc);
293 } else {
294 m_r2c->forward(*rhs);
295 m_r2c->post_forward_doit_0(f);
296 m_r2c->backward_doit(*soln, ng, m_geom.periodicity());
297 }
298}

314template <typename MF>
315void PoissonOpenBC<MF>::define_doit ()
316{
317 using T = typename MF::value_type;
318 auto const& lo = m_grown_domain.smallEnd();
319 auto const dx = T(m_geom.CellSize(0));
320 auto const dy = T(m_geom.CellSize(1));
321 auto const dz = T(m_geom.CellSize(2));
322 auto const gfac = T(1)/T(std::sqrt(T(12)));
323 // 0.125 comes from that there are 8 Gauss quadrature points
324 auto const fac = T(-0.125) * (dx*dy*dz) / (T(4)*Math::pi<T>());
325 m_solver.setGreensFunction([=] AMREX_GPU_DEVICE (int i, int j, int k) -> T
326 {
327 auto x = (T(i-lo[0]) - gfac) * dx; // first Gauss quadrature point
328 auto y = (T(j-lo[1]) - gfac) * dy;
329 auto z = (T(k-lo[2]) - gfac) * dz;
330 T r = 0;
331 for (int gx = 0; gx < 2; ++gx) {
332 for (int gy = 0; gy < 2; ++gy) {
333 for (int gz = 0; gz < 2; ++gz) {
334 auto xg = x + 2*gx*gfac*dx;
335 auto yg = y + 2*gy*gfac*dy;
336 auto zg = z + 2*gz*gfac*dz;
337 r += T(1)/std::sqrt(xg*xg+yg*yg+zg*zg);
338 }}}
339 return fac * r;
340 });
341}

353 template <typename T>
354 struct Tri_Uniform {
355 [[nodiscard]] AMREX_GPU_DEVICE AMREX_FORCE_INLINE
356 T operator() (int, int, int) const
357 {
358 return m_dz2inv;
359 }
360 T m_dz2inv;
361 };
362
363 template <typename T>
364 struct TriA {
365 [[nodiscard]] AMREX_GPU_DEVICE AMREX_FORCE_INLINE
366 T operator() (int, int, int k) const
367 {
368 return (k > 0) ? T(2.0) / (m_dz[k]*(m_dz[k]+m_dz[k-1]))
369 : T(1.0) / (m_dz[k]* m_dz[k]);
370 }
371 T const* m_dz;
372 };
373
374 template <typename T>
375 struct TriC {
376 [[nodiscard]] AMREX_GPU_DEVICE AMREX_FORCE_INLINE
377 T operator() (int, int, int k) const
378 {
379 return (k < m_size-1) ? T(2.0) / (m_dz[k]*(m_dz[k]+m_dz[k+1]))
380 : T(1.0) / (m_dz[k]* m_dz[k]);
381 }
382 T const* m_dz;
383 int m_size;
384 };
385}

398template <typename MF>
399void PoissonHybrid<MF>::build_spmf ()
400{
401#if (AMREX_SPACEDIM == 3)
402 AMREX_ALWAYS_ASSERT(m_geom.Domain().length(2) > 1 &&
403 (m_geom.Domain().length(0) > 1 ||
404 m_geom.Domain().length(1) > 1));
405
406 if (m_r2c) {
407 Box cdomain = m_geom.Domain();
408 if (cdomain.length(0) > 1) {
409 cdomain.setBig(0,cdomain.length(0)/2);
410 } else {
411 cdomain.setBig(1,cdomain.length(1)/2);
412 }
413 auto cba = amrex::decompose(cdomain, ParallelContext::NProcsSub(),
414 {AMREX_D_DECL(true,true,false)});
415 DistributionMapping dm = detail::make_iota_distromap(cba.size());
416 m_spmf_c.define(cba, dm, 1, 0);
417 } else if (m_geom.Domain().length(0) > 1 &&
418 m_geom.Domain().length(1) > 1) {
419 if (m_r2x->m_cy.empty()) { // spectral data is real
420 auto sba = amrex::decompose(m_geom.Domain(),ParallelContext::NProcsSub(),
421 {AMREX_D_DECL(true,true,false)});
422 DistributionMapping dm = detail::make_iota_distromap(sba.size());
423 m_spmf_r.define(sba, dm, 1, 0);
424 } else { // spectral data is complex. one of the first two dimensions is periodic.
425 Box cdomain = m_geom.Domain();
426 if (m_bc[0].first == Boundary::periodic) {
427 cdomain.setBig(0,cdomain.length(0)/2);
428 } else {
429 cdomain.setBig(1,cdomain.length(1)/2);
430 }
431 auto cba = amrex::decompose(cdomain, ParallelContext::NProcsSub(),
432 {AMREX_D_DECL(true,true,false)});
433 DistributionMapping dm = detail::make_iota_distromap(cba.size());
434 m_spmf_c.define(cba, dm, 1, 0);
435 }
436 } else {
437 // spectral data is real
438 auto sba = amrex::decompose(m_geom.Domain(),ParallelContext::NProcsSub(),
439 {AMREX_D_DECL(true,true,false)});
440 DistributionMapping dm = detail::make_iota_distromap(sba.size());
441 m_spmf_r.define(sba, dm, 1, 0);
442 }
443#else
444 amrex::ignore_unused(this);
445#endif

492#if (AMREX_SPACEDIM < 3)

519 } else { // spectral data is complex.
520 m_r2x->forward(*rhs, m_spmf_c);
521 solve_z(m_spmf_c, tria, tric);
522 m_r2x->backward(m_spmf_c, *soln, ng, m_geom.periodicity());
523 }
524 }
525
526 detail::fill_physbc(*soln, m_geom, m_bc);
527#endif

530template <typename MF>
531template <typename FA, typename TRIA, typename TRIC>
532void PoissonHybrid<MF>::solve_z (FA& spmf, TRIA const& tria, TRIC const& tric)
533{
534 BL_PROFILE("PoissonHybrid::solve_z");
535
536#if (AMREX_SPACEDIM < 3)
537 amrex::ignore_unused(spmf, tria, tric);
538#else
539 auto facx = Math::pi<T>()/T(m_geom.Domain().length(0));
540 auto facy = Math::pi<T>()/T(m_geom.Domain().length(1));
541 if (m_bc[0].first == Boundary::periodic) { facx *= T(2); }
542 if (m_bc[1].first == Boundary::periodic) { facy *= T(2); }
543 auto dxfac = T(2)/T(m_geom.CellSize(0)*m_geom.CellSize(0));
544 auto dyfac = T(2)/T(m_geom.CellSize(1)*m_geom.CellSize(1));
545 auto scale = (m_r2x) ? m_r2x->scalingFactor() : m_r2c->scalingFactor();
546
547 if (m_geom.Domain().length(0) == 1) { dxfac = 0; }
548 if (m_geom.Domain().length(1) == 1) { dyfac = 0; }
549
550 GpuArray<T,AMREX_SPACEDIM-1> offset{T(0),T(0)};
551 for (int idim = 0; idim < AMREX_SPACEDIM-1; ++idim) {
552 if (m_geom.Domain().length(idim) > 1) {
553 if (m_bc[idim].first == Boundary::odd &&
554 m_bc[idim].second == Boundary::odd)
555 {
556 offset[idim] = T(1);
557 }
558 else if ((m_bc[idim].first == Boundary::odd &&
559 m_bc[idim].second == Boundary::even) ||
560 (m_bc[idim].first == Boundary::even &&
561 m_bc[idim].second == Boundary::odd))
562 {
563 offset[idim] = T(0.5);
564 }
565 }
566 }
567
568 bool zlo_neumann = m_bc[2].first == Boundary::even;
569 bool zhi_neumann = m_bc[2].second == Boundary::even;
570 bool is_singular = (offset[0] == T(0)) && (offset[1] == T(0))
571 && zlo_neumann && zhi_neumann;
572
573 auto nz = m_geom.Domain().length(2);
574
575 for (MFIter mfi(spmf); mfi.isValid(); ++mfi)
576 {
577 auto const& spectral = spmf.array(mfi);
578 auto const& box = mfi.validbox();
579 auto const& xybox = amrex::makeSlab(box, 2, 0);
580
581#ifdef AMREX_USE_GPU
582 // xxxxx TODO: We need to explore how to optimize this
583 // function. Maybe we can use cusparse. Maybe we should make
584 // z-direction to be the unit stride direction.
585
586 FArrayBox tridiag_workspace(box,4);
587 auto const& ald = tridiag_workspace.array(0);
588 auto const& bd = tridiag_workspace.array(1);
