188 template <typename MF>
189 void PoissonOpenBC<MF>::define_doit ()
190 {
191  using T = typename MF::value_type;
192  auto const& lo = m_grown_domain.smallEnd();
193  auto const dx = T(m_geom.CellSize(0));
194  auto const dy = T(m_geom.CellSize(1));
195  auto const dz = T(m_geom.CellSize(2));
196  auto const gfac = T(1)/T(std::sqrt(T(12)));
197  // 0.125 comes from that there are 8 Gauss quadrature points
198  auto const fac = T(-0.125) * (dx*dy*dz) / (T(4)*Math::pi<T>());
199  m_solver.setGreensFunction([=] AMREX_GPU_DEVICE (int i, int j, int k) -> T
200  {
201  auto x = (T(i-lo[0]) - gfac) * dx; // first Gauss quadrature point
202  auto y = (T(j-lo[1]) - gfac) * dy;
203  auto z = (T(k-lo[2]) - gfac) * dz;
204  T r = 0;
205  for (int gx = 0; gx < 2; ++gx) {
206  for (int gy = 0; gy < 2; ++gy) {
207  for (int gz = 0; gz < 2; ++gz) {
208  auto xg = x + 2*gx*gfac*dx;
209  auto yg = y + 2*gy*gfac*dy;
210  auto zg = z + 2*gz*gfac*dz;
211  r += T(1)/std::sqrt(xg*xg+yg*yg+zg*zg);
212  }}}
213  return fac * r;
214  });
215 }
