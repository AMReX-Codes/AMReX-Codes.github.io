119 template<std::size_t... idx>
120 constexpr auto make_oob_message_impl (std::index_sequence<idx...>) {
121 constexpr std::size_t N = sizeof...(idx);
122 constexpr char prefix[] = " (";
123 constexpr char middle[] = ") is out of bound (";
124 constexpr char suffix[] = ")\n";
125
126 constexpr std::size_t size =
127 sizeof(prefix) - 1 +
128 (2*N + (N-1)) +
129 sizeof(middle) - 1 +
130 (5*N + (N-1)) +
131 sizeof(suffix); // include null terminator
132
133 std::array<char, size> buf{};
134
135 std::size_t pos = 0;
136
137 // prefix
138 for (char c : prefix) {
139 if (c != '\0') {
140 buf[pos++] = c;
141 }
142 }
143 // %d,%d,...
144 ((buf[pos++] = '%', buf[pos++] = 'd', idx + 1 < N ? buf[pos++] = ',' : 0), ...);
145
146 // ") is out of bound ("
147 for (char c : middle) {
148 if (c != '\0') {
149 buf[pos++] = c;
150 }
151 }
152
153 // %d:%d,%d:%d,...
154 ((buf[pos++] = '%', buf[pos++] = 'd', buf[pos++] = ':', buf[pos++] = '%', buf[pos++] = 'd',
155 idx + 1 < N ? buf[pos++] = ',' : 0), ...);
156
157 // suffix
158 for (char c : suffix) {
159 buf[pos++] = c;
160 }
161 return buf;
162 }

652 template <int M, std::enable_if_t<(M == N)
653 || (last_dim_component && (M + 1 == N || M == AMREX_SPACEDIM)), int> = 0>
654 [[nodiscard]] AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
655 constexpr Long get_offset (IntVectND<M> const& iv) const noexcept
656 {
657 Long offset = iv[0] - begin[0];
658 // If M == N and we have a component at the end, we only loop up to N-1 for spatial.
659 // Otherwise, we loop up to M.
660 constexpr int idx = (last_dim_component && M == N) ? N - 1 : M;
661 constexpr_for<1, idx>([&](int d) {
662 offset += (iv[d] - begin[d]) * nstride[d-1];
663 });
664
665 // Handle the component offset if the input is the full N dimensions
666 // and the last dimension is a component.
667 if constexpr (last_dim_component && M == N) {
668 offset += iv[N-1] * nstride[N-2];
669 }
670 return offset;
671 }

709 ss << iv[d];

711 }
712 ss << ") is out of bound (";
