314template <typename MF>
315void
316MLCellLinOpT<MF>::BndryCondLoc::
317setLOBndryConds (const Geometry& geom, const Real* dx,
318 const Vector<Array<BCType,AMREX_SPACEDIM> >& lobc,
319 const Vector<Array<BCType,AMREX_SPACEDIM> >& hibc,
320 IntVect const& ratio, const RealVect& interior_bloc,
321 const Array<Real,AMREX_SPACEDIM>& domain_bloc_lo,
322 const Array<Real,AMREX_SPACEDIM>& domain_bloc_hi,
323 LinOpBCType crse_fine_bc_type)
324{
325 const Box& domain = geom.Domain();
326
327#ifdef AMREX_USE_OMP
328#pragma omp parallel
329#endif
330 for (MFIter mfi(bcloc); mfi.isValid(); ++mfi)
331 {
332 const Box& bx = mfi.validbox();
333 for (int icomp = 0; icomp < m_ncomp; ++icomp) {
334 RealTuple & bloc = bcloc[mfi][icomp];
335 BCTuple & bctag = bcond[mfi][icomp];
336 MLMGBndryT<MF>::setBoxBC(bloc, bctag, bx, domain,
337 lobc[icomp], hibc[icomp],
338 dx, ratio, interior_bloc,
339 domain_bloc_lo, domain_bloc_hi,
340 geom.isPeriodicArray(),
341 crse_fine_bc_type);
342 }
343 }
344
345 Gpu::PinnedVector<GpuArray<BCTL,2*AMREX_SPACEDIM> > hv;
346 hv.reserve(bctl_dv.size());
347 for (MFIter mfi(bctl); mfi.isValid(); ++mfi)
348 {
349 for (int icomp = 0; icomp < m_ncomp; ++icomp) {
350 GpuArray<BCTL,2*AMREX_SPACEDIM> tmp;
351 for (int m = 0; m < 2*AMREX_SPACEDIM; ++m) {
352 tmp[m].type = bcond[mfi][icomp][m];
353 tmp[m].location = bcloc[mfi][icomp][m];
354 }
355 hv.push_back(std::move(tmp));
356 }
357 }
358 Gpu::copyAsync(Gpu::hostToDevice, hv.begin(), hv.end(), bctl_dv.begin());
359 Gpu::streamSynchronize();
360}

527template <typename MF>
528void
529MLCellLinOpT<MF>::setLevelBC (int amrlev, const MF* a_levelbcdata, const MF* robinbc_a,
530 const MF* robinbc_b, const MF* robinbc_f)
531{
532 BL_PROFILE("MLCellLinOp::setLevelBC()");
533
534 AMREX_ALWAYS_ASSERT(amrlev >= 0 && amrlev < this->m_num_amr_levels);
535
536 const int ncomp = this->getNComp();
537
538 MF zero;
539 IntVect ng(1);
540 if (this->hasHiddenDimension()) { ng[this->hiddenDirection()] = 0; }
541 if (a_levelbcdata == nullptr) {
542 zero.define(this->m_grids[amrlev][0], this->m_dmap[amrlev][0], ncomp, ng);
543 zero.setVal(RT(0.0));
544 } else {
545 AMREX_ALWAYS_ASSERT(a_levelbcdata->nGrowVect().allGE(ng));
546 }
547 const MF& bcdata = (a_levelbcdata == nullptr) ? zero : *a_levelbcdata;
548
549 IntVect br_ref_ratio(-1);
550
551 if (amrlev == 0)
552 {
553 if (this->needsCoarseDataForBC())
554 {
555 // AMREX_ALWAYS_ASSERT(!this->hasHiddenDimension());
556 if (this->hasHiddenDimension()) {
557 int hidden_dir = this->hiddenDirection();
558 AMREX_ALWAYS_ASSERT(this->m_coarse_data_crse_ratio[hidden_dir] == 1);
559 }
560 br_ref_ratio = this->m_coarse_data_crse_ratio.allGT(0) ? this->m_coarse_data_crse_ratio : IntVect(2);
561 if (this->m_crse_sol_br[amrlev] == nullptr && br_ref_ratio.allGT(0))
562 {
563 const int in_rad = 0;
564 const int out_rad = 1;
565 const int extent_rad = 2;
566 const IntVect crse_ratio = br_ref_ratio;
567 BoxArray cba = this->m_grids[amrlev][0];
568 cba.coarsen(crse_ratio);
569 this->m_crse_sol_br[amrlev] = std::make_unique<BndryRegisterT<MF>>
570 (cba, this->m_dmap[amrlev][0], in_rad, out_rad, extent_rad, ncomp);
571 }
572 if (this->m_coarse_data_for_bc != nullptr) {
573 AMREX_ALWAYS_ASSERT(this->m_coarse_data_crse_ratio.allGT(0));
574 const Box& cbx = amrex::coarsen(this->m_geom[0][0].Domain(), this->m_coarse_data_crse_ratio);
575 this->m_crse_sol_br[amrlev]->copyFrom(*(this->m_coarse_data_for_bc), 0, 0, 0, ncomp,
576 this->m_geom[0][0].periodicity(cbx));
577 } else {
578 this->m_crse_sol_br[amrlev]->setVal(RT(0.0));
579 }
580 this->m_bndry_sol[amrlev]->setBndryValues(*(this->m_crse_sol_br[amrlev]), 0,
581 bcdata, 0, 0, ncomp, br_ref_ratio,
582 InterpBndryDataT<MF>::IBD_max_order_DEF,
583 this->m_interpbndry_halfwidth);
584 br_ref_ratio = this->m_coarse_data_crse_ratio;
585 }
586 else
587 {
588 this->m_bndry_sol[amrlev]->setPhysBndryValues(bcdata,0,0,ncomp);
589 br_ref_ratio = IntVect(1);
590 }
591 }
592 else
593 {
594 this->m_bndry_sol[amrlev]->setPhysBndryValues(bcdata,0,0,ncomp);
595 br_ref_ratio = IntVect(this->m_amr_ref_ratio[amrlev-1]);
596 }
597
598 auto crse_fine_bc_type = (amrlev == 0) ? this->m_coarse_fine_bc_type : LinOpBCType::Dirichlet;
599 this->m_bndry_sol[amrlev]->setLOBndryConds(this->m_lobc, this->m_hibc, br_ref_ratio,
600 this->m_coarse_bc_loc, crse_fine_bc_type);
601
602 const Real* dx = this->m_geom[amrlev][0].CellSize();
603 for (int mglev = 0; mglev < this->m_num_mg_levels[amrlev]; ++mglev)
604 {
605 this->m_bcondloc[amrlev][mglev]->setLOBndryConds(this->m_geom[amrlev][mglev], dx,
606 this->m_lobc, this->m_hibc,
607 br_ref_ratio, this->m_coarse_bc_loc,
608 this->m_domain_bloc_lo, this->m_domain_bloc_hi,
609 crse_fine_bc_type);
610 }
611
612 if (this->hasRobinBC()) {
613 AMREX_ASSERT(robinbc_a != nullptr && robinbc_b != nullptr && robinbc_f != nullptr);
614 this->m_robin_bcval[amrlev] = std::make_unique<MF>(this->m_grids[amrlev][0],
615 this->m_dmap[amrlev][0],
616 ncomp*3, 1);
617 const Box& domain = this->m_geom[amrlev][0].Domain();
618 MFItInfo mfi_info;
619 if (Gpu::notInLaunchRegion()) { mfi_info.SetDynamic(true); }
620#ifdef AMREX_USE_OMP
621#pragma omp parallel if (Gpu::notInLaunchRegion())
622#endif
623 for (MFIter mfi(*(this->m_robin_bcval[amrlev]), mfi_info); mfi.isValid(); ++mfi) {
624 Box const& vbx = mfi.validbox();
625 Array4<RT const> const& ra = robinbc_a->const_array(mfi);
626 Array4<RT const> const& rb = robinbc_b->const_array(mfi);
627 Array4<RT const> const& rf = robinbc_f->const_array(mfi);
628 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
629 const Box& blo = amrex::adjCellLo(vbx, idim);
630 const Box& bhi = amrex::adjCellHi(vbx, idim);
631 bool outside_domain_lo = !(domain.contains(blo));
632 bool outside_domain_hi = !(domain.contains(bhi));
633 if ((!outside_domain_lo) && (!outside_domain_hi)) { continue; }
634 for (int icomp = 0; icomp < ncomp; ++icomp) {
635 Array4<RT> const& rbc = (*(this->m_robin_bcval[amrlev]))[mfi].array(icomp*3);
636 if (this->m_lobc_orig[icomp][idim] == LinOpBCType::Robin && outside_domain_lo)
637 {
638 AMREX_HOST_DEVICE_PARALLEL_FOR_3D(blo, i, j, k,
639 {
640 rbc(i,j,k,0) = ra(i,j,k,icomp);
641 rbc(i,j,k,1) = rb(i,j,k,icomp);
642 rbc(i,j,k,2) = rf(i,j,k,icomp);
643 });
644 }
645 if (this->m_hibc_orig[icomp][idim] == LinOpBCType::Robin && outside_domain_hi)
646 {
647 AMREX_HOST_DEVICE_PARALLEL_FOR_3D(bhi, i, j, k,
648 {
649 rbc(i,j,k,0) = ra(i,j,k,icomp);
650 rbc(i,j,k,1) = rb(i,j,k,icomp);
651 rbc(i,j,k,2) = rf(i,j,k,icomp);
652 });
653 }
654 }
655 }

698template <typename MF>
699void
700MLCellLinOpT<MF>::applyBC (int amrlev, int mglev, MF& in, BCMode bc_mode, StateMode,
701 const MLMGBndryT<MF>* bndry, bool skip_fillboundary) const
702{
703 BL_PROFILE("MLCellLinOp::applyBC()");
704 // No coarsened boundary values, cannot apply inhomog at mglev>0.
705 BL_ASSERT(mglev == 0 || bc_mode == BCMode::Homogeneous);
706 BL_ASSERT(bndry != nullptr || bc_mode == BCMode::Homogeneous);
707
708 const int ncomp = this->getNComp();
709 const int cross = isCrossStencil();
710 const int tensorop = isTensorOp();
711 if (!skip_fillboundary) {
712 in.FillBoundary(0, ncomp, this->m_geom[amrlev][mglev].periodicity(), cross);
713 }
714
715 int flagbc = bc_mode == BCMode::Inhomogeneous;
716 const int imaxorder = this->maxorder;
717
718 const Real* dxinv = this->m_geom[amrlev][mglev].InvCellSize();
719 const RT dxi = static_cast<RT>(dxinv[0]);
720 const RT dyi = (AMREX_SPACEDIM >= 2) ? static_cast<RT>(dxinv[1]) : RT(1.0);
721 const RT dzi = (AMREX_SPACEDIM == 3) ? static_cast<RT>(dxinv[2]) : RT(1.0);
722
723 const auto& maskvals = m_maskvals[amrlev][mglev];
724 const auto& bcondloc = *m_bcondloc[amrlev][mglev];
725
726 FAB foofab(Box::TheUnitBox(),ncomp);
727 const auto& foo = foofab.const_array();
728
729 MFItInfo mfi_info;
730 if (Gpu::notInLaunchRegion()) { mfi_info.SetDynamic(true); }
731
732 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(cross || tensorop || Gpu::notInLaunchRegion(),
733 "non-cross stencil not support for gpu");
734
735 const int hidden_direction = this->hiddenDirection();
736
737#ifdef AMREX_USE_GPU
738 if ((cross || tensorop) && Gpu::inLaunchRegion())
739 {
740 Vector<MLMGABCTag<RT>> tags;
741 tags.reserve(in.local_size()*AMREX_SPACEDIM*ncomp);
742
743 for (MFIter mfi(in); mfi.isValid(); ++mfi) {
744 const Box& vbx = mfi.validbox();
745 const auto& iofab = in.array(mfi);
746 const auto & bdlv = bcondloc.bndryLocs(mfi);
747 const auto & bdcv = bcondloc.bndryConds(mfi);
748
749 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
750 if (idim != hidden_direction) {
751 const Orientation olo(idim,Orientation::low);
752 const Orientation ohi(idim,Orientation::high);
753 const auto& bvlo = (bndry != nullptr) ?
754 bndry->bndryValues(olo).const_array(mfi) : foo;
755 const auto& bvhi = (bndry != nullptr) ?
756 bndry->bndryValues(ohi).const_array(mfi) : foo;
757 for (int icomp = 0; icomp < ncomp; ++icomp) {
758 tags.emplace_back(MLMGABCTag<RT>{iofab, bvlo, bvhi,
759 maskvals[olo].const_array(mfi),
760 maskvals[ohi].const_array(mfi),
761 bdlv[icomp][olo], bdlv[icomp][ohi],
762 amrex::adjCell(vbx,olo),
763 bdcv[icomp][olo], bdcv[icomp][ohi],
764 vbx.length(idim), icomp, idim});
765 }
766 }
767 }
768 }
769
770 ParallelFor(tags,
771 [=] AMREX_GPU_DEVICE (int i, int j, int k, MLMGABCTag<RT> const& tag) noexcept
772 {
773 if (tag.dir == 0)
774 {
775 mllinop_apply_bc_x(0, i, j, k, tag.blen, tag.fab,
776 tag.mask_lo, tag.bctype_lo, tag.bcloc_lo, tag.bcval_lo,
777 imaxorder, dxi, flagbc, tag.comp);
778 mllinop_apply_bc_x(1, i+tag.blen+1, j, k, tag.blen, tag.fab,
779 tag.mask_hi, tag.bctype_hi, tag.bcloc_hi, tag.bcval_hi,
780 imaxorder, dxi, flagbc, tag.comp);
781 }
782#if (AMREX_SPACEDIM > 1)
783 else
784#if (AMREX_SPACEDIM > 2)
785 if (tag.dir == 1)
786#endif
787 {
788 mllinop_apply_bc_y(0, i, j, k, tag.blen, tag.fab,
789 tag.mask_lo, tag.bctype_lo, tag.bcloc_lo, tag.bcval_lo,
790 imaxorder, dyi, flagbc, tag.comp);
791 mllinop_apply_bc_y(1, i, j+tag.blen+1, k, tag.blen, tag.fab,
792 tag.mask_hi, tag.bctype_hi, tag.bcloc_hi, tag.bcval_hi,
793 imaxorder, dyi, flagbc, tag.comp);
794 }
795#if (AMREX_SPACEDIM > 2)
796 else {
797 mllinop_apply_bc_z(0, i, j, k, tag.blen, tag.fab,
798 tag.mask_lo, tag.bctype_lo, tag.bcloc_lo, tag.bcval_lo,
799 imaxorder, dzi, flagbc, tag.comp);
800 mllinop_apply_bc_z(1, i, j, k+tag.blen+1, tag.blen, tag.fab,
801 tag.mask_hi, tag.bctype_hi, tag.bcloc_hi, tag.bcval_hi,
802 imaxorder, dzi, flagbc, tag.comp);
803 }
804#endif
805#endif
806 });
807 } else
808#endif
809 if (cross || tensorop)
810 {
811#ifdef AMREX_USE_OMP
812#pragma omp parallel if (Gpu::notInLaunchRegion())
813#endif
814 for (MFIter mfi(in, mfi_info); mfi.isValid(); ++mfi)
815 {
816 const Box& vbx = mfi.validbox();
817 const auto& iofab = in.array(mfi);
818
819 const auto & bdlv = bcondloc.bndryLocs(mfi);
820 const auto & bdcv = bcondloc.bndryConds(mfi);
821
822 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
823 {
824 if (hidden_direction == idim) { continue; }
825 const Orientation olo(idim,Orientation::low);
826 const Orientation ohi(idim,Orientation::high);
827 const Box blo = amrex::adjCellLo(vbx, idim);
828 const Box bhi = amrex::adjCellHi(vbx, idim);
829 const int blen = vbx.length(idim);
830 const auto& mlo = maskvals[olo].array(mfi);
831 const auto& mhi = maskvals[ohi].array(mfi);
832 const auto& bvlo = (bndry != nullptr) ? bndry->bndryValues(olo).const_array(mfi) : foo;
833 const auto& bvhi = (bndry != nullptr) ? bndry->bndryValues(ohi).const_array(mfi) : foo;
834 for (int icomp = 0; icomp < ncomp; ++icomp) {
835 const BoundCond bctlo = bdcv[icomp][olo];
836 const BoundCond bcthi = bdcv[icomp][ohi];
837 const RT bcllo = bdlv[icomp][olo];
838 const RT bclhi = bdlv[icomp][ohi];
839 if (idim == 0) {
840 mllinop_apply_bc_x(0, blo, blen, iofab, mlo,
841 bctlo, bcllo, bvlo,
842 imaxorder, dxi, flagbc, icomp);
843 mllinop_apply_bc_x(1, bhi, blen, iofab, mhi,
844 bcthi, bclhi, bvhi,
845 imaxorder, dxi, flagbc, icomp);
846 } else if (idim == 1) {
847 mllinop_apply_bc_y(0, blo, blen, iofab, mlo,
848 bctlo, bcllo, bvlo,
849 imaxorder, dyi, flagbc, icomp);
850 mllinop_apply_bc_y(1, bhi, blen, iofab, mhi,
851 bcthi, bclhi, bvhi,
852 imaxorder, dyi, flagbc, icomp);
853 } else {
854 mllinop_apply_bc_z(0, blo, blen, iofab, mlo,
855 bctlo, bcllo, bvlo,
856 imaxorder, dzi, flagbc, icomp);
857 mllinop_apply_bc_z(1, bhi, blen, iofab, mhi,
858 bcthi, bclhi, bvhi,
859 imaxorder, dzi, flagbc, icomp);
860 }
861 }
862 }
863 }
864 }
865 else
866 {
867#ifdef BL_NO_FORT
868 amrex::Abort("amrex_mllinop_apply_bc not available when BL_NO_FORT=TRUE");
869#else
870 if constexpr (std::is_same_v<Real,RT>) {
871#ifdef AMREX_USE_OMP
872#pragma omp parallel
873#endif
874 for (MFIter mfi(in, mfi_info); mfi.isValid(); ++mfi)
875 {
876 const Box& vbx = mfi.validbox();
877
878 const auto & bdlv = bcondloc.bndryLocs(mfi);
879 const auto & bdcv = bcondloc.bndryConds(mfi);
880
881 const RealTuple & bdl = bdlv[0];
882 const BCTuple & bdc = bdcv[0];
883
884 for (OrientationIter oitr; oitr; ++oitr)
885 {
886 const Orientation ori = oitr();
887
888 int cdr = ori;
889 RT bcl = bdl[ori];
890 int bct = bdc[ori];
891
892 const auto& fsfab = (bndry != nullptr) ? bndry->bndryValues(ori)[mfi] : foofab;
893
894 const Mask& m = maskvals[ori][mfi];
895
896 amrex_mllinop_apply_bc(BL_TO_FORTRAN_BOX(vbx),
897 BL_TO_FORTRAN_ANYD(in[mfi]),
898 BL_TO_FORTRAN_ANYD(m),
899 cdr, bct, bcl,
900 BL_TO_FORTRAN_ANYD(fsfab),
901 imaxorder, dxinv, flagbc, ncomp, cross);
902 }
903 }
904 } else {
905 amrex::Abort("Not supported");
906 }
907#endif
908 }

911template <typename MF>
912BoxArray
913MLCellLinOpT<MF>::makeNGrids (int grid_size) const
914{
915 const Box& dombx = this->m_geom[0].back().Domain();
916
917 const BoxArray& old_ba = this->m_grids[0].back();
918 const int N = old_ba.size();
919 Vector<Box> bv;
920 bv.reserve(N);
921 for (int i = 0; i < N; ++i)
922 {
923 Box b = old_ba[i];
924 b.coarsen(grid_size);
925 b.refine(grid_size);
926 IntVect sz = b.size();
927 const IntVect nblks {AMREX_D_DECL(sz[0]/grid_size, sz[1]/grid_size, sz[2]/grid_size)};
928
929 IntVect big = b.smallEnd() + grid_size - 1;
930 b.setBig(big);
931
932#if (AMREX_SPACEDIM == 3)
933 for (int kk = 0; kk < nblks[2]; ++kk) {
934#endif
935#if (AMREX_SPACEDIM >= 2)
936 for (int jj = 0; jj < nblks[1]; ++jj) {
937#endif
938 for (int ii = 0; ii < nblks[0]; ++ii)
939 {
940 IntVect shft{AMREX_D_DECL(ii*grid_size,jj*grid_size,kk*grid_size)};
941 Box bb = amrex::shift(b,shft);
942 bb &= dombx;
943 bv.push_back(bb);
944 }
945#if (AMREX_SPACEDIM >= 2)
946 }
947#endif
948#if (AMREX_SPACEDIM == 3)
949 }
950#endif
951 }
952
953 std::sort(bv.begin(), bv.end());
954 bv.erase(std::unique(bv.begin(), bv.end()), bv.end());
955
956 BoxList bl(std::move(bv));
957
958 return BoxArray{std::move(bl)};
959}

1017template <typename MF>
1018void
1019MLCellLinOpT<MF>::interpAssign (int amrlev, int fmglev, MF& fine, MF& crse) const
1020{
1021 const int ncomp = this->getNComp();
1022
1023 const Geometry& crse_geom = this->Geom(amrlev,fmglev+1);
1024 const IntVect refratio = (amrlev > 0) ? IntVect(2) : this->mg_coarsen_ratio_vec[fmglev];
1025 const IntVect ng = crse.nGrowVect();
1026
1027 MF cfine;
1028 const MF* cmf;
1029
1030 if (amrex::isMFIterSafe(crse, fine))
1031 {
1032 crse.FillBoundary(crse_geom.periodicity());
1033 cmf = &crse;
1034 }
1035 else
1036 {
1037 BoxArray cba = fine.boxArray();
1038 cba.coarsen(refratio);
1039 cfine.define(cba, fine.DistributionMap(), ncomp, ng);
1040 cfine.setVal(RT(0.0));
1041 cfine.ParallelCopy(crse, 0, 0, ncomp, IntVect(0), ng, crse_geom.periodicity());
1042 cmf = & cfine;
1043 }
1044
1045 bool isEB = fine.hasEBFabFactory();
1046 ignore_unused(isEB);
1047
1048#ifdef AMREX_USE_EB
1049 const auto *factory = dynamic_cast<EBFArrayBoxFactory const*>(&(fine.Factory()));
1050 const FabArray<EBCellFlagFab>* flags = (factory) ? &(factory->getMultiEBCellFlagFab()) : nullptr;
1051#endif
1052
1053 MFItInfo mfi_info;
1054 if (Gpu::notInLaunchRegion()) { mfi_info.EnableTiling().SetDynamic(true); }
1055#ifdef AMREX_USE_OMP
1056#pragma omp parallel if (Gpu::notInLaunchRegion())
1057#endif
1058 for (MFIter mfi(fine, mfi_info); mfi.isValid(); ++mfi)
1059 {
1060 const Box& bx = mfi.tilebox();
1061 const auto& ff = fine.array(mfi);
1062 const auto& cc = cmf->array(mfi);
1063#ifdef AMREX_USE_EB
1064 bool call_lincc;
1065 if (isEB)
1066 {
1067 const auto& flag = (*flags)[mfi];
1068 if (flag.getType(amrex::grow(bx,1)) == FabType::regular) {
1069 call_lincc = true;
1070 } else {
1071 Array4<EBCellFlag const> const& flg = flag.const_array();
1072 AMREX_GPU_LAUNCH_HOST_DEVICE_LAMBDA_RANGE (bx, tbx,
1073 {
1074 mlmg_eb_cc_interp_r<2>(tbx, ff, cc, flg, ncomp);
1075 });
1076
1077 call_lincc = false;
1078 }
1079 }
1080 else
1081 {
1082 call_lincc = true;
1083 }
1084#else
1085 const bool call_lincc = true;
1086#endif
1087 if (call_lincc)
1088 {
1089#if (AMREX_SPACEDIM == 3)
1090 if (this->hasHiddenDimension()) {
