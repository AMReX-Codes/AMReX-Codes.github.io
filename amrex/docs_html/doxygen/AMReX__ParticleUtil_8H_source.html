130  if (a_do_tiling == false) {

135  // This function must be consistent with FabArrayBase::buildTileArray function!!!
136  //
137  auto tiling_1d = [](int i, int lo, int hi, int tilesize,
138  int& ntile, int& tileidx, int& tlo, int& thi) {
139  int ncells = hi-lo+1;
140  ntile = amrex::max(ncells/tilesize, 1);
141  int ts_right = ncells/ntile;
142  int ts_left = ts_right+1;
143  int nleft = ncells - ntile*ts_right;
144  int ii = i - lo;
145  int nbndry = nleft*ts_left;
146  if (ii < nbndry) {
147  tileidx = ii / ts_left; // tiles on the left of nbndry have size of ts_left
148  tlo = lo + tileidx * ts_left;
149  thi = tlo + ts_left - 1;
150  } else {
151  tileidx = nleft + (ii-nbndry) / ts_right; // tiles on the right: ts_right
152  tlo = lo + tileidx * ts_right + nleft;
153  thi = tlo + ts_right - 1;
154  }
155  };
156  const IntVect& small = box.smallEnd();
157  const IntVect& big = box.bigEnd();
158  IntVect ntiles, ivIndex, tilelo, tilehi;
159 
160  AMREX_D_TERM(int iv0 = amrex::min(amrex::max(iv[0], small[0]), big[0]);,
161  int iv1 = amrex::min(amrex::max(iv[1], small[1]), big[1]);,
162  int iv2 = amrex::min(amrex::max(iv[2], small[2]), big[2]););
163 
164  AMREX_D_TERM(tiling_1d(iv0, small[0], big[0], a_tile_size[0], ntiles[0], ivIndex[0], tilelo[0], tilehi[0]);,

167

211 #else

213 #endif
214 
215  for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
216  {
217  if (not is_per[idim]) continue;
218  if (p.pos(idim) >= phi[idim]) {
219  while (p.pos(idim) >= phi[idim]) {
220  p.pos(idim) -= (phi[idim] - plo[idim]);
221  }
222  if (p.pos(idim) < plo[idim]) p.pos(idim) = plo[idim]; // clamp to avoid precision issues;
223  }
224  else if (p.pos(idim) < plo[idim]) {

226  p.pos(idim) += (phi[idim] - plo[idim]);
227  }
228  if (p.pos(idim) == phi[idim]) p.pos(idim) = plo[idim]; // clamp to avoid precision issues;
229  if (p.pos(idim) > phi[idim]) p.pos(idim) = phi[idim]-eps; // clamp to avoid precision issues;
230  }
