265 int getTileIndex (const IntVect& iv, const Box& box, const bool a_do_tiling,
266  const IntVect& a_tile_size, Box& tbx)
267 {
268  if (a_do_tiling == false) {
269  tbx = box;
270  return 0;
271  } else {
272  //
273  // This function must be consistent with FabArrayBase::buildTileArray function!!!
274  //
275  auto tiling_1d = [](int i, int lo, int hi, int tilesize,
276  int& ntile, int& tileidx, int& tlo, int& thi) {
277  int ncells = hi-lo+1;
278  ntile = amrex::max(ncells/tilesize, 1);
279  int ts_right = ncells/ntile;
280  int ts_left = ts_right+1;
281  int nleft = ncells - ntile*ts_right;
282  int ii = i - lo;
283  int nbndry = nleft*ts_left;
284  if (ii < nbndry) {
285  tileidx = ii / ts_left; // tiles on the left of nbndry have size of ts_left
286  tlo = lo + tileidx * ts_left;
287  thi = tlo + ts_left - 1;
288  } else {
289  tileidx = nleft + (ii-nbndry) / ts_right; // tiles on the right: ts_right
290  tlo = lo + tileidx * ts_right + nleft;
291  thi = tlo + ts_right - 1;
292  }
293  };
294  const IntVect& small = box.smallEnd();
295  const IntVect& big = box.bigEnd();
296  IntVect ntiles, ivIndex, tilelo, tilehi;
297 
298  AMREX_D_TERM(int iv0 = amrex::min(amrex::max(iv[0], small[0]), big[0]);,
299  int iv1 = amrex::min(amrex::max(iv[1], small[1]), big[1]);,
300  int iv2 = amrex::min(amrex::max(iv[2], small[2]), big[2]););
301 
302  AMREX_D_TERM(tiling_1d(iv0, small[0], big[0], a_tile_size[0], ntiles[0], ivIndex[0], tilelo[0], tilehi[0]);,
303  tiling_1d(iv1, small[1], big[1], a_tile_size[1], ntiles[1], ivIndex[1], tilelo[1], tilehi[1]);,
304  tiling_1d(iv2, small[2], big[2], a_tile_size[2], ntiles[2], ivIndex[2], tilelo[2], tilehi[2]););
305 
306  tbx = Box(tilelo, tilehi);
307 
308  return AMREX_D_TERM(ivIndex[0], + ntiles[0]*ivIndex[1], + ntiles[0]*ntiles[1]*ivIndex[2]);
309  }

384 template <typename P>
385 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
386 bool enforcePeriodic (P& p,
387  amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& plo,
388  amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& phi,
389  amrex::GpuArray<int,AMREX_SPACEDIM> const& is_per) noexcept
390 {
391  bool shifted = false;
392  for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
393  {
394  if (! is_per[idim]) continue;
395  if (p.pos(idim) >= phi[idim]) {
396  while (p.pos(idim) >= phi[idim]) {
397  p.pos(idim) -= static_cast<ParticleReal>(phi[idim] - plo[idim]);
398  }
399  // clamp to avoid precision issues;
400  if (p.pos(idim) < plo[idim]) p.pos(idim) = static_cast<ParticleReal>(plo[idim]);
401  shifted = true;
402  }
403  else if (p.pos(idim) < plo[idim]) {
404  while (p.pos(idim) < plo[idim]) {
405  p.pos(idim) += static_cast<ParticleReal>(phi[idim] - plo[idim]);
406  }
407  // clamp to avoid precision issues;
408  if (p.pos(idim) == phi[idim]) p.pos(idim) = static_cast<ParticleReal>(plo[idim]);
409  if (p.pos(idim) > phi[idim]) p.pos(idim) = std::nextafter( (amrex::ParticleReal) phi[idim], (amrex::ParticleReal) plo[idim]);
410  shifted = true;
411  }
412  AMREX_ASSERT( (p.pos(idim) >= plo[idim] ) && ( p.pos(idim) < phi[idim] ));
413  }
414 
415  return shifted;
416 }
