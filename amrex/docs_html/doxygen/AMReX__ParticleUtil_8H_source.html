395 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
396 int getTileIndex (const IntVect& iv, const Box& box, const bool a_do_tiling,
397  const IntVect& a_tile_size, Box& tbx)
398 {
399  if (a_do_tiling == false) {
400  tbx = box;
401  return 0;
402  } else {
403  //
404  // This function must be consistent with FabArrayBase::buildTileArray function!!!
405  //
406  auto tiling_1d = [](int i, int lo, int hi, int tilesize,
407  int& ntile, int& tileidx, int& tlo, int& thi) {
408  int ncells = hi-lo+1;
409  ntile = amrex::max(ncells/tilesize, 1);
410  int ts_right = ncells/ntile;
411  int ts_left = ts_right+1;
412  int nleft = ncells - ntile*ts_right;
413  int ii = i - lo;
414  int nbndry = nleft*ts_left;
415  if (ii < nbndry) {
416  tileidx = ii / ts_left; // tiles on the left of nbndry have size of ts_left
417  tlo = lo + tileidx * ts_left;
418  thi = tlo + ts_left - 1;
419  } else {
420  tileidx = nleft + (ii-nbndry) / ts_right; // tiles on the right: ts_right
421  tlo = lo + tileidx * ts_right + nleft;
422  thi = tlo + ts_right - 1;
423  }
424  };
425  const IntVect& small = box.smallEnd();
426  const IntVect& big = box.bigEnd();
427  IntVect ntiles, ivIndex, tilelo, tilehi;
428 
429  AMREX_D_TERM(int iv0 = amrex::min(amrex::max(iv[0], small[0]), big[0]);,
430  int iv1 = amrex::min(amrex::max(iv[1], small[1]), big[1]);,
431  int iv2 = amrex::min(amrex::max(iv[2], small[2]), big[2]););
432 
433  AMREX_D_TERM(tiling_1d(iv0, small[0], big[0], a_tile_size[0], ntiles[0], ivIndex[0], tilelo[0], tilehi[0]);,
434  tiling_1d(iv1, small[1], big[1], a_tile_size[1], ntiles[1], ivIndex[1], tilelo[1], tilehi[1]);,
435  tiling_1d(iv2, small[2], big[2], a_tile_size[2], ntiles[2], ivIndex[2], tilelo[2], tilehi[2]););
436 
437  tbx = Box(tilelo, tilehi);
438 
439  return AMREX_D_TERM(ivIndex[0], + ntiles[0]*ivIndex[1], + ntiles[0]*ntiles[1]*ivIndex[2]);
440  }

515 template <typename P>
516 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
517 bool enforcePeriodic (P& p,
518  amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& plo,
519  amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& phi,
520  amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& rhi,
521  amrex::GpuArray<int,AMREX_SPACEDIM> const& is_per) noexcept
522 {
523  bool shifted = false;
524  for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
525  {
526  if (! is_per[idim]) continue;
527  if (p.pos(idim) >= phi[idim]) {
528  while (p.pos(idim) >= phi[idim]) {
529  p.pos(idim) -= static_cast<ParticleReal>(phi[idim] - plo[idim]);
530  }
531  // clamp to avoid precision issues;
532  if (p.pos(idim) < plo[idim]) p.pos(idim) = static_cast<ParticleReal>(plo[idim]);
533  shifted = true;
534  }
535  else if (p.pos(idim) < plo[idim]) {
536  while (p.pos(idim) < plo[idim]) {
537  p.pos(idim) += static_cast<ParticleReal>(phi[idim] - plo[idim]);
538  }
539  // clamp to avoid precision issues;
540  if (p.pos(idim) >= rhi[idim]) {
541  p.pos(idim) = static_cast<ParticleReal>(plo[idim]);
542  }
543  shifted = true;
544  }
545  AMREX_ASSERT( (p.pos(idim) >= plo[idim] ) && ( p.pos(idim) < phi[idim] ));
546  }
547 
548  return shifted;
549 }
