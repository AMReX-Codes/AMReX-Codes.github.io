226 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
227 int getTileIndex (const IntVect& iv, const Box& box, const bool a_do_tiling,
228  const IntVect& a_tile_size, Box& tbx)
229 {
230  if (a_do_tiling == false) {
231  tbx = box;
232  return 0;
233  } else {
234  //
235  // This function must be consistent with FabArrayBase::buildTileArray function!!!
236  //
237  auto tiling_1d = [](int i, int lo, int hi, int tilesize,
238  int& ntile, int& tileidx, int& tlo, int& thi) {
239  int ncells = hi-lo+1;
240  ntile = amrex::max(ncells/tilesize, 1);
241  int ts_right = ncells/ntile;
242  int ts_left = ts_right+1;
243  int nleft = ncells - ntile*ts_right;
244  int ii = i - lo;
245  int nbndry = nleft*ts_left;
246  if (ii < nbndry) {
247  tileidx = ii / ts_left; // tiles on the left of nbndry have size of ts_left
248  tlo = lo + tileidx * ts_left;
249  thi = tlo + ts_left - 1;
250  } else {
251  tileidx = nleft + (ii-nbndry) / ts_right; // tiles on the right: ts_right
252  tlo = lo + tileidx * ts_right + nleft;
253  thi = tlo + ts_right - 1;
254  }
255  };
256  const IntVect& small = box.smallEnd();
257  const IntVect& big = box.bigEnd();
258  IntVect ntiles, ivIndex, tilelo, tilehi;
259 
260  AMREX_D_TERM(int iv0 = amrex::min(amrex::max(iv[0], small[0]), big[0]);,
261  int iv1 = amrex::min(amrex::max(iv[1], small[1]), big[1]);,
262  int iv2 = amrex::min(amrex::max(iv[2], small[2]), big[2]););
263 
264  AMREX_D_TERM(tiling_1d(iv0, small[0], big[0], a_tile_size[0], ntiles[0], ivIndex[0], tilelo[0], tilehi[0]);,
265  tiling_1d(iv1, small[1], big[1], a_tile_size[1], ntiles[1], ivIndex[1], tilelo[1], tilehi[1]);,
266  tiling_1d(iv2, small[2], big[2], a_tile_size[2], ntiles[2], ivIndex[2], tilelo[2], tilehi[2]););
267 
268  tbx = Box(tilelo, tilehi);
269 
270  return AMREX_D_TERM(ivIndex[0], + ntiles[0]*ivIndex[1], + ntiles[0]*ntiles[1]*ivIndex[2]);
271  }

407 bool enforcePeriodic (P& p,
408  amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& plo,
409  amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& phi,

413 {
414  bool shifted = false;

416  {
417  if (! is_per[idim]) { continue; }
418  if (p.pos(idim) > rhi[idim]) {

420  p.pos(idim) -= static_cast<ParticleReal>(phi[idim] - plo[idim]);

424  p.pos(idim) = rlo[idim];
425  }

427  }
428  else if (p.pos(idim) < rlo[idim]) {

432  // clamp to avoid precision issues;

434  p.pos(idim) = rhi[idim];

437  }

610 template <class index_type, typename F>
611 void PermutationForDeposition (Gpu::DeviceVector<index_type>& perm, index_type nitems,
612  index_type nbins, F&& f)
613 {
614  BL_PROFILE("PermutationForDeposition()");
615 
616  constexpr index_type gpu_block_size = 1024;
617  constexpr index_type gpu_block_size_m1 = gpu_block_size - 1;
618  constexpr index_type llist_guard = std::numeric_limits<index_type>::max();
619 
620  // round up to gpu_block_size
621  nbins = (nbins + gpu_block_size_m1) / gpu_block_size * gpu_block_size;
622 
623  Gpu::DeviceVector<index_type> llist_start(nbins, llist_guard);
624  Gpu::DeviceVector<index_type> llist_next(nitems);
625  perm.resize(nitems);
626  Gpu::DeviceScalar<index_type> global_idx(0);
627 
628  index_type* pllist_start = llist_start.dataPtr();
629  index_type* pllist_next = llist_next.dataPtr();
630  index_type* pperm = perm.dataPtr();
631  index_type* pglobal_idx = global_idx.dataPtr();
632 
633  amrex::ParallelFor(nitems, [=] AMREX_GPU_DEVICE (index_type i) noexcept
634  {
635  i = nitems - i - 1;
636  pllist_next[i] = Gpu::Atomic::Exch(pllist_start + f(i), i);
637  });
638 
639 #if defined(AMREX_USE_CUDA) || defined(AMREX_USE_HIP)
640  amrex::launch<gpu_block_size>(nbins / gpu_block_size, Gpu::gpuStream(),
641  [=] AMREX_GPU_DEVICE () {
642  __shared__ index_type sdata[gpu_block_size];
643  index_type current_idx = pllist_start[threadIdx.x + gpu_block_size * blockIdx.x];
644 
645  while (true) {
646  sdata[threadIdx.x] = index_type(current_idx != llist_guard);
647  index_type x = 0;
648 
649  // simple block wide prefix sum
650  for (index_type i = 1; i<gpu_block_size; i*=2) {
651  __syncthreads();
652  if (threadIdx.x >= i) {
653  x = sdata[threadIdx.x - i];
654  }
655  __syncthreads();
656  if (threadIdx.x >= i) {
657  sdata[threadIdx.x] += x;
658  }
659  }
660  __syncthreads();
661  if (sdata[gpu_block_size_m1] == 0) {
662  break;
663  }
664  __syncthreads();
665  if (threadIdx.x == gpu_block_size_m1) {
666  x = sdata[gpu_block_size_m1];
667  sdata[gpu_block_size_m1] = Gpu::Atomic::Add(pglobal_idx, x);
668  }
669  __syncthreads();
670  if (threadIdx.x < gpu_block_size_m1) {
671  sdata[threadIdx.x] += sdata[gpu_block_size_m1];
672  }
673  __syncthreads();
674  if (threadIdx.x == gpu_block_size_m1) {
675  sdata[gpu_block_size_m1] += x;
676  }
677  __syncthreads();
678 
679  if (current_idx != llist_guard) {
680  pperm[sdata[threadIdx.x] - 1] = current_idx;
681  current_idx = pllist_next[current_idx];
682  }
683  }
684  });
685 #else
686  Abort("Not implemented");
687 #endif
688 
689  Gpu::Device::streamSynchronize();
690 }

692 template <class index_type, class PTile>
693 void PermutationForDeposition (Gpu::DeviceVector<index_type>& perm, index_type nitems,
694  const PTile& ptile, Box bx, Geometry geom, const IntVect idx_type)
695 {
696  AMREX_ALWAYS_ASSERT(idx_type.allGE(IntVect(0)) && idx_type.allLE(IntVect(2)));
697 
698  const IntVect refine_vect = max(idx_type, IntVect(1)).min(IntVect(2));
699  const IntVect type_vect = idx_type - idx_type / 2 * 2;
700 
701  geom.refine(refine_vect);
702 
703  Box domain = geom.Domain();
704 
705  bx.convert(type_vect);
706  domain.convert(type_vect);
707 
708  const RealVect dxi(geom.InvCellSize());
709  const RealVect pos_offset = Real(0.5) * (RealVect(geom.ProbLo()) + RealVect(geom.ProbHi())
710  - RealVect(geom.CellSize()) * RealVect(domain.smallEnd() + domain.bigEnd()));
711 
712  const int ref_product = AMREX_D_TERM(refine_vect[0], * refine_vect[1], * refine_vect[2]);
713  const IntVect ref_offset(AMREX_D_DECL(1, refine_vect[0], refine_vect[0] * refine_vect[1]));
714 
715  auto ptd = ptile.getConstParticleTileData();
716  using ParticleType = typename PTile::ParticleType::ConstType;
717  PermutationForDeposition<index_type>(perm, nitems, bx.numPts() * ref_product,
718  [=] AMREX_GPU_DEVICE (index_type idx) noexcept
719  {
720  const auto& p = make_particle<ParticleType>{}(ptd,idx);
721 
722  IntVect iv = ((p.pos() - pos_offset) * dxi).round();
723 
724  IntVect iv_coarse = iv / refine_vect;
725  IntVect iv_remainder = iv - iv_coarse * refine_vect;
726 
727  iv_coarse = iv_coarse.max(bx.smallEnd());
728  iv_coarse = iv_coarse.min(bx.bigEnd());
729  return bx.index(iv_coarse) + bx.numPts() * (iv_remainder * ref_offset).sum();
730  });
731 }
